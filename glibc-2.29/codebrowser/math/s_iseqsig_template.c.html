19#include <errno.h>
20#include <fenv.h>
21#include <math.h>
22#include <stdbool.h>
23#include <fix-fp-int-compare-invalid.h>
24
25int
26M_DECL_FUNC (__iseqsig) (FLOAT x, FLOAT y)
27{
28 /* Comparing <= and >= is sufficient to determine both whether X and
29 Y are equal, and whether they are unordered, while raising the
30 "invalid" exception if they are unordered. */
31 bool cmp1 = x <= y;
32 bool cmp2 = x >= y;
33 if (cmp1 && cmp2)
34 return 1;
35 else if (!cmp1 && !cmp2)
36 {
37 if (FIX_COMPARE_INVALID)
38 __feraiseexcept (FE_INVALID);
39 __set_errno (EDOM);
40 }
41 return 0;
42}
