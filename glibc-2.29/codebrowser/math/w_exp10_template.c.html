19/* Only build wrappers from the templates for the types that define the macro
20 below. This macro is set in math-type-macros-<type>.h in sysdeps/generic
21 for each floating-point type. */
22#if __USE_WRAPPER_TEMPLATE
23
24# include <errno.h>
25# include <fenv.h>
26# include <math.h>
27# include <math_private.h>
28
29FLOAT
30M_DECL_FUNC (__exp10) (FLOAT x)
31{
32 FLOAT z = M_SUF (__ieee754_exp10) (x);
33 if (__glibc_unlikely (!isfinite (z) || z == 0) && isfinite (x))
34 /* Overflow or underflow. */
35 __set_errno (ERANGE);
36 return z;
37}
38declare_mgen_alias (__exp10, exp10)
39
40#endif /* __USE_WRAPPER_TEMPLATE. */
