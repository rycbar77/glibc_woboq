22FLOAT
23M_DECL_FUNC (__w_scalbln) (FLOAT x, long int n)
24{
25 if (!isfinite (x) || x == 0)
26 return x + x;
27
28 x = M_SUF (__scalbln) (x, n);
29
30 if (!isfinite (x) || x == 0)
31 __set_errno (ERANGE);
32
33 return x;
34}
