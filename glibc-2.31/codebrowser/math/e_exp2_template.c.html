19#include <math.h>
20#include <math_private.h>
21#include <math-underflow.h>
22#include <float.h>
23
24FLOAT
25M_DECL_FUNC (__ieee754_exp2) (FLOAT x)
26{
27 if (__glibc_likely (isless (x, (FLOAT) M_MAX_EXP)))
28 {
29 if (__builtin_expect (isgreaterequal (x, (FLOAT) (M_MIN_EXP - M_MANT_DIG
30 - 1)), 1))
31 {
32 int intx = (int) x;
33 FLOAT fractx = x - intx;
34 FLOAT result;
35 if (M_FABS (fractx) < M_EPSILON / 4)
36 result = M_SCALBN (1 + fractx, intx);
37 else
38 result = M_SCALBN (M_EXP (M_MLIT (M_LN2) * fractx), intx);
39 math_check_force_underflow_nonneg (result);
40 return result;
41 }
42 else
43 {
44 /* Underflow or exact zero. */
45 if (isinf (x))
46 return 0;
47 else
48 return M_MIN * M_MIN;
49 }
50 }
51 else
52 /* Infinity, NaN or overflow. */
53 return M_MAX * x;
54}
