21#include <math.h>

24CFLOAT
25M_DECL_FUNC (__cproj) (CFLOAT x)
26{
27 if (isinf (__real__ x) || isinf (__imag__ x))
28 {
29 CFLOAT res;
30
31 __real__ res = INFINITY;
32 __imag__ res = M_COPYSIGN (0, __imag__ x);
33
34 return res;
35 }
36
37 return x;
38}
