19#include <math.h>
20
21FLOAT
22M_DECL_FUNC (__fminmag) (FLOAT x, FLOAT y)
23{
24 FLOAT ax = M_FABS (x);
25 FLOAT ay = M_FABS (y);
26 if (isless (ax, ay))
27 return x;
28 else if (isgreater (ax, ay))
29 return y;
30 else if (ax == ay)
31 return x < y ? x : y;
32 else if (issignaling (x) || issignaling (y))
33 return x + y;
34 else
35 return isnan (y) ? x : y;
36}
37
38declare_mgen_alias (__fminmag, fminmag);
