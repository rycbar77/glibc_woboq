19#include <math.h>
20#include <errno.h>
21#include <limits.h>
22#include <math_private.h>
23#include <fenv.h>
24
25long int
26M_DECL_FUNC (__llogb) (FLOAT x)
27{
28 int r = M_SUF (__ieee754_ilogb) (x);
29 long int lr = r;
30 if (__glibc_unlikely (r == FP_ILOGB0)
31 || __glibc_unlikely (r == FP_ILOGBNAN)
32 || __glibc_unlikely (r == INT_MAX))
33 {
34#if LONG_MAX != INT_MAX
35 if (r == FP_ILOGB0)
36 lr = FP_LLOGB0;
37 else if (r == FP_ILOGBNAN)
38 lr = FP_LLOGBNAN;
39 else
40 lr = LONG_MAX;
41#endif
42 __set_errno (EDOM);
43 __feraiseexcept (FE_INVALID);
44 }
45 return lr;
46}
