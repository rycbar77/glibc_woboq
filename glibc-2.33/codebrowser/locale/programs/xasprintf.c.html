18#include <stdlib.h>
19#include <stdio.h>
20#include <stdarg.h>
21#include <libintl.h>
22#include <error.h>
23
24char *
25xasprintf (const char *format, ...)
26{
27 va_list ap;
28 va_start (ap, format);
29 char *result;
30 if (vasprintf (&result, format, ap) < 0)
31 error (EXIT_FAILURE, 0, _("memory exhausted"));
32 va_end (ap);
33 return result;
34}
