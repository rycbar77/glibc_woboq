21int
22M_DECL_FUNC (__canonicalize) (FLOAT *cx, const FLOAT *x)
23{
24 FLOAT val = *x;
25 /* For all binary formats supported by glibc, iscanonical only fails
26 if the representation is not a valid representation of the type,
27 so the only work to do is for signaling NaNs. */
28 if (!iscanonical (val))
29 return 1;
30 if (issignaling (val))
31 *cx = val + val;
32 else
33 *cx = val;
34 return 0;
35}
