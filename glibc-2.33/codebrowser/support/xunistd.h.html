19/* These wrapper functions use POSIX types and therefore cannot be
20 declared in <support/support.h>. */
21
22#ifndef SUPPORT_XUNISTD_H
23#define SUPPORT_XUNISTD_H
24
25#include <sys/cdefs.h>
26#include <sys/types.h>
27#include <unistd.h>
28
29__BEGIN_DECLS
30
31struct stat64;
32
33pid_t xfork (void);
34pid_t xwaitpid (pid_t, int *status, int flags);
35void xpipe (int[2]);
36void xdup2 (int, int);
37int xopen (const char *path, int flags, mode_t);
38void xstat (const char *path, struct stat64 *);
39void xlstat (const char *path, struct stat64 *);
40void xfstat (int fd, struct stat64 *);
41void xmkdir (const char *path, mode_t);
42void xchroot (const char *path);
43void xunlink (const char *path);
44long xsysconf (int name);
45long long xlseek (int fd, long long offset, int whence);
46void xftruncate (int fd, long long length);
47void xsymlink (const char *target, const char *linkpath);
48void xchdir (const char *path);
49void xfchmod (int fd, mode_t mode);
50void xchmod (const char *pathname, mode_t mode);
51
52/* Equivalent of "mkdir -p". */
53void xmkdirp (const char *, mode_t);
54
55/* Read the link at PATH. The caller should free the returned string
56 with free. */
57char *xreadlink (const char *path);
58
59/* Close the file descriptor. Ignore EINTR errors, but terminate the
60 process on other errors. */
61void xclose (int);
62
63/* Write the buffer. Retry on short writes. */
64void xwrite (int, const void *, size_t);
65
66/* Invoke mmap with a zero file offset. */
67void *xmmap (void *addr, size_t length, int prot, int flags, int fd);
68void xmprotect (void *addr, size_t length, int prot);
69void xmunmap (void *addr, size_t length);
70
71ssize_t xcopy_file_range(int fd_in, loff_t *off_in, int fd_out,
72 loff_t *off_out, size_t len, unsigned int flags);
73
74__END_DECLS
75
76#endif /* SUPPORT_XUNISTD_H */
