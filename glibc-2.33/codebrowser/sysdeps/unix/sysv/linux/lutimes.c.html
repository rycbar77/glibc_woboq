20#include <errno.h>
21#include <fcntl.h>
22#include <time.h>
23
24int
25__lutimes64 (const char *file, const struct __timeval64 tvp64[2])
26{
27 struct __timespec64 ts64[2];
28 if (tvp64 != NULL)
29 {
30 ts64[0] = timeval64_to_timespec64 (tvp64[0]);
31 ts64[1] = timeval64_to_timespec64 (tvp64[1]);
32 }
33
34 return __utimensat64_helper (AT_FDCWD, file, tvp64 ? &ts64[0] : NULL,
35 AT_SYMLINK_NOFOLLOW);
36}
37
38#if __TIMESIZE != 64
39libc_hidden_def (__lutimes64)

41int
42__lutimes (const char *file, const struct timeval tvp[2])
43{
44 struct __timeval64 tv64[2];
45
46 if (tvp != NULL)
47 {
48 tv64[0] = valid_timeval_to_timeval64 (tvp[0]);
49 tv64[1] = valid_timeval_to_timeval64 (tvp[1]);
50 }
51
52 return __lutimes64 (file, tvp ? &tv64[0] : NULL);
53}
54#endif
