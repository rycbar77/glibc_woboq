41static void
42update_usable (struct cpu_features *cpu_features)
43{
44 /* Copy the cpuid bits to usable bits for CPU featuress whose usability
45 in user space can be detected without additonal OS support. */
46 CPU_FEATURE_SET_USABLE (cpu_features, SSE3);
47 CPU_FEATURE_SET_USABLE (cpu_features, PCLMULQDQ);
48 CPU_FEATURE_SET_USABLE (cpu_features, SSSE3);
49 CPU_FEATURE_SET_USABLE (cpu_features, CMPXCHG16B);
50 CPU_FEATURE_SET_USABLE (cpu_features, SSE4_1);
51 CPU_FEATURE_SET_USABLE (cpu_features, SSE4_2);
52 CPU_FEATURE_SET_USABLE (cpu_features, MOVBE);
53 CPU_FEATURE_SET_USABLE (cpu_features, POPCNT);
54 CPU_FEATURE_SET_USABLE (cpu_features, AES);
55 CPU_FEATURE_SET_USABLE (cpu_features, OSXSAVE);
56 CPU_FEATURE_SET_USABLE (cpu_features, TSC);
57 CPU_FEATURE_SET_USABLE (cpu_features, CX8);
58 CPU_FEATURE_SET_USABLE (cpu_features, CMOV);
59 CPU_FEATURE_SET_USABLE (cpu_features, CLFSH);
60 CPU_FEATURE_SET_USABLE (cpu_features, MMX);
61 CPU_FEATURE_SET_USABLE (cpu_features, FXSR);
62 CPU_FEATURE_SET_USABLE (cpu_features, SSE);
63 CPU_FEATURE_SET_USABLE (cpu_features, SSE2);
64 CPU_FEATURE_SET_USABLE (cpu_features, HTT);
65 CPU_FEATURE_SET_USABLE (cpu_features, BMI1);
66 CPU_FEATURE_SET_USABLE (cpu_features, HLE);
67 CPU_FEATURE_SET_USABLE (cpu_features, BMI2);
68 CPU_FEATURE_SET_USABLE (cpu_features, ERMS);
69 CPU_FEATURE_SET_USABLE (cpu_features, RTM);
70 CPU_FEATURE_SET_USABLE (cpu_features, RDSEED);
71 CPU_FEATURE_SET_USABLE (cpu_features, ADX);
72 CPU_FEATURE_SET_USABLE (cpu_features, CLFLUSHOPT);
73 CPU_FEATURE_SET_USABLE (cpu_features, CLWB);
74 CPU_FEATURE_SET_USABLE (cpu_features, SHA);
75 CPU_FEATURE_SET_USABLE (cpu_features, PREFETCHWT1);
76 CPU_FEATURE_SET_USABLE (cpu_features, OSPKE);
77 CPU_FEATURE_SET_USABLE (cpu_features, WAITPKG);
78 CPU_FEATURE_SET_USABLE (cpu_features, SHSTK);
79 CPU_FEATURE_SET_USABLE (cpu_features, GFNI);
80 CPU_FEATURE_SET_USABLE (cpu_features, RDPID);
81 CPU_FEATURE_SET_USABLE (cpu_features, RDRAND);
82 CPU_FEATURE_SET_USABLE (cpu_features, CLDEMOTE);
83 CPU_FEATURE_SET_USABLE (cpu_features, MOVDIRI);
84 CPU_FEATURE_SET_USABLE (cpu_features, MOVDIR64B);
85 CPU_FEATURE_SET_USABLE (cpu_features, FSRM);
86 CPU_FEATURE_SET_USABLE (cpu_features, SERIALIZE);
87 CPU_FEATURE_SET_USABLE (cpu_features, TSXLDTRK);
88 CPU_FEATURE_SET_USABLE (cpu_features, IBT);
89 CPU_FEATURE_SET_USABLE (cpu_features, LAHF64_SAHF64);
90 CPU_FEATURE_SET_USABLE (cpu_features, LZCNT);
91 CPU_FEATURE_SET_USABLE (cpu_features, SSE4A);
92 CPU_FEATURE_SET_USABLE (cpu_features, PREFETCHW);
93 CPU_FEATURE_SET_USABLE (cpu_features, TBM);
94 CPU_FEATURE_SET_USABLE (cpu_features, RDTSCP);
95 CPU_FEATURE_SET_USABLE (cpu_features, WBNOINVD);
96 CPU_FEATURE_SET_USABLE (cpu_features, FZLRM);
97 CPU_FEATURE_SET_USABLE (cpu_features, FSRS);
98 CPU_FEATURE_SET_USABLE (cpu_features, FSRCS);
99
100 /* Can we call xgetbv? */
101 if (CPU_FEATURES_CPU_P (cpu_features, OSXSAVE))
102 {
103 unsigned int xcrlow;
104 unsigned int xcrhigh;
105 asm ("xgetbv" : "=a" (xcrlow), "=d" (xcrhigh) : "c" (0));
106 /* Is YMM and XMM state usable? */
107 if ((xcrlow & (bit_YMM_state | bit_XMM_state))
108 == (bit_YMM_state | bit_XMM_state))
109 {
110 /* Determine if AVX is usable. */
111 if (CPU_FEATURES_CPU_P (cpu_features, AVX))
112 {
113 CPU_FEATURE_SET (cpu_features, AVX);
114 /* The following features depend on AVX being usable. */
115 /* Determine if AVX2 is usable. */
116 if (CPU_FEATURES_CPU_P (cpu_features, AVX2))
117 {
118 CPU_FEATURE_SET (cpu_features, AVX2);
119
120 /* Unaligned load with 256-bit AVX registers are faster
121 on Intel/AMD processors with AVX2. */
122 cpu_features->preferred[index_arch_AVX_Fast_Unaligned_Load]
123 |= bit_arch_AVX_Fast_Unaligned_Load;
124 }
125 /* Determine if AVX-VNNI is usable. */
126 CPU_FEATURE_SET_USABLE (cpu_features, AVX_VNNI);
127 /* Determine if FMA is usable. */
128 CPU_FEATURE_SET_USABLE (cpu_features, FMA);
129 /* Determine if VAES is usable. */
130 CPU_FEATURE_SET_USABLE (cpu_features, VAES);
131 /* Determine if VPCLMULQDQ is usable. */
132 CPU_FEATURE_SET_USABLE (cpu_features, VPCLMULQDQ);
133 /* Determine if XOP is usable. */
134 CPU_FEATURE_SET_USABLE (cpu_features, XOP);
135 /* Determine if F16C is usable. */
136 CPU_FEATURE_SET_USABLE (cpu_features, F16C);
137 }
138
139 /* Check if OPMASK state, upper 256-bit of ZMM0-ZMM15 and
140 ZMM16-ZMM31 state are enabled. */
141 if ((xcrlow & (bit_Opmask_state | bit_ZMM0_15_state
142 | bit_ZMM16_31_state))
143 == (bit_Opmask_state | bit_ZMM0_15_state | bit_ZMM16_31_state))
144 {
145 /* Determine if AVX512F is usable. */
146 if (CPU_FEATURES_CPU_P (cpu_features, AVX512F))
147 {
148 CPU_FEATURE_SET (cpu_features, AVX512F);
149 /* Determine if AVX512CD is usable. */
150 CPU_FEATURE_SET_USABLE (cpu_features, AVX512CD);
151 /* Determine if AVX512ER is usable. */
152 CPU_FEATURE_SET_USABLE (cpu_features, AVX512ER);
153 /* Determine if AVX512PF is usable. */
154 CPU_FEATURE_SET_USABLE (cpu_features, AVX512PF);
155 /* Determine if AVX512VL is usable. */
156 CPU_FEATURE_SET_USABLE (cpu_features, AVX512VL);
157 /* Determine if AVX512DQ is usable. */
158 CPU_FEATURE_SET_USABLE (cpu_features, AVX512DQ);
159 /* Determine if AVX512BW is usable. */
160 CPU_FEATURE_SET_USABLE (cpu_features, AVX512BW);
161 /* Determine if AVX512_4FMAPS is usable. */
162 CPU_FEATURE_SET_USABLE (cpu_features, AVX512_4FMAPS);
163 /* Determine if AVX512_4VNNIW is usable. */
164 CPU_FEATURE_SET_USABLE (cpu_features, AVX512_4VNNIW);
165 /* Determine if AVX512_BITALG is usable. */
166 CPU_FEATURE_SET_USABLE (cpu_features, AVX512_BITALG);
167 /* Determine if AVX512_IFMA is usable. */
168 CPU_FEATURE_SET_USABLE (cpu_features, AVX512_IFMA);
169 /* Determine if AVX512_VBMI is usable. */
170 CPU_FEATURE_SET_USABLE (cpu_features, AVX512_VBMI);
171 /* Determine if AVX512_VBMI2 is usable. */
172 CPU_FEATURE_SET_USABLE (cpu_features, AVX512_VBMI2);
173 /* Determine if is AVX512_VNNI usable. */
174 CPU_FEATURE_SET_USABLE (cpu_features, AVX512_VNNI);
175 /* Determine if AVX512_VPOPCNTDQ is usable. */
176 CPU_FEATURE_SET_USABLE (cpu_features,
177 AVX512_VPOPCNTDQ);
178 /* Determine if AVX512_VP2INTERSECT is usable. */
179 CPU_FEATURE_SET_USABLE (cpu_features,
180 AVX512_VP2INTERSECT);
181 /* Determine if AVX512_BF16 is usable. */
182 CPU_FEATURE_SET_USABLE (cpu_features, AVX512_BF16);
183 /* Determine if AVX512_FP16 is usable. */
184 CPU_FEATURE_SET_USABLE (cpu_features, AVX512_FP16);
185 }
186 }
187 }
188
189 /* Are XTILECFG and XTILEDATA states usable? */
190 if ((xcrlow & (bit_XTILECFG_state | bit_XTILEDATA_state))
191 == (bit_XTILECFG_state | bit_XTILEDATA_state))
192 {
193 /* Determine if AMX_BF16 is usable. */
194 CPU_FEATURE_SET_USABLE (cpu_features, AMX_BF16);
195 /* Determine if AMX_TILE is usable. */
196 CPU_FEATURE_SET_USABLE (cpu_features, AMX_TILE);
197 /* Determine if AMX_INT8 is usable. */
198 CPU_FEATURE_SET_USABLE (cpu_features, AMX_INT8);
199 }
200
201 /* These features are usable only when OSXSAVE is enabled. */
202 CPU_FEATURE_SET (cpu_features, XSAVE);
203 CPU_FEATURE_SET_USABLE (cpu_features, XSAVEOPT);
204 CPU_FEATURE_SET_USABLE (cpu_features, XSAVEC);
205 CPU_FEATURE_SET_USABLE (cpu_features, XGETBV_ECX_1);
206 CPU_FEATURE_SET_USABLE (cpu_features, XFD);
207
208 /* For _dl_runtime_resolve, set xsave_state_size to xsave area
209 size + integer register save size and align it to 64 bytes. */
210 if (cpu_features->basic.max_cpuid >= 0xd)
211 {
212 unsigned int eax, ebx, ecx, edx;
213
214 __cpuid_count (0xd, 0, eax, ebx, ecx, edx);
215 if (ebx != 0)
216 {
217 unsigned int xsave_state_full_size
218 = ALIGN_UP (ebx + STATE_SAVE_OFFSET, 64);
219
220 cpu_features->xsave_state_size
221 = xsave_state_full_size;
222 cpu_features->xsave_state_full_size
223 = xsave_state_full_size;
224
225 /* Check if XSAVEC is available. */
226 if (CPU_FEATURES_CPU_P (cpu_features, XSAVEC))
227 {
228 unsigned int xstate_comp_offsets[32];
229 unsigned int xstate_comp_sizes[32];
230 unsigned int i;
231
232 xstate_comp_offsets[0] = 0;
233 xstate_comp_offsets[1] = 160;
234 xstate_comp_offsets[2] = 576;
235 xstate_comp_sizes[0] = 160;
236 xstate_comp_sizes[1] = 256;
237
238 for (i = 2; i < 32; i++)
239 {
240 if ((STATE_SAVE_MASK & (1 << i)) != 0)
241 {
242 __cpuid_count (0xd, i, eax, ebx, ecx, edx);
243 xstate_comp_sizes[i] = eax;
244 }
245 else
246 {
247 ecx = 0;
248 xstate_comp_sizes[i] = 0;
249 }
250
251 if (i > 2)
252 {
253 xstate_comp_offsets[i]
254 = (xstate_comp_offsets[i - 1]
255 + xstate_comp_sizes[i -1]);
256 if ((ecx & (1 << 1)) != 0)
257 xstate_comp_offsets[i]
258 = ALIGN_UP (xstate_comp_offsets[i], 64);
259 }
260 }
261
262 /* Use XSAVEC. */
263 unsigned int size
264 = xstate_comp_offsets[31] + xstate_comp_sizes[31];
265 if (size)
266 {
267 cpu_features->xsave_state_size
268 = ALIGN_UP (size + STATE_SAVE_OFFSET, 64);
269 CPU_FEATURE_SET (cpu_features, XSAVEC);
270 }
271 }
272 }
273 }
274 }
275
276 /* Determine if PKU is usable. */
277 if (CPU_FEATURES_CPU_P (cpu_features, OSPKE))
278 CPU_FEATURE_SET (cpu_features, PKU);
279
280 /* Determine if Key Locker instructions are usable. */
281 if (CPU_FEATURES_CPU_P (cpu_features, AESKLE))
282 {
283 CPU_FEATURE_SET (cpu_features, AESKLE);
284 CPU_FEATURE_SET_USABLE (cpu_features, KL);
285 CPU_FEATURE_SET_USABLE (cpu_features, WIDE_KL);
286 }
287
288 cpu_features->isa_1 = get_isa_level (cpu_features);
289}

291static void
292get_extended_indices (struct cpu_features *cpu_features)
293{
294 unsigned int eax, ebx, ecx, edx;
295 __cpuid (0x80000000, eax, ebx, ecx, edx);
296 if (eax >= 0x80000001)
297 __cpuid (0x80000001,
298 cpu_features->features[CPUID_INDEX_80000001].cpuid.eax,
299 cpu_features->features[CPUID_INDEX_80000001].cpuid.ebx,
300 cpu_features->features[CPUID_INDEX_80000001].cpuid.ecx,
301 cpu_features->features[CPUID_INDEX_80000001].cpuid.edx);
302 if (eax >= 0x80000007)
303 __cpuid (0x80000007,
304 cpu_features->features[CPUID_INDEX_80000007].cpuid.eax,
305 cpu_features->features[CPUID_INDEX_80000007].cpuid.ebx,
306 cpu_features->features[CPUID_INDEX_80000007].cpuid.ecx,
307 cpu_features->features[CPUID_INDEX_80000007].cpuid.edx);
308 if (eax >= 0x80000008)
309 __cpuid (0x80000008,
310 cpu_features->features[CPUID_INDEX_80000008].cpuid.eax,
311 cpu_features->features[CPUID_INDEX_80000008].cpuid.ebx,
312 cpu_features->features[CPUID_INDEX_80000008].cpuid.ecx,
313 cpu_features->features[CPUID_INDEX_80000008].cpuid.edx);
314}

316static void
317get_common_indices (struct cpu_features *cpu_features,
318 unsigned int *family, unsigned int *model,
319 unsigned int *extended_model, unsigned int *stepping)
320{
321 if (family)
322 {
323 unsigned int eax;
324 __cpuid (1, eax,
325 cpu_features->features[CPUID_INDEX_1].cpuid.ebx,
326 cpu_features->features[CPUID_INDEX_1].cpuid.ecx,
327 cpu_features->features[CPUID_INDEX_1].cpuid.edx);
328 cpu_features->features[CPUID_INDEX_1].cpuid.eax = eax;
329 *family = (eax >> 8) & 0x0f;
330 *model = (eax >> 4) & 0x0f;
331 *extended_model = (eax >> 12) & 0xf0;
332 *stepping = eax & 0x0f;
333 if (*family == 0x0f)
334 {
335 *family += (eax >> 20) & 0xff;
336 *model += *extended_model;
337 }
338 }
339
340 if (cpu_features->basic.max_cpuid >= 7)
341 {
342 __cpuid_count (7, 0,
343 cpu_features->features[CPUID_INDEX_7].cpuid.eax,
344 cpu_features->features[CPUID_INDEX_7].cpuid.ebx,
345 cpu_features->features[CPUID_INDEX_7].cpuid.ecx,
346 cpu_features->features[CPUID_INDEX_7].cpuid.edx);
347 __cpuid_count (7, 1,
348 cpu_features->features[CPUID_INDEX_7_ECX_1].cpuid.eax,
349 cpu_features->features[CPUID_INDEX_7_ECX_1].cpuid.ebx,
350 cpu_features->features[CPUID_INDEX_7_ECX_1].cpuid.ecx,
351 cpu_features->features[CPUID_INDEX_7_ECX_1].cpuid.edx);
352 }
353
354 if (cpu_features->basic.max_cpuid >= 0xd)
355 __cpuid_count (0xd, 1,
356 cpu_features->features[CPUID_INDEX_D_ECX_1].cpuid.eax,
357 cpu_features->features[CPUID_INDEX_D_ECX_1].cpuid.ebx,
358 cpu_features->features[CPUID_INDEX_D_ECX_1].cpuid.ecx,
359 cpu_features->features[CPUID_INDEX_D_ECX_1].cpuid.edx);
360
361 if (cpu_features->basic.max_cpuid >= 0x19)
362 __cpuid_count (0x19, 0,
363 cpu_features->features[CPUID_INDEX_19].cpuid.eax,
364 cpu_features->features[CPUID_INDEX_19].cpuid.ebx,
365 cpu_features->features[CPUID_INDEX_19].cpuid.ecx,
366 cpu_features->features[CPUID_INDEX_19].cpuid.edx);
367}

369_Static_assert (((index_arch_Fast_Unaligned_Load
370 == index_arch_Fast_Unaligned_Copy)
371 && (index_arch_Fast_Unaligned_Load
372 == index_arch_Prefer_PMINUB_for_stringop)
373 && (index_arch_Fast_Unaligned_Load
374 == index_arch_Slow_SSE4_2)
375 && (index_arch_Fast_Unaligned_Load
376 == index_arch_Fast_Rep_String)
377 && (index_arch_Fast_Unaligned_Load
378 == index_arch_Fast_Copy_Backward)),
379 "Incorrect index_arch_Fast_Unaligned_Load");

381static inline void
382init_cpu_features (struct cpu_features *cpu_features)
383{
384 unsigned int ebx, ecx, edx;
385 unsigned int family = 0;
386 unsigned int model = 0;
387 unsigned int stepping = 0;
388 enum cpu_features_kind kind;
389
390#if !HAS_CPUID
391 if (__get_cpuid_max (0, 0) == 0)
392 {
393 kind = arch_kind_other;
394 goto no_cpuid;
395 }
396#endif
397
398 __cpuid (0, cpu_features->basic.max_cpuid, ebx, ecx, edx);
399
400 /* This spells out "GenuineIntel". */
401 if (ebx == 0x756e6547 && ecx == 0x6c65746e && edx == 0x49656e69)
402 {
403 unsigned int extended_model;
404
405 kind = arch_kind_intel;
406
407 get_common_indices (cpu_features, &family, &model, &extended_model,
408 &stepping);
409
410 get_extended_indices (cpu_features);
411
412 update_usable (cpu_features);
413
414 if (family == 0x06)
415 {
416 model += extended_model;
417 switch (model)
418 {
419 case 0x1c:
420 case 0x26:
421 /* BSF is slow on Atom. */
422 cpu_features->preferred[index_arch_Slow_BSF]
423 |= bit_arch_Slow_BSF;
424 break;
425
426 case 0x57:
427 /* Knights Landing. Enable Silvermont optimizations. */
428
429 case 0x7a:
430 /* Unaligned load versions are faster than SSSE3
431 on Goldmont Plus. */
432
433 case 0x5c:
434 case 0x5f:
435 /* Unaligned load versions are faster than SSSE3
436 on Goldmont. */
437
438 case 0x4c:
439 case 0x5a:
440 case 0x75:
441 /* Airmont is a die shrink of Silvermont. */
442
443 case 0x37:
444 case 0x4a:
445 case 0x4d:
446 case 0x5d:
447 /* Unaligned load versions are faster than SSSE3
448 on Silvermont. */
449 cpu_features->preferred[index_arch_Fast_Unaligned_Load]
450 |= (bit_arch_Fast_Unaligned_Load
451 | bit_arch_Fast_Unaligned_Copy
452 | bit_arch_Prefer_PMINUB_for_stringop
453 | bit_arch_Slow_SSE4_2);
454 break;
455
456 case 0x86:
457 case 0x96:
458 case 0x9c:
459 /* Enable rep string instructions, unaligned load, unaligned
460 copy, pminub and avoid SSE 4.2 on Tremont. */
461 cpu_features->preferred[index_arch_Fast_Rep_String]
462 |= (bit_arch_Fast_Rep_String
463 | bit_arch_Fast_Unaligned_Load
464 | bit_arch_Fast_Unaligned_Copy
465 | bit_arch_Prefer_PMINUB_for_stringop
466 | bit_arch_Slow_SSE4_2);
467 break;
468
469 default:
470 /* Unknown family 0x06 processors. Assuming this is one
471 of Core i3/i5/i7 processors if AVX is available. */
472 if (!CPU_FEATURES_CPU_P (cpu_features, AVX))
473 break;
474 /* Fall through. */
475
476 case 0x1a:
477 case 0x1e:
478 case 0x1f:
479 case 0x25:
480 case 0x2c:
481 case 0x2e:
482 case 0x2f:
483 /* Rep string instructions, unaligned load, unaligned copy,
484 and pminub are fast on Intel Core i3, i5 and i7. */
485 cpu_features->preferred[index_arch_Fast_Rep_String]
486 |= (bit_arch_Fast_Rep_String
487 | bit_arch_Fast_Unaligned_Load
488 | bit_arch_Fast_Unaligned_Copy
489 | bit_arch_Prefer_PMINUB_for_stringop);
490 break;
491 }
492
493 /* Disable TSX on some Haswell processors to avoid TSX on kernels that
494 weren't updated with the latest microcode package (which disables
495 broken feature by default). */
496 switch (model)
497 {
498 case 0x3f:
499 /* Xeon E7 v3 with stepping >= 4 has working TSX. */
500 if (stepping >= 4)
501 break;
502 /* Fall through. */
503 case 0x3c:
504 case 0x45:
505 case 0x46:
506 /* Disable Intel TSX on Haswell processors (except Xeon E7 v3
507 with stepping >= 4) to avoid TSX on kernels that weren't
508 updated with the latest microcode package (which disables
509 broken feature by default). */
510 CPU_FEATURE_UNSET (cpu_features, RTM);
511 break;
512 }
513 }
514
515
516 /* Since AVX512ER is unique to Xeon Phi, set Prefer_No_VZEROUPPER
517 if AVX512ER is available. Don't use AVX512 to avoid lower CPU
518 frequency if AVX512ER isn't available. */
519 if (CPU_FEATURES_CPU_P (cpu_features, AVX512ER))
520 cpu_features->preferred[index_arch_Prefer_No_VZEROUPPER]
521 |= bit_arch_Prefer_No_VZEROUPPER;
522 else
523 cpu_features->preferred[index_arch_Prefer_No_AVX512]
524 |= bit_arch_Prefer_No_AVX512;
525 }
526 /* This spells out "AuthenticAMD" or "HygonGenuine". */
527 else if ((ebx == 0x68747541 && ecx == 0x444d4163 && edx == 0x69746e65)
528 || (ebx == 0x6f677948 && ecx == 0x656e6975 && edx == 0x6e65476e))
529 {
530 unsigned int extended_model;
531
532 kind = arch_kind_amd;
533
534 get_common_indices (cpu_features, &family, &model, &extended_model,
535 &stepping);
536
537 get_extended_indices (cpu_features);
538
539 update_usable (cpu_features);
540
541 ecx = cpu_features->features[CPUID_INDEX_1].cpuid.ecx;
542
543 if (CPU_FEATURE_USABLE_P (cpu_features, AVX))
544 {
545 /* Since the FMA4 bit is in CPUID_INDEX_80000001 and
546 FMA4 requires AVX, determine if FMA4 is usable here. */
547 CPU_FEATURE_SET_USABLE (cpu_features, FMA4);
548 }
549
550 if (family == 0x15)
551 {
552 /* "Excavator" */
553 if (model >= 0x60 && model <= 0x7f)
554 {
555 cpu_features->preferred[index_arch_Fast_Unaligned_Load]
556 |= (bit_arch_Fast_Unaligned_Load
557 | bit_arch_Fast_Copy_Backward);
558
559 /* Unaligned AVX loads are slower.*/
560 cpu_features->preferred[index_arch_AVX_Fast_Unaligned_Load]
561 &= ~bit_arch_AVX_Fast_Unaligned_Load;
562 }
563 }
564 }
565 /* This spells out "CentaurHauls" or " Shanghai ". */
566 else if ((ebx == 0x746e6543 && ecx == 0x736c7561 && edx == 0x48727561)
567 || (ebx == 0x68532020 && ecx == 0x20206961 && edx == 0x68676e61))
568 {
569 unsigned int extended_model, stepping;
570
571 kind = arch_kind_zhaoxin;
572
573 get_common_indices (cpu_features, &family, &model, &extended_model,
574 &stepping);
575
576 get_extended_indices (cpu_features);
577
578 update_usable (cpu_features);
579
580 model += extended_model;
581 if (family == 0x6)
582 {
583 if (model == 0xf || model == 0x19)
584 {
585 CPU_FEATURE_UNSET (cpu_features, AVX);
586 CPU_FEATURE_UNSET (cpu_features, AVX2);
587
588 cpu_features->preferred[index_arch_Slow_SSE4_2]
589 |= bit_arch_Slow_SSE4_2;
590
591 cpu_features->preferred[index_arch_AVX_Fast_Unaligned_Load]
592 &= ~bit_arch_AVX_Fast_Unaligned_Load;
593 }
594 }
595 else if (family == 0x7)
596 {
597 if (model == 0x1b)
598 {
599 CPU_FEATURE_UNSET (cpu_features, AVX);
600 CPU_FEATURE_UNSET (cpu_features, AVX2);
601
602 cpu_features->preferred[index_arch_Slow_SSE4_2]
603 |= bit_arch_Slow_SSE4_2;
604
605 cpu_features->preferred[index_arch_AVX_Fast_Unaligned_Load]
606 &= ~bit_arch_AVX_Fast_Unaligned_Load;
607 }
608 else if (model == 0x3b)
609 {
610 CPU_FEATURE_UNSET (cpu_features, AVX);
611 CPU_FEATURE_UNSET (cpu_features, AVX2);
612
613 cpu_features->preferred[index_arch_AVX_Fast_Unaligned_Load]
614 &= ~bit_arch_AVX_Fast_Unaligned_Load;
615 }
616 }
617 }
618 else
619 {
620 kind = arch_kind_other;
621 get_common_indices (cpu_features, NULL, NULL, NULL, NULL);
622 update_usable (cpu_features);
623 }
624
625 /* Support i586 if CX8 is available. */
626 if (CPU_FEATURES_CPU_P (cpu_features, CX8))
627 cpu_features->preferred[index_arch_I586] |= bit_arch_I586;
628
629 /* Support i686 if CMOV is available. */
630 if (CPU_FEATURES_CPU_P (cpu_features, CMOV))
631 cpu_features->preferred[index_arch_I686] |= bit_arch_I686;
632
633#if !HAS_CPUID
634no_cpuid:
635#endif
636
637 cpu_features->basic.kind = kind;
638 cpu_features->basic.family = family;
639 cpu_features->basic.model = model;
640 cpu_features->basic.stepping = stepping;
641
642 dl_init_cacheinfo (cpu_features);
643
644#if HAVE_TUNABLES
645 TUNABLE_GET (hwcaps, tunable_val_t *, TUNABLE_CALLBACK (set_hwcaps));
646#elif defined SHARED
647 /* Reuse dl_platform, dl_hwcap and dl_hwcap_mask for x86. The
648 glibc.cpu.hwcap_mask tunable is initialized already, so no
649 need to do this. */
650 GLRO(dl_hwcap_mask) = HWCAP_IMPORTANT;
651#endif
652
653#ifdef __x86_64__
654 GLRO(dl_hwcap) = HWCAP_X86_64;
655 if (cpu_features->basic.kind == arch_kind_intel)
656 {
657 const char *platform = NULL;
658
659 if (CPU_FEATURE_USABLE_P (cpu_features, AVX512CD))
660 {
661 if (CPU_FEATURE_USABLE_P (cpu_features, AVX512ER))
662 {
663 if (CPU_FEATURE_USABLE_P (cpu_features, AVX512PF))
664 platform = "xeon_phi";
665 }
666 else
667 {
668 if (CPU_FEATURE_USABLE_P (cpu_features, AVX512BW)
669 && CPU_FEATURE_USABLE_P (cpu_features, AVX512DQ)
670 && CPU_FEATURE_USABLE_P (cpu_features, AVX512VL))
671 GLRO(dl_hwcap) |= HWCAP_X86_AVX512_1;
672 }
673 }
674
675 if (platform == NULL
676 && CPU_FEATURE_USABLE_P (cpu_features, AVX2)
677 && CPU_FEATURE_USABLE_P (cpu_features, FMA)
678 && CPU_FEATURE_USABLE_P (cpu_features, BMI1)
679 && CPU_FEATURE_USABLE_P (cpu_features, BMI2)
680 && CPU_FEATURE_USABLE_P (cpu_features, LZCNT)
681 && CPU_FEATURE_USABLE_P (cpu_features, MOVBE)
682 && CPU_FEATURE_USABLE_P (cpu_features, POPCNT))
683 platform = "haswell";
684
685 if (platform != NULL)
686 GLRO(dl_platform) = platform;
687 }
688#else
689 GLRO(dl_hwcap) = 0;
690 if (CPU_FEATURE_USABLE_P (cpu_features, SSE2))
691 GLRO(dl_hwcap) |= HWCAP_X86_SSE2;
692
693 if (CPU_FEATURES_ARCH_P (cpu_features, I686))
694 GLRO(dl_platform) = "i686";
695 else if (CPU_FEATURES_ARCH_P (cpu_features, I586))
696 GLRO(dl_platform) = "i586";
697#endif
698
699#if CET_ENABLED
700# if HAVE_TUNABLES
701 TUNABLE_GET (x86_ibt, tunable_val_t *,
702 TUNABLE_CALLBACK (set_x86_ibt));
703 TUNABLE_GET (x86_shstk, tunable_val_t *,
704 TUNABLE_CALLBACK (set_x86_shstk));
705# endif
706
707 /* Check CET status. */
708 unsigned int cet_status = get_cet_status ();
709
710 if ((cet_status & GNU_PROPERTY_X86_FEATURE_1_IBT) == 0)
711 CPU_FEATURE_UNSET (cpu_features, IBT)
712 if ((cet_status & GNU_PROPERTY_X86_FEATURE_1_SHSTK) == 0)
713 CPU_FEATURE_UNSET (cpu_features, SHSTK)
714
715 if (cet_status)
716 {
717 GL(dl_x86_feature_1) = cet_status;
718
719# ifndef SHARED
720 /* Check if IBT and SHSTK are enabled by kernel. */
721 if ((cet_status & GNU_PROPERTY_X86_FEATURE_1_IBT)
722 || (cet_status & GNU_PROPERTY_X86_FEATURE_1_SHSTK))
723 {
724 /* Disable IBT and/or SHSTK if they are enabled by kernel, but
725 disabled by environment variable:
726
727 GLIBC_TUNABLES=glibc.cpu.hwcaps=-IBT,-SHSTK
728 */
729 unsigned int cet_feature = 0;
730 if (!CPU_FEATURE_USABLE (IBT))
731 cet_feature |= GNU_PROPERTY_X86_FEATURE_1_IBT;
732 if (!CPU_FEATURE_USABLE (SHSTK))
733 cet_feature |= GNU_PROPERTY_X86_FEATURE_1_SHSTK;
734
735 if (cet_feature)
736 {
737 int res = dl_cet_disable_cet (cet_feature);
738
739 /* Clear the disabled bits in dl_x86_feature_1. */
740 if (res == 0)
741 GL(dl_x86_feature_1) &= ~cet_feature;
742 }
743
744 /* Lock CET if IBT or SHSTK is enabled in executable. Don't
745 lock CET if IBT or SHSTK is enabled permissively. */
746 if (GL(dl_x86_feature_control).ibt != cet_permissive
747 && GL(dl_x86_feature_control).shstk != cet_permissive)
748 dl_cet_lock_cet ();
749 }
750# endif
751 }
752#endif
753
754#ifndef SHARED
755 /* NB: In libc.a, call init_cacheinfo. */
756 init_cacheinfo ();
757#endif
758}
