22/* Return true if the ISA level in ENTRY is compatible with CPU. */
23static inline bool
24dl_cache_hwcap_isa_level_compatible (struct file_entry_new *entry)
25{
26 const struct cpu_features *cpu_features = __get_cpu_features ();
27 unsigned int isa_level
28 = 1 << ((entry->hwcap >> 32) & DL_CACHE_HWCAP_ISA_LEVEL_MASK);
29
30 return (isa_level & cpu_features->isa_1) == isa_level;
31}
