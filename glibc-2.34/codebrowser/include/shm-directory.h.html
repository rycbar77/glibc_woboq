19#ifndef _SHM_DIRECTORY_H
20
21#include <limits.h>
22#include <paths.h>
23#include <stdbool.h>
24
25/* The directory that contains shared POSIX objects. */
26#define SHMDIR _PATH_DEV "shm/"
27
28struct shmdir_name
29{
30 /* The combined prefix/name. The sizeof includes the terminating
31 NUL byte. 4 bytes are needed for the optional "sem." prefix. */
32 char name[sizeof (SHMDIR) + 4 + NAME_MAX];
33};
34
35/* Sets RESULT->name to the constructed name and returns 0 on success,
36 or -1 on failure. Includes the "sem." prefix in the name if
37 SEM_PREFIX is true. */
38int __shm_get_name (struct shmdir_name *result, const char *name,
39 bool sem_prefix);
40libc_hidden_proto (__shm_get_name)
41
42#endif /* shm-directory.h */
