20#include <errno.h>
21#include <math.h>
22#include <math-narrow-eval.h>
23
24FLOAT
25M_DECL_FUNC (__fdim) (FLOAT x, FLOAT y)
26{
27 if (islessequal (x, y))
28 return 0;
29
30 FLOAT r = math_narrow_eval (x - y);
31 if (isinf (r) && !isinf (x) && !isinf (y))
32 __set_errno (ERANGE);
33
34 return r;
35}
36declare_mgen_alias (__fdim, fdim);
