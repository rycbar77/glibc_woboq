19#include <math.h>
20#include <errno.h>

22#include <math_private.h>
23#include <fenv.h>

25/* wrapper ilogb */
26int
27M_DECL_FUNC (__ilogb) (FLOAT x)
28{
29 int r = M_SUF (__ieee754_ilogb) (x);
30 if (__builtin_expect (r == FP_ILOGB0, 0)
31 || __builtin_expect (r == FP_ILOGBNAN, 0)
32 || __builtin_expect (r == INT_MAX, 0))
33 {
34 __set_errno (EDOM);
35 __feraiseexcept (FE_INVALID);
36 }
37 return r;
38}
