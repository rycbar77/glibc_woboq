19/* Only build wrappers from the templates for the types that define the macro
20 below. This macro is set in math-type-macros-<type>.h in sysdeps/generic
21 for each floating-point type. */
22#if __USE_WRAPPER_TEMPLATE
23
24# include <errno.h>
25# include <fenv.h>
26# include <math.h>
27# include <math_private.h>
28
29#define M_DECL_FUNC_R_X(x) x ## _r
30#define M_DECL_FUNC_R_S(x) M_DECL_FUNC_R_X (x)
31#define M_DECL_FUNC_R(x) M_DECL_FUNC_R_S (M_SUF (x))
32
33#define M_CALL_FUNC_R_X(x) x ## _r
34#define M_CALL_FUNC_R_S(x) M_CALL_FUNC_R_X (x)
35#define M_CALL_FUNC_R(x) M_CALL_FUNC_R_S (M_SUF (x))
36
37FLOAT
38M_DECL_FUNC_R (__lgamma) (FLOAT x, int *signgamp)
39{
40 FLOAT y = M_CALL_FUNC_R (__ieee754_lgamma) (x, signgamp);
41 if (__glibc_unlikely (!isfinite (y)) && isfinite (x))
42 /* Pole error: lgamma_r(integer x<0). Or overflow. */
43 __set_errno (ERANGE);
44 return y;
45}
46declare_mgen_alias_r (__lgamma, lgamma)
47
48#endif /* __USE_WRAPPER_TEMPLATE. */
