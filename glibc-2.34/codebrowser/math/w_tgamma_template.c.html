19/* Only build wrappers from the templates for the types that define the macro
20 below. This macro is set in math-type-macros-<type>.h in sysdeps/generic
21 for each floating-point type. */
22#if __USE_WRAPPER_TEMPLATE
23
24# include <errno.h>
25# include <fenv.h>
26# include <math.h>
27# include <math_private.h>
28
29#define M_CALL_FUNC_X(x) x ## _r
30#define M_CALL_FUNC(x) M_CALL_FUNC_X (x)
31
32FLOAT
33M_DECL_FUNC (__tgamma) (FLOAT x)
34{
35 int local_signgam;
36 FLOAT y = M_CALL_FUNC (M_SUF (__ieee754_gamma)) (x, &local_signgam);
37
38 if (__glibc_unlikely (!isfinite (y) || y == 0)
39 && (isfinite (x) || (isinf (x) && x < 0)))
40 {
41 if (x == 0)
42 /* Pole error: tgamma(x=0). */
43 __set_errno (ERANGE);
44 else if (M_SUF (floor) (x) == x && x < 0)
45 /* Domain error: tgamma(integer x<0). */
46 __set_errno (EDOM);
47 else
48 /* Overflow or underflow. */
49 __set_errno (ERANGE);
50 }
51 return local_signgam < 0 ? -y : y;
52}
53declare_mgen_alias (__tgamma, tgamma)
54
55#endif /* __USE_WRAPPER_TEMPLATE. */
