23/* See pthread_rwlock_common.c. */
24int
25___pthread_rwlock_clockwrlock64 (pthread_rwlock_t *rwlock, clockid_t clockid,
26 const struct __timespec64 *abstime)
27{
28 return __pthread_rwlock_wrlock_full64 (rwlock, clockid, abstime);
29}

37int
38___pthread_rwlock_clockwrlock (pthread_rwlock_t *rwlock, clockid_t clockid,
39 const struct timespec *abstime)
40{
41 struct __timespec64 ts64 = valid_timespec_to_timespec64 (*abstime);
42
43 return __pthread_rwlock_clockwrlock64 (rwlock, clockid, &ts64);
44}
