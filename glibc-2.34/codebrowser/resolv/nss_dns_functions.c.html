23void
24__nss_dns_functions (nss_module_functions_untyped pointers)
25{
26 struct nss_module_functions typed =
27 {
28 .getcanonname_r = &_nss_dns_getcanonname_r,
29 .gethostbyname3_r = &_nss_dns_gethostbyname3_r,
30 .gethostbyname2_r = &_nss_dns_gethostbyname2_r,
31 .gethostbyname_r = &_nss_dns_gethostbyname_r,
32 .gethostbyname4_r = &_nss_dns_gethostbyname4_r,
33 .gethostbyaddr2_r = &_nss_dns_gethostbyaddr2_r,
34 .gethostbyaddr_r = &_nss_dns_gethostbyaddr_r,
35 .getnetbyname_r = &_nss_dns_getnetbyname_r,
36 .getnetbyaddr_r = &_nss_dns_getnetbyaddr_r,
37 };
38
39 memcpy (pointers, &typed, sizeof (nss_module_functions_untyped));
40}
