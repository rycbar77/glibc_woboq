20#include <bits/wordsize.h>
21#if __WORDSIZE != 64
22
23# include <aio.h>
24# include <aio_misc.h>
25# include <shlib-compat.h>
26
27int
28__aio_read64 (struct aiocb64 *aiocbp)
29{
30 return (__aio_enqueue_request ((aiocb_union *) aiocbp, LIO_READ64) == NULL
31 ? -1 : 0);
32}

38# endif
39# else /* !PTHREAD_IN_LIBC */
40strong_alias (__aio_read64, aio_read64)
41# endif
