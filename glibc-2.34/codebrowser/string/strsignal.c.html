18#include <signal.h>
19#include <stdio.h>
20#include <stdlib.h>
21#include <string.h>
22#include <libintl.h>
23#include <tls-internal.h>
24
25/* Return a string describing the meaning of the signal number SIGNUM. */
26char *
27strsignal (int signum)
28{
29 const char *desc = __sigdescr_np (signum);
30 if (desc != NULL)
31 return _(desc);
32
33 struct tls_internal_t *tls_internal = __glibc_tls_internal ();
34 free (tls_internal->strsignal_buf);
35
36 int r;
37#ifdef SIGRTMIN
38 if (signum >= SIGRTMIN && signum <= SIGRTMAX)
39 r = __asprintf (&tls_internal->strsignal_buf, _("Real-time signal %d"),
40 signum - SIGRTMIN);
41 else
42#endif
43 r = __asprintf (&tls_internal->strsignal_buf, _("Unknown signal %d"),
44 signum);
45
46 if (r == -1)
47 tls_internal->strsignal_buf = NULL;
48
49 return tls_internal->strsignal_buf;
50}
