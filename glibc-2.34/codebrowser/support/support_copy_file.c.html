19#include <fcntl.h>
20#include <support/check.h>
21#include <support/support.h>
22#include <support/xunistd.h>
23
24void
25support_copy_file (const char *from, const char *to)
26{
27 struct stat64 st;
28 xstat (from, &st);
29 int fd_from = xopen (from, O_RDONLY, 0);
30 mode_t mode = st.st_mode & 0777;
31 int fd_to = xopen (to, O_WRONLY | O_TRUNC | O_CREAT, mode);
32 ssize_t ret = support_copy_file_range (fd_from, NULL, fd_to, NULL,
33 st.st_size, 0);
34 if (ret < 0)
35 FAIL_EXIT1 ("copying from \"%s\" to \"%s\": %m", from, to);
36 if (ret != st.st_size)
37 FAIL_EXIT1 ("copying from \"%s\" to \"%s\": only %zd of %llu bytes copied",
38 from, to, ret, (unsigned long long int) st.st_size);
39 if (fchmod (fd_to, mode) < 0)
40 FAIL_EXIT1 ("fchmod on %s to 0%o: %m", to, mode);
41 xclose (fd_to);
42 xclose (fd_from);
43}
