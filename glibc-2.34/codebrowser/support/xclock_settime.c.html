19#include <stdlib.h>
20#include <support/check.h>
21#include <support/xtime.h>
22
23void
24xclock_settime (clockid_t clockid,
25 const struct timespec *ts)
26{
27 if (getenv (SETTIME_ENV_NAME) == NULL)
28 FAIL_UNSUPPORTED ("clock_settime is executed only when "\
29 SETTIME_ENV_NAME" is set\n");
30
31 int ret = clock_settime (clockid, ts);
32 if (ret < 0)
33 FAIL_EXIT1 ("clock_settime (%d): %m", clockid);
34}
