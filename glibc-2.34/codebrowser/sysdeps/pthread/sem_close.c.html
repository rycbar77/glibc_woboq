19#include <errno.h>
20#include "semaphoreP.h"
21#include <sem_routines.h>
22
23int
24__sem_close (sem_t *sem)
25{
26 if (!__sem_remove_mapping (sem))
27 {
28 __set_errno (EINVAL);
29 return -1;
30 }
31
32 return 0;
33}

35versioned_symbol (libc, __sem_close, sem_close, GLIBC_2_34);
36# if OTHER_SHLIB_COMPAT (libpthread, GLIBC_2_1_1, GLIBC_2_34)
37compat_symbol (libpthread, __sem_close, sem_close, GLIBC_2_1_1);
38# endif
39#else /* !PTHREAD_IN_LIBC */
40strong_alias (__sem_close, sem_close)
41#endif
