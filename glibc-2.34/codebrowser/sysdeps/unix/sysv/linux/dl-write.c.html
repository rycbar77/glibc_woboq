21#include <ldsodefs.h>

23ssize_t
24_dl_write (int fd, const void *buffer, size_t length)
25{
26 long int r = INTERNAL_SYSCALL_CALL (write, fd, buffer, length);
27 if (INTERNAL_SYSCALL_ERROR_P (r))
28 r = - INTERNAL_SYSCALL_ERRNO (r);
29 return r;
30}
