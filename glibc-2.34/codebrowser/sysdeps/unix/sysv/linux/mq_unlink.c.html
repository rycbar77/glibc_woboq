18#include <errno.h>
19#include <mqueue.h>
20#include <sysdep.h>
21#include <shlib-compat.h>
22
23/* Remove message queue named NAME. */
24int
25__mq_unlink (const char *name)
26{
27 if (name[0] != '/')
28 return INLINE_SYSCALL_ERROR_RETURN_VALUE (EINVAL);
29
30 int ret = INTERNAL_SYSCALL_CALL (mq_unlink, name + 1);
31
32 /* While unlink can return either EPERM or EACCES, mq_unlink should
33 return just EACCES. */
34 if (__glibc_unlikely (INTERNAL_SYSCALL_ERROR_P (ret)))
35 {
36 ret = INTERNAL_SYSCALL_ERRNO (ret);
37 if (ret == EPERM)
38 ret = EACCES;
39 return INLINE_SYSCALL_ERROR_RETURN_VALUE (ret);
40 }
41
42 return ret;
43}
