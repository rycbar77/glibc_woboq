23static inline int __attribute__ ((always_inline))
24get_cet_status (void)
25{
26 unsigned long long cet_status[3];
27 if (INTERNAL_SYSCALL_CALL (arch_prctl, ARCH_CET_STATUS, cet_status) == 0)
28 return cet_status[0];
29 return 0;
30}
