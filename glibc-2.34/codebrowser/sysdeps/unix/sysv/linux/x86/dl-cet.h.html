18#include <sys/prctl.h>
19#include <asm/prctl.h>
20
21static inline int __attribute__ ((always_inline))
22dl_cet_disable_cet (unsigned int cet_feature)
23{
24 return (int) INTERNAL_SYSCALL_CALL (arch_prctl, ARCH_CET_DISABLE,
25 cet_feature);
26}
27
28static inline int __attribute__ ((always_inline))
29dl_cet_lock_cet (void)
30{
31 return (int) INTERNAL_SYSCALL_CALL (arch_prctl, ARCH_CET_LOCK, 0);
32}
