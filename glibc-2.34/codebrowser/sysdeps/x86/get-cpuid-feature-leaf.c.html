21const struct cpuid_feature *
22__x86_get_cpuid_feature_leaf (unsigned int leaf)
23{
24 static const struct cpuid_feature feature = {};
25 if (leaf < CPUID_INDEX_MAX)
26 return ((const struct cpuid_feature *)
27 &GLRO(dl_x86_cpu_features).features[leaf]);
28 else
29 return &feature;
30}
