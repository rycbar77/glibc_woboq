21ENTRY(__pthread_spin_lock)
221: LOCK
23 decl 0(%rdi)
24 jne 2f
25 xor %eax, %eax
26 ret

28 .align 16
292: rep
30 nop
31 cmpl $0, 0(%rdi)
32 jg 1b
33 jmp 2b
34END(__pthread_spin_lock)
