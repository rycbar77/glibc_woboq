19#include <sysdep.h>
20#include <shlib-compat.h>

22ENTRY(__pthread_spin_unlock)
23 movl $1, (%rdi)
24 xorl %eax, %eax
25 retq
26END(__pthread_spin_unlock)
27versioned_symbol (libc, __pthread_spin_unlock, pthread_spin_unlock, GLIBC_2_34)
28/* The implementation of pthread_spin_init is identical. */
29versioned_symbol (libc, __pthread_spin_unlock, pthread_spin_init, GLIBC_2_34)
