19#include <math.h>
20
21FLOAT
22M_DECL_FUNC (__fmaximum_num) (FLOAT x, FLOAT y)
23{
24 if (isgreater (x, y))
25 return x;
26 else if (isless (x, y))
27 return y;
28 else if (x == y)
29 return (M_COPYSIGN (1, x) >= M_COPYSIGN (1, y) ? x : y);
30 else
31 return isnan (y) ? (isnan (x) ? x + y : x) : y;
32}
33declare_mgen_alias (__fmaximum_num, fmaximum_num);
