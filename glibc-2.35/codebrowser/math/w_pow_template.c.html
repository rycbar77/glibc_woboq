19/* Only build wrappers from the templates for the types that define the macro
20 below. This macro is set in math-type-macros-<type>.h in sysdeps/generic
21 for each floating-point type. */
22#if __USE_WRAPPER_TEMPLATE
23
24# include <errno.h>
25# include <fenv.h>
26# include <math.h>
27# include <math_private.h>
28
29FLOAT
30M_DECL_FUNC (__pow) (FLOAT x, FLOAT y)
31{
32 FLOAT z = M_SUF (__ieee754_pow) (x, y);
33 if (__glibc_unlikely (!isfinite (z)))
34 {
35 if (isfinite (x) && isfinite (y))
36 {
37 if (isnan (z))
38 /* Domain error: pow(x<0,y=non-integer). */
39 __set_errno (EDOM);
40 else
41 /* Pole error: pow(x=0,y<0). Or overflow. */
42 __set_errno (ERANGE);
43 }
44 }
45 else if (__glibc_unlikely (z == 0) && isfinite (x) && x != 0 && isfinite (y))
46 /* Underflow. */
47 __set_errno (ERANGE);
48 return z;
49}
50declare_mgen_alias (__pow, pow)
51
52#endif /* __USE_WRAPPER_TEMPLATE. */
