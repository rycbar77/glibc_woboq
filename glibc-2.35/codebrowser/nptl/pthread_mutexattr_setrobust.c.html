18#include <errno.h>
19#include <pthreadP.h>
20#include <shlib-compat.h>
21
22int
23__pthread_mutexattr_setrobust (pthread_mutexattr_t *attr, int robustness)
24{
25 if (robustness != PTHREAD_MUTEX_STALLED_NP
26 && __builtin_expect (robustness != PTHREAD_MUTEX_ROBUST_NP, 0))
27 return EINVAL;
28
29 struct pthread_mutexattr *iattr = (struct pthread_mutexattr *) attr;
30
31 /* We use bit 30 to signal whether the mutex is going to be
32 robust or not. */
33 if (robustness == PTHREAD_MUTEX_STALLED_NP)
34 iattr->mutexkind &= ~PTHREAD_MUTEXATTR_FLAG_ROBUST;
35 else
36 iattr->mutexkind |= PTHREAD_MUTEXATTR_FLAG_ROBUST;
37
38 return 0;
39}
40versioned_symbol (libc, __pthread_mutexattr_setrobust,
41 pthread_mutexattr_setrobust, GLIBC_2_34);
