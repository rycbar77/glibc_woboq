18#include <stdlib.h>
19#include "pthreadP.h"
20#include <shlib-compat.h>
21
22void
23___pthread_testcancel (void)
24{
25 struct pthread *self = THREAD_SELF;
26 int cancelhandling = THREAD_GETMEM (self, cancelhandling);
27 if (self->cancelstate == PTHREAD_CANCEL_ENABLE
28 && (cancelhandling & CANCELED_BITMASK)
29 && !(cancelhandling & EXITING_BITMASK)
30 && !(cancelhandling & TERMINATED_BITMASK))
31 {
32 THREAD_SETMEM (self, result, PTHREAD_CANCELED);
33 __do_cancel ();
34 }
35}
