32bool
33__td_ta_rtld_global (td_thragent_t *ta)
34{
35 if (ta->ta_addr__rtld_global == 0)
36 {
37 psaddr_t rtldglobalp;
38 if (DB_GET_VALUE (rtldglobalp, ta, __nptl_rtld_global, 0) == TD_OK)
39 ta->ta_addr__rtld_global = rtldglobalp;
40 else
41 ta->ta_addr__rtld_global = (void *) -1;
42 }
43
44 return ta->ta_addr__rtld_global != (void *)-1;
45}
