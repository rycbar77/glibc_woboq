31 {
32 size_t cnt;
33

35 {
36 unsigned int number;
37
38 if (cnt < 5)
39 INT_FIELD (number, ISCOLON , 0, 16, (unsigned int))
40 else
41 INT_FIELD (number, isspace, 1, 16, (unsigned int))
42
43 if (number > 0xff)
44 return 0;
45 result->e_addr.ether_addr_octet[cnt] = number;
46 }
47 };
