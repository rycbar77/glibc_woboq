21#include <errno.h>
22#include <stdio_ext.h>

24FILE *
25__nss_files_fopen (const char *path)
26{
27 FILE *fp = fopen (path, "rce");
28 if (fp == NULL)
29 return NULL;
30
31 /* The stream is not shared across threads. */
32 __fsetlocking (fp, FSETLOCKING_BYCALLER);
33
34 /* This tells libio that the file is seekable, and that fp->_offset
35 is correct, ensuring that __ftello64 is efficient (bug 26257). */
36 if (__fseeko64 (fp, 0, SEEK_SET) < 0)
37 {
38 /* nss_files requires seekable files, to deal with repeated
39 reads of the same line after reporting ERANGE. */
40 fclose (fp);
41 __set_errno (ESPIPE);
42 return NULL;
43 }
44
45 return fp;
46}
