23/* Return a string describing the errno code in ERRNUM. */
24char *
25__strerror_r (int errnum, char *buf, size_t buflen)
26{
27 char *err = (char *) __get_errlist (errnum);
28 if (__glibc_unlikely (err == NULL))
29 {
30 __snprintf (buf, buflen, "%s%d", _("Unknown error "), errnum);
31 return buf;
32 }
33
34 return _(err);
35}
36weak_alias (__strerror_r, strerror_r)
37libc_hidden_def (__strerror_r)
