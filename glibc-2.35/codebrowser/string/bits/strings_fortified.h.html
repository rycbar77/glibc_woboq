19#ifndef __STRINGS_FORTIFIED
20# define __STRINGS_FORTIFIED 1
21
22__fortify_function void
23__NTH (bcopy (const void *__src, void *__dest, size_t __len))
24{
25 (void) __builtin___memmove_chk (__dest, __src, __len,
26 __glibc_objsize0 (__dest));
27}
28
29__fortify_function void
30__NTH (bzero (void *__dest, size_t __len))
31{
32 (void) __builtin___memset_chk (__dest, '\0', __len,
33 __glibc_objsize0 (__dest));
34}
35
36#endif
