19#include <string.h>
20#include <signal.h>
21#include <array_length.h>

23const char *
24__sigdescr_np (int signum)
25{
26 const char *descr = NULL;
27
28 if (signum >= 0 && signum <= NSIG && signum < array_length (__sys_siglist))
29 descr = __sys_siglist[signum];
30
31 return descr;
32}
