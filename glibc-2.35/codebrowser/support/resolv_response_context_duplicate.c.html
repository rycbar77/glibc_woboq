20#include <support/resolv_test.h>

23struct resolv_response_context *
24resolv_response_context_duplicate (const struct resolv_response_context *ctx)
25{
26 struct resolv_response_context *result = xmalloc (sizeof (*result));
27 memcpy (result, ctx, sizeof (*result));
28 if (result->client_address != NULL)
29 {
30 result->client_address = xmalloc (result->client_address_length);
31 memcpy (result->client_address, ctx->client_address,
32 result->client_address_length);
33 }
34 result->query_buffer = xmalloc (result->query_length);
35 memcpy (result->query_buffer, ctx->query_buffer, result->query_length);
36 return result;
37}
