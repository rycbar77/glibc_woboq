20#include <support/check.h>
21
22size_t
23xgetline (char **lineptr, size_t *n, FILE *stream)
24{
25 TEST_VERIFY (!ferror (stream));
26 ssize_t ret = getline (lineptr, n, stream);
27 if (ferror (stream))
28 {
29 TEST_VERIFY (ret < 0);
30 FAIL_EXIT1 ("getline: %m");
31 }
32 if (feof (stream))
33 {
34 TEST_VERIFY (ret <= 0);
35 return 0;
36 }
37 TEST_VERIFY (ret > 0);
38 return ret;
39}
