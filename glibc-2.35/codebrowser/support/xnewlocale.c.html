23locale_t
24xnewlocale (int category_mask, const char *locale, locale_t base)
25{
26 locale_t r = newlocale (category_mask, locale, base);
27 if (r == (locale_t) 0)
28 FAIL_EXIT1 ("error: newlocale (%d, \"%s\", %p)\n", category_mask,
29 locale, base);
30 return r;
31}
