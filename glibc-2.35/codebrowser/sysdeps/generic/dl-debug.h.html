19#ifndef _DL_DEBUG_H
20#define _DL_DEBUG_H
21
22/* There is a DT_DEBUG entry in the dynamic section. Fill it in with the
23 run-time address of the r_debug structure */
24
25static inline void
26__attribute ((always_inline))
27elf_setup_debug_entry (struct link_map *l, struct r_debug *r)
28{
29 if (l->l_info[DT_DEBUG] != NULL)
30 l->l_info[DT_DEBUG]->d_un.d_ptr = (ElfW(Addr)) r;
31}
32
33#endif /* _DL_DEBUG_H */
