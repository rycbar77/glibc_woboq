19/* Return the address of the struct r_debug after relocation. */
20
21static inline EW(Addr)
22E(r_debug_address) (EW(Dyn) *d)
23{
24 if (d->d_tag == DT_DEBUG)
25 return (EW(Addr)) d->d_un.d_ptr;
26
27 return 0;
28}
29
30/* Return the offset of the struct r_debug before relocation. */
31
32static inline EW(Addr)
33E(r_debug_offset) (EW(Dyn) *d, int fd, EW(Addr) offset)
34{
35 return E(r_debug_address) (d);
36}
