19#include "thrd_priv.h"
20#include <shlib-compat.h>
21
22int
23__thrd_join (thrd_t thr, int *res)
24{
25 void *pthread_res;
26 int err_code = __pthread_join (thr, &pthread_res);
27 if (res)
28 *res = (int) (uintptr_t) pthread_res;
29
30 return thrd_err_map (err_code);
31}
32#if PTHREAD_IN_LIBC
33versioned_symbol (libc, __thrd_join, thrd_join, GLIBC_2_34);
34# if OTHER_SHLIB_COMPAT (libpthread, GLIBC_2_28, GLIBC_2_34)
35compat_symbol (libc, __thrd_join, thrd_join, GLIBC_2_28);
36# endif
37#else /* !PTHREAD_IN_LIBC */
38strong_alias (__thrd_join, thrd_join)
39#endif
