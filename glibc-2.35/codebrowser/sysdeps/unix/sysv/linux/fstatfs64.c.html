27/* Return information about the filesystem on which FD resides. */
28int
29__fstatfs64 (int fd, struct statfs64 *buf)
30{
31#ifdef __NR_fstatfs64
32 return INLINE_SYSCALL_CALL (fstatfs64, fd, sizeof (*buf), buf);
33#else
34 return INLINE_SYSCALL_CALL (fstatfs, fd, buf);
35#endif
36}
