18#include <fcntl.h>
19#include <unistd.h>
20#include <paths.h>
21#include <stdlib.h>
22
23/* Path to the master pseudo terminal cloning device. */
24#define _PATH_DEVPTMX _PATH_DEV "ptmx"
25
26/* Open a master pseudo terminal and return its file descriptor. */
27int
28__posix_openpt (int oflag)
29{
30 return __open (_PATH_DEVPTMX, oflag);
31}
32weak_alias (__posix_openpt, posix_openpt)
33
34
35int
36__getpt (void)
37{
38 return __posix_openpt (O_RDWR);
39}
40libc_hidden_def (__getpt)
41weak_alias (__getpt, getpt)
