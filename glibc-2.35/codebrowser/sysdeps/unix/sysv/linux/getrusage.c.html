19#include <sys/time.h>
20#include <sys/resource.h>
21#include <sys/types.h>
22#include <sysdep.h>
23#include <tv32-compat.h>
24
25int
26__getrusage64 (enum __rusage_who who, struct __rusage64 *usage)
27{
28#if __KERNEL_OLD_TIMEVAL_MATCHES_TIMEVAL64
29 return INLINE_SYSCALL_CALL (getrusage, who, usage);
30#else
31 struct __rusage32 usage32;
32 if (INLINE_SYSCALL_CALL (getrusage, who, &usage32) == -1)
33 return -1;
34
35 rusage32_to_rusage64 (&usage32, usage);
36 return 0;
37#endif
38}
39
40#if __TIMESIZE != 64
41libc_hidden_def (__getrusage64)
42int
43__getrusage (enum __rusage_who who, struct rusage *usage)
44{
45 int ret ;
46 struct __rusage64 usage64;
47
48 ret = __getrusage64 (who, &usage64);
49
50 if (ret != 0)
51 return ret;
52
53 rusage64_to_rusage (&usage64, usage);
54
55 return ret;
56}
57#endif
58weak_alias (__getrusage, getrusage)
