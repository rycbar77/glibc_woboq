19#include <time.h>
20#include <errno.h>
21
22int
23__nanosleep64 (const struct __timespec64 *req, struct __timespec64 *rem)
24{
25 int ret = __clock_nanosleep_time64 (CLOCK_REALTIME, 0, req, rem);
26 if (ret != 0)
27 {
28 __set_errno (ret);
29 return -1;
30 }
31 return 0;
32}
33#if __TIMESIZE != 64
34libc_hidden_def (__nanosleep64)
35
36int
37__nanosleep (const struct timespec *req, struct timespec *rem)
38{
39 struct __timespec64 treq64, trem64;
40
41 treq64 = valid_timespec_to_timespec64 (*req);
42 int ret = __nanosleep64 (&treq64, rem != NULL ? &trem64 : NULL);
43
44 if (ret != 0 && errno == EINTR && rem != NULL)
45 *rem = valid_timespec64_to_timespec (trem64);
46 return ret;
47}
48#endif
49libc_hidden_def (__nanosleep)
