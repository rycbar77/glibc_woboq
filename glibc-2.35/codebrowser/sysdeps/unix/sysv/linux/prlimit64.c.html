24int
25__prlimit64 (pid_t pid, enum __rlimit_resource resource,
26 const struct rlimit64 *new_rlimit, struct rlimit64 *old_rlimit)
27{
28 return INLINE_SYSCALL_CALL (prlimit64, pid, resource, new_rlimit,
29 old_rlimit);
30}
