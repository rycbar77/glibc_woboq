26int
27__stat64_time64 (const char *file, struct __stat64_t64 *buf)
28{
29 return __fstatat64_time64 (AT_FDCWD, file, buf, 0);
30}

34int
35__stat64 (const char *file, struct stat64 *buf)
36{

40}
