18#include <dirent.h>

20#include <dirstream.h>

22/* Return the current position of DIRP. */
23long int
24telldir (DIR *dirp)
25{
26 long int ret;
27
28 __libc_lock_lock (dirp->lock);
29 ret = dirp->filepos;
30 __libc_lock_unlock (dirp->lock);
31
32 return ret;
33}
