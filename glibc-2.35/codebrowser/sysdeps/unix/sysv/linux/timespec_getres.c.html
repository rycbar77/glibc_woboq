19#include <time.h>
20
21/* Set TS to resolution of time base BASE. */
22int
23__timespec_getres64 (struct __timespec64 *ts, int base)
24{
25 if (base == TIME_UTC)
26 {
27 __clock_getres64 (CLOCK_REALTIME, ts);
28 return base;
29 }
30 return 0;
31}
32
33#if __TIMESIZE != 64
34libc_hidden_def (__timespec_getres64)
35
36int
37__timespec_getres (struct timespec *ts, int base)
38{
39 int ret;
40 struct __timespec64 tp64;
41
42 ret = __timespec_getres64 (&tp64, base);
43
44 if (ret == TIME_UTC && ts != NULL)
45 *ts = valid_timespec64_to_timespec (tp64);
46
47 return ret;
48}
49#endif
50strong_alias (__timespec_getres, timespec_getres);
