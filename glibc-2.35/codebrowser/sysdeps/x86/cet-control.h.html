19#ifndef _CET_CONTROL_H
20#define _CET_CONTROL_H
21
22/* For each CET feature, IBT and SHSTK, valid control values. */
23enum dl_x86_cet_control
24{
25 /* Enable CET features based on ELF property note. */
26 cet_elf_property = 0,
27 /* Always enable CET features. */
28 cet_always_on,
29 /* Always disable CET features. */
30 cet_always_off,
31 /* Enable CET features permissively. */
32 cet_permissive
33};
34
35struct dl_x86_feature_control
36{
37 enum dl_x86_cet_control ibt : 2;
38 enum dl_x86_cet_control shstk : 2;
39};
40
41#endif /* cet-control.h */
