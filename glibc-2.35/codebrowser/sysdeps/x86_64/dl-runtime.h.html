20/* The ABI calls for the PLT stubs to pass the index of the relocation
21 and not its offset. In _dl_profile_fixup and _dl_audit_pltexit we
22 also use the index. Therefore it is wasteful to compute the offset
23 in the trampoline just to reverse the operation immediately
24 afterwards. */
25static inline uintptr_t
26reloc_offset (uintptr_t plt0, uintptr_t pltn)
27{
28 return pltn * sizeof (ElfW(Rela));
29}
30
31static inline uintptr_t
32reloc_index (uintptr_t plt0, uintptr_t pltn, size_t size)
33{
34 return pltn;
35}
