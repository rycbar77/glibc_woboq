19/*
20 * ALGORITHM DESCRIPTION:
21 *
22 * Compute atanh(x) as 0.5 * log((1 + x)/(1 - x))
23 * using small lookup table that map to AVX-512 permute instructions
24 *
25 * Special cases:
26 *
27 * atanh(0) = 0
28 * atanh(+1) = +INF
29 * atanh(-1) = -INF
30 * atanh(x) = NaN if |x| > 1, or if x is a NaN or INF
31 *
32 */
33
34/* Offsets for data table __svml_datanh_data_internal_avx512
35 */
36#define Log_tbl_H 0
37#define Log_tbl_L 128
38#define One 256
39#define AbsMask 320
40#define AddB5 384
41#define RcpBitMask 448
42#define poly_coeff8 512
43#define poly_coeff7 576
44#define poly_coeff6 640
45#define poly_coeff5 704
46#define poly_coeff4 768
47#define poly_coeff3 832
48#define poly_coeff2 896
49#define poly_coeff1 960
50#define poly_coeff0 1024
51#define Half 1088
52#define L2H 1152
53#define L2L 1216

59ENTRY(_ZGVeN8v_atanh_skx)
60 pushq %rbp
61 cfi_def_cfa_offset(16)
62 movq %rsp, %rbp
63 cfi_def_cfa(6, 16)
64 cfi_offset(6, -16)
65 andq $-64, %rsp
66 subq $192, %rsp
67 vmovups One+__svml_datanh_data_internal_avx512(%rip), %zmm15
68
69/* round reciprocals to 1+4b mantissas */
70 vmovups AddB5+__svml_datanh_data_internal_avx512(%rip), %zmm6
71 vmovups RcpBitMask+__svml_datanh_data_internal_avx512(%rip), %zmm9
72 vmovaps %zmm0, %zmm2
73 vandpd AbsMask+__svml_datanh_data_internal_avx512(%rip), %zmm2, %zmm13
74
75/* 1+y */
76 vaddpd {rn-sae}, %zmm15, %zmm13, %zmm0
77
78/* 1-y */
79 vsubpd {rn-sae}, %zmm13, %zmm15, %zmm4
80 vxorpd %zmm13, %zmm2, %zmm1
81
82/* Yp_high */
83 vsubpd {rn-sae}, %zmm15, %zmm0, %zmm7
84
85/* -Ym_high */
86 vsubpd {rn-sae}, %zmm15, %zmm4, %zmm12
87
88/* RcpP ~ 1/Yp */
89 vrcp14pd %zmm0, %zmm3
90
91/* RcpM ~ 1/Ym */
92 vrcp14pd %zmm4, %zmm5
93
94/* input outside (-1, 1) ? */
95 vcmppd $21, {sae}, %zmm15, %zmm13, %k0
96 vpaddq %zmm6, %zmm3, %zmm11
97 vpaddq %zmm6, %zmm5, %zmm10
98
99/* Yp_low */
100 vsubpd {rn-sae}, %zmm7, %zmm13, %zmm8
101 vandpd %zmm9, %zmm11, %zmm14
102 vandpd %zmm9, %zmm10, %zmm3
103
104/* Ym_low */
105 vaddpd {rn-sae}, %zmm12, %zmm13, %zmm12
106
107/* Reduced argument: Rp = (RcpP*Yp - 1)+RcpP*Yp_low */
108 vfmsub213pd {rn-sae}, %zmm15, %zmm14, %zmm0
109
110/* Reduced argument: Rm = (RcpM*Ym - 1)+RcpM*Ym_low */
111 vfmsub231pd {rn-sae}, %zmm3, %zmm4, %zmm15
112
113/* exponents */
114 vgetexppd {sae}, %zmm14, %zmm5
115 vgetexppd {sae}, %zmm3, %zmm4
116
117/* Table lookups */
118 vmovups __svml_datanh_data_internal_avx512(%rip), %zmm9
119 vmovups Log_tbl_H+64+__svml_datanh_data_internal_avx512(%rip), %zmm13
120 vmovups Log_tbl_L+__svml_datanh_data_internal_avx512(%rip), %zmm7
121 vfmadd231pd {rn-sae}, %zmm14, %zmm8, %zmm0
122 vfnmadd231pd {rn-sae}, %zmm3, %zmm12, %zmm15
123
124/* Prepare table index */
125 vpsrlq $48, %zmm14, %zmm11
126 vpsrlq $48, %zmm3, %zmm8
127 vmovups Log_tbl_L+64+__svml_datanh_data_internal_avx512(%rip), %zmm14
128
129/* polynomials */
130 vmovups poly_coeff8+__svml_datanh_data_internal_avx512(%rip), %zmm3
131
132/* Km-Kp */
133 vsubpd {rn-sae}, %zmm5, %zmm4, %zmm5
134 vmovups poly_coeff7+__svml_datanh_data_internal_avx512(%rip), %zmm4
135 kmovw %k0, %edx
136 vmovaps %zmm11, %zmm10
137 vmovaps %zmm4, %zmm6
138 vpermi2pd %zmm13, %zmm9, %zmm10
139 vpermi2pd %zmm14, %zmm7, %zmm11
140 vpermt2pd %zmm13, %zmm8, %zmm9
141 vpermt2pd %zmm14, %zmm8, %zmm7
142 vmovups poly_coeff6+__svml_datanh_data_internal_avx512(%rip), %zmm8
143 vfmadd231pd {rn-sae}, %zmm0, %zmm3, %zmm6
144 vfmadd231pd {rn-sae}, %zmm15, %zmm3, %zmm4
145 vmovups poly_coeff3+__svml_datanh_data_internal_avx512(%rip), %zmm13
146 vmovups poly_coeff2+__svml_datanh_data_internal_avx512(%rip), %zmm14
147 vfmadd213pd {rn-sae}, %zmm8, %zmm0, %zmm6
148 vfmadd213pd {rn-sae}, %zmm8, %zmm15, %zmm4
149 vmovups poly_coeff0+__svml_datanh_data_internal_avx512(%rip), %zmm8
150 vsubpd {rn-sae}, %zmm11, %zmm7, %zmm12
151
152/* table values */
153 vsubpd {rn-sae}, %zmm10, %zmm9, %zmm3
154 vmovups poly_coeff5+__svml_datanh_data_internal_avx512(%rip), %zmm7
155 vmovups poly_coeff4+__svml_datanh_data_internal_avx512(%rip), %zmm9
156
157/* K*L2H + Th */
158 vmovups L2H+__svml_datanh_data_internal_avx512(%rip), %zmm10
159
160/* K*L2L + Tl */
161 vmovups L2L+__svml_datanh_data_internal_avx512(%rip), %zmm11
162 vfmadd213pd {rn-sae}, %zmm7, %zmm0, %zmm6
163 vfmadd213pd {rn-sae}, %zmm7, %zmm15, %zmm4
164 vmovups poly_coeff1+__svml_datanh_data_internal_avx512(%rip), %zmm7
165 vfmadd231pd {rn-sae}, %zmm5, %zmm10, %zmm3
166 vfmadd213pd {rn-sae}, %zmm12, %zmm11, %zmm5
167 vfmadd213pd {rn-sae}, %zmm9, %zmm0, %zmm6
168 vfmadd213pd {rn-sae}, %zmm9, %zmm15, %zmm4
169 vfmadd213pd {rn-sae}, %zmm13, %zmm0, %zmm6
170 vfmadd213pd {rn-sae}, %zmm13, %zmm15, %zmm4
171 vfmadd213pd {rn-sae}, %zmm14, %zmm0, %zmm6
172 vfmadd213pd {rn-sae}, %zmm14, %zmm15, %zmm4
173 vfmadd213pd {rn-sae}, %zmm7, %zmm0, %zmm6
174 vfmadd213pd {rn-sae}, %zmm7, %zmm15, %zmm4
175 vfmadd213pd {rn-sae}, %zmm8, %zmm0, %zmm6
176 vfmadd213pd {rn-sae}, %zmm8, %zmm15, %zmm4
177
178/* (K*L2L + Tl) + Rp*PolyP */
179 vfmadd213pd {rn-sae}, %zmm5, %zmm0, %zmm6
180 vorpd Half+__svml_datanh_data_internal_avx512(%rip), %zmm1, %zmm0
181
182/* (K*L2L + Tl) + Rp*PolyP -Rm*PolyM */
183 vfnmadd213pd {rn-sae}, %zmm6, %zmm15, %zmm4
184 vaddpd {rn-sae}, %zmm4, %zmm3, %zmm1
185 vmulpd {rn-sae}, %zmm0, %zmm1, %zmm0
186 testl %edx, %edx
187
188/* Go to special inputs processing branch */
189 jne L(SPECIAL_VALUES_BRANCH)
190 # LOE rbx r12 r13 r14 r15 edx zmm0 zmm2
191
192/* Restore registers
193 * and exit the function
194 */
195
196L(EXIT):
197 movq %rbp, %rsp
198 popq %rbp
199 cfi_def_cfa(7, 8)
200 cfi_restore(6)
201 ret
202 cfi_def_cfa(6, 16)
203 cfi_offset(6, -16)
204
205/* Branch to process
206 * special inputs
207 */
208
209L(SPECIAL_VALUES_BRANCH):
210 vmovups %zmm2, 64(%rsp)
211 vmovups %zmm0, 128(%rsp)
212 # LOE rbx r12 r13 r14 r15 edx zmm0
213
214 xorl %eax, %eax
215 # LOE rbx r12 r13 r14 r15 eax edx
216
217 vzeroupper
218 movq %r12, 16(%rsp)
219 /* DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -176; DW_OP_plus) */
220 .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x50, 0xff, 0xff, 0xff, 0x22
221 movl %eax, %r12d
222 movq %r13, 8(%rsp)
223 /* DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -184; DW_OP_plus) */
224 .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x48, 0xff, 0xff, 0xff, 0x22
225 movl %edx, %r13d
226 movq %r14, (%rsp)
227 /* DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -192; DW_OP_plus) */
228 .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x40, 0xff, 0xff, 0xff, 0x22
229 # LOE rbx r15 r12d r13d
230
231/* Range mask
232 * bits check
233 */
234
235L(RANGEMASK_CHECK):
236 btl %r12d, %r13d
237
238/* Call scalar math function */
239 jc L(SCALAR_MATH_CALL)
240 # LOE rbx r15 r12d r13d
241
242/* Special inputs
243 * processing loop
244 */
245
246L(SPECIAL_VALUES_LOOP):
247 incl %r12d
248 cmpl $8, %r12d
249
250/* Check bits in range mask */
251 jl L(RANGEMASK_CHECK)
252 # LOE rbx r15 r12d r13d
253
254 movq 16(%rsp), %r12
255 cfi_restore(12)
256 movq 8(%rsp), %r13
257 cfi_restore(13)
258 movq (%rsp), %r14
259 cfi_restore(14)
260 vmovups 128(%rsp), %zmm0
261
262/* Go to exit */
263 jmp L(EXIT)
264 /* DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -176; DW_OP_plus) */
265 .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x50, 0xff, 0xff, 0xff, 0x22
266 /* DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -184; DW_OP_plus) */
267 .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x48, 0xff, 0xff, 0xff, 0x22
268 /* DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -192; DW_OP_plus) */
269 .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x40, 0xff, 0xff, 0xff, 0x22
270 # LOE rbx r12 r13 r14 r15 zmm0
271
272/* Scalar math fucntion call
273 * to process special input
274 */
275
276L(SCALAR_MATH_CALL):
277 movl %r12d, %r14d
278 movsd 64(%rsp,%r14,8), %xmm0
279 call atanh@PLT
280 # LOE rbx r14 r15 r12d r13d xmm0
281
282 movsd %xmm0, 128(%rsp,%r14,8)
283
284/* Process special inputs in loop */
285 jmp L(SPECIAL_VALUES_LOOP)
286 # LOE rbx r15 r12d r13d
287END(_ZGVeN8v_atanh_skx)
