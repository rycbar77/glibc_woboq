19/*
20 * ALGORITHM DESCRIPTION:
21 *
22 * x=2^{3*k+j} * 1.b1 b2 ... b5 b6 ... b52
23 * Let r=(x*2^{-3k-j} - 1.b1 b2 ... b5 1)* rcp[b1 b2 ..b5],
24 * where rcp[b1 b2 .. b5]=1/(1.b1 b2 b3 b4 b5 1) in double precision
25 * cbrt(2^j * 1. b1 b2 .. b5 1) is approximated as T[j][b1..b5]+D[j][b1..b5]
26 * (T stores the high 53 bits, D stores the low order bits)
27 * Result=2^k*T+(2^k*T*r)*P+2^k*D
28 * where P=p1+p2*r+..+p8*r^7
29 *
30 */
31

61ENTRY(_ZGVbN2v_cbrt_sse4)
62 subq $72, %rsp
63 cfi_def_cfa_offset(80)
64
65/* Calculate CbrtIndex */
66 movaps %xmm0, %xmm10
67 psrlq $52, %xmm10
68
69/* Load 1/(1+iRcpIndex/32+1/64) reciprocal table value */
70 lea __svml_dcbrt_data_internal(%rip), %r8
71 pand _lExpLoMask+__svml_dcbrt_data_internal(%rip), %xmm10
72 movdqu _l1556+__svml_dcbrt_data_internal(%rip), %xmm9
73 pmuludq %xmm10, %xmm9
74
75/* If the exponent field is zero - go to callout to process denormals */
76 movq _iAbsMask+__svml_dcbrt_data_internal(%rip), %xmm7
77
78/* Calculate Rcp table index */
79 movq _iRcpIndexMask+__svml_dcbrt_data_internal(%rip), %xmm13
80
81/* Get iX - high part of argument */
82 pshufd $221, %xmm0, %xmm4
83
84/*
85 * Declarations
86 * Load constants
87 */
88 movq _iSignMask+__svml_dcbrt_data_internal(%rip), %xmm1
89 pand %xmm4, %xmm7
90 pand %xmm4, %xmm13
91
92/* Compute 2^k */
93 psrld $20, %xmm4
94 movq _iBias+__svml_dcbrt_data_internal(%rip), %xmm2
95 pand %xmm1, %xmm4
96 pshufd $136, %xmm9, %xmm15
97 por %xmm2, %xmm4
98 psrld $14, %xmm15
99 psrld $12, %xmm13
100 paddd %xmm15, %xmm4
101 pxor %xmm2, %xmm2
102 pslld $20, %xmm4
103 movdqa %xmm15, %xmm11
104 movd %xmm13, %edx
105 paddd %xmm15, %xmm11
106 pshufd $1, %xmm13, %xmm8
107 punpckldq %xmm4, %xmm2
108
109/*
110 * VAND( L, l2k, = l2k, lExpHiMask );
111 * Argument reduction Z
112 */
113 movups _dMantissaMask+__svml_dcbrt_data_internal(%rip), %xmm1
114 movups _dSgnf6Mask+__svml_dcbrt_data_internal(%rip), %xmm4
115 andps %xmm0, %xmm1
116 movd %xmm8, %ecx
117 andps %xmm0, %xmm4
118 orps _dNegOne+__svml_dcbrt_data_internal(%rip), %xmm1
119 orps _dNeg65Div64+__svml_dcbrt_data_internal(%rip), %xmm4
120 movslq %edx, %rdx
121 subpd %xmm4, %xmm1
122 movslq %ecx, %rcx
123 movsd (%r8,%rdx), %xmm3
124 movq _iSub+__svml_dcbrt_data_internal(%rip), %xmm5
125 psubd %xmm5, %xmm7
126 movhpd (%r8,%rcx), %xmm3
127 mulpd %xmm1, %xmm3
128
129/* Polynomial */
130 movups _dA7+__svml_dcbrt_data_internal(%rip), %xmm5
131 mulpd %xmm3, %xmm5
132 addpd _dA6+__svml_dcbrt_data_internal(%rip), %xmm5
133 mulpd %xmm3, %xmm5
134 addpd _dA5+__svml_dcbrt_data_internal(%rip), %xmm5
135 mulpd %xmm3, %xmm5
136 addpd _dA4+__svml_dcbrt_data_internal(%rip), %xmm5
137 mulpd %xmm3, %xmm5
138 addpd _dA3+__svml_dcbrt_data_internal(%rip), %xmm5
139 pshufd $136, %xmm10, %xmm12
140 psubd %xmm15, %xmm12
141 psubd %xmm11, %xmm12
142 mulpd %xmm3, %xmm5
143 pslld $8, %xmm12
144 paddd %xmm12, %xmm13
145
146/* Load cbrt(2^j*(1+iRcpIndex/32+1/64)) Hi & Lo values */
147 movd %xmm13, %esi
148 pshufd $1, %xmm13, %xmm14
149 movq _iCmp+__svml_dcbrt_data_internal(%rip), %xmm6
150 movd %xmm14, %edi
151 pcmpgtd %xmm6, %xmm7
152 movmskps %xmm7, %eax
153 addpd _dA2+__svml_dcbrt_data_internal(%rip), %xmm5
154 movslq %esi, %rsi
155 movslq %edi, %rdi
156 mulpd %xmm3, %xmm5
157 movsd 256(%r8,%rsi), %xmm6
158 movhpd 256(%r8,%rdi), %xmm6
159
160/* THi*2^k, TLo*2^k */
161 mulpd %xmm2, %xmm6
162 addpd _dA1+__svml_dcbrt_data_internal(%rip), %xmm5
163
164/* THi*2^k*Z */
165 mulpd %xmm6, %xmm3
166
167/* Final reconstruction */
168 mulpd %xmm3, %xmm5
169 addpd %xmm5, %xmm6
170 andl $3, %eax
171
172/* Go to special inputs processing branch */
173 jne L(SPECIAL_VALUES_BRANCH)
174 # LOE rbx rbp r12 r13 r14 r15 eax xmm0 xmm6
175
176/* Restore registers
177 * and exit the function
178 */
179
180L(EXIT):
181 movaps %xmm6, %xmm0
182 addq $72, %rsp
183 cfi_def_cfa_offset(8)
184 ret
185 cfi_def_cfa_offset(80)
186
187/* Branch to process
188 * special inputs
189 */
190
191L(SPECIAL_VALUES_BRANCH):
192 movups %xmm0, 32(%rsp)
193 movups %xmm6, 48(%rsp)
194 # LOE rbx rbp r12 r13 r14 r15 eax xmm6
195
196 xorl %edx, %edx
197 movq %r12, 16(%rsp)
198 cfi_offset(12, -64)
199 movl %edx, %r12d
200 movq %r13, 8(%rsp)
201 cfi_offset(13, -72)
202 movl %eax, %r13d
203 movq %r14, (%rsp)
204 cfi_offset(14, -80)
205 # LOE rbx rbp r15 r12d r13d
206
207/* Range mask
208 * bits check
209 */
210
211L(RANGEMASK_CHECK):
212 btl %r12d, %r13d
213
214/* Call scalar math function */
215 jc L(SCALAR_MATH_CALL)
216 # LOE rbx rbp r15 r12d r13d
217
218/* Special inputs
219 * processing loop
220 */
221
222L(SPECIAL_VALUES_LOOP):
223 incl %r12d
224 cmpl $2, %r12d
225
226/* Check bits in range mask */
227 jl L(RANGEMASK_CHECK)
228 # LOE rbx rbp r15 r12d r13d
229
230 movq 16(%rsp), %r12
231 cfi_restore(12)
232 movq 8(%rsp), %r13
233 cfi_restore(13)
234 movq (%rsp), %r14
235 cfi_restore(14)
236 movups 48(%rsp), %xmm6
237
238/* Go to exit */
239 jmp L(EXIT)
240 cfi_offset(12, -64)
241 cfi_offset(13, -72)
242 cfi_offset(14, -80)
243 # LOE rbx rbp r12 r13 r14 r15 xmm6
244
245/* Scalar math fucntion call
246 * to process special input
247 */
248
249L(SCALAR_MATH_CALL):
250 movl %r12d, %r14d
251 movsd 32(%rsp,%r14,8), %xmm0
252 call cbrt@PLT
253 # LOE rbx rbp r14 r15 r12d r13d xmm0
254
255 movsd %xmm0, 48(%rsp,%r14,8)
256
257/* Process special inputs in loop */
258 jmp L(SPECIAL_VALUES_LOOP)
259 # LOE rbx rbp r15 r12d r13d
260END(_ZGVbN2v_cbrt_sse4)
