19/*
20 * ALGORITHM DESCRIPTION:
21 *
22 * Basic formula is
23 * erf(x) ~ erf(x0) +
24 * + exp(-x0*x0)*D*(1+c0+T*P1(T)+D^2*P3(T)+D^4*P5(T)+D^6*p7+D^8*p9)
25 * where D=x-x0, T=x0*D
26 * x0 is x rounded to a specified number of fractional bits (in this case 7),
27 * except that x0=0 for |x|<3.5/128.0 (using x0=0 for first 4 table entries)
28 *
29 * Data table packs both erf(x0)_high and a few bits of erf(x0)_low in one
30 * entry (in place of redundant exponent bits)
31 *
32 */
33

56ENTRY(_ZGVbN2v_erf_sse4)
57/*
58 * vector gather: erf(x0),
59 * second value is exp(-x0*x0)
60 */
61 lea __svml_derf_data_internal(%rip), %rcx
62 movups _AbsMask+__svml_derf_data_internal(%rip), %xmm5
63 andps %xmm0, %xmm5
64
65/*
66 * erf(x) rounds to 1.0 for x>_MaxThreshold (5.9921875)
67 * can compute all results in the main path
68 */
69 movaps %xmm5, %xmm9
70
71/* save sign */
72 pxor %xmm5, %xmm0
73 minpd _MaxThreshold+__svml_derf_data_internal(%rip), %xmm9
74 movups _SRound+__svml_derf_data_internal(%rip), %xmm1
75 movaps %xmm1, %xmm2
76 addpd %xmm9, %xmm2
77 movaps %xmm2, %xmm8
78 psllq $4, %xmm2
79 subpd %xmm1, %xmm8
80 movd %xmm2, %eax
81 movups _U2Threshold+__svml_derf_data_internal(%rip), %xmm11
82 cmpltpd %xmm9, %xmm11
83 subpd %xmm8, %xmm9
84 mulpd %xmm9, %xmm8
85
86/*
87 * _LA_ polynomial computation
88 * Start polynomial evaluation
89 */
90 movups _poly1_0+__svml_derf_data_internal(%rip), %xmm7
91 andps %xmm9, %xmm11
92 mulpd %xmm8, %xmm7
93
94/* D2 = Diff^2 */
95 mulpd %xmm11, %xmm11
96 addpd _poly1_1+__svml_derf_data_internal(%rip), %xmm7
97
98/* NaN fixup */
99 minpd %xmm5, %xmm9
100 mulpd %xmm8, %xmm7
101 movups _poly3_0+__svml_derf_data_internal(%rip), %xmm6
102
103/* T^2 */
104 movaps %xmm8, %xmm12
105 mulpd %xmm8, %xmm6
106 addpd _poly1_2+__svml_derf_data_internal(%rip), %xmm7
107 addpd _poly3_1+__svml_derf_data_internal(%rip), %xmm6
108 mulpd %xmm8, %xmm12
109 mulpd %xmm8, %xmm6
110 mulpd %xmm8, %xmm7
111 addpd _poly3_2+__svml_derf_data_internal(%rip), %xmm6
112 addpd _poly1_3+__svml_derf_data_internal(%rip), %xmm7
113 mulpd %xmm8, %xmm6
114
115/* P1 = T^2*P1 - T */
116 mulpd %xmm7, %xmm12
117 movups _poly5_0+__svml_derf_data_internal(%rip), %xmm10
118
119/* Sign | Diff */
120 pxor %xmm0, %xmm9
121 mulpd %xmm8, %xmm10
122 subpd %xmm8, %xmm12
123 addpd _poly5_1+__svml_derf_data_internal(%rip), %xmm10
124 mulpd %xmm11, %xmm10
125 addpd _poly3_3+__svml_derf_data_internal(%rip), %xmm10
126 addpd %xmm6, %xmm10
127 pshufd $2, %xmm2, %xmm3
128 movd %xmm3, %edx
129
130/* P1 + P3*D2 */
131 mulpd %xmm10, %xmm11
132 movslq %eax, %rax
133 movslq %edx, %rdx
134 addpd %xmm11, %xmm12
135 movups (%rcx,%rax), %xmm13
136 movups (%rcx,%rdx), %xmm4
137 movaps %xmm13, %xmm14
138 unpckhpd %xmm4, %xmm13
139
140/* exp_h(x0) * Diff */
141 mulpd %xmm9, %xmm13
142
143/*
144 * branch-free
145 * low part of result: exp_h(x0) * Diff*(1+P1)
146 */
147 mulpd %xmm13, %xmm12
148 addpd %xmm12, %xmm13
149 unpcklpd %xmm4, %xmm14
150
151/* Sign | _Erf_H */
152 pxor %xmm0, %xmm14
153
154/* Final result */
155 addpd %xmm13, %xmm14
156
157/* Fix erf(-0) = -0 */
158 orps %xmm14, %xmm0
159 ret
160
161END(_ZGVbN2v_erf_sse4)
