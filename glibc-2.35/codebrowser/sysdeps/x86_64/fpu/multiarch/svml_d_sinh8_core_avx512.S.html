19/*
20 * ALGORITHM DESCRIPTION:
21 *
22 * Compute sinh(x) as (exp(x)-exp(-x))/2,
23 * where exp is calculated as
24 * exp(M*ln2 + ln2*(j/2^k) + r) = 2^M * 2^(j/2^k) * exp(r)
25 *
26 * Special cases:
27 *
28 * sinh(NaN) = quiet NaN, and raise invalid exception
29 * sinh(INF) = that INF
30 * sinh(x) = x for subnormals
31 * sinh(x) overflows for big x and returns MAXLOG+log(2)
32 *
33 */
34

42#define _dbShifter 2304
43#define _iDomainRange 2368
44#define _dPC2 2432
45#define _dPC3 2496
46#define _dPC4 2560
47#define _dPC5 2624
48#define _lIndexMask 2688

54ENTRY(_ZGVeN8v_sinh_skx)
55 pushq %rbp
56 cfi_def_cfa_offset(16)
57 movq %rsp, %rbp
58 cfi_def_cfa(6, 16)
59 cfi_offset(6, -16)
60 andq $-64, %rsp
61 subq $192, %rsp
62 lea _dbT+8+__svml_dsinh_data_internal(%rip), %rax
63 vmovaps %zmm0, %zmm8
64
65/* Abs argument */
66 vandpd _dSign+__svml_dsinh_data_internal(%rip), %zmm8, %zmm7
67 vmovups _dbShifter+__svml_dsinh_data_internal(%rip), %zmm13
68
69/*
70 * Load argument
71 * dM = x*2^K/log(2) + RShifter
72 */
73 vmovups _dbInvLn2+__svml_dsinh_data_internal(%rip), %zmm12
74 vmovups _dbLn2hi+__svml_dsinh_data_internal(%rip), %zmm14
75 vmovups _dPC5+__svml_dsinh_data_internal(%rip), %zmm6
76
77/* VLOAD_CONST( D, dPC[0], TAB._dPC1 ); */
78 vmovups _dPC4+__svml_dsinh_data_internal(%rip), %zmm4
79 vxorpd %zmm8, %zmm7, %zmm5
80 kxnorw %k0, %k0, %k1
81 kxnorw %k0, %k0, %k2
82 vfmadd213pd {rn-sae}, %zmm13, %zmm5, %zmm12
83
84/*
85 * Check for overflow\underflow
86 *
87 */
88 vpsrlq $32, %zmm5, %zmm9
89
90/*
91 * R
92 * dN = dM - RShifter
93 */
94 vsubpd {rn-sae}, %zmm13, %zmm12, %zmm2
95 vpmovqd %zmm9, %ymm10
96 vmovups _dbLn2lo+__svml_dsinh_data_internal(%rip), %zmm9
97
98/* dR = dX - dN*Log2_hi/2^K */
99 vfnmadd231pd {rn-sae}, %zmm14, %zmm2, %zmm5
100
101/*
102 * sinh(r) = r*((a1=1)+r^2*(a3+r^2*a5)) = r + r*(r^2*(a3+r^2*a5)) ....
103 * dSinh_r = (a3+r^2*a5)
104 */
105 vmovups _dPC3+__svml_dsinh_data_internal(%rip), %zmm14
106
107/* dR = (dX - dN*Log2_hi/2^K) - dN*Log2_lo/2^K */
108 vfnmadd231pd {rn-sae}, %zmm9, %zmm2, %zmm5
109 vpcmpgtd _iDomainRange+__svml_dsinh_data_internal(%rip), %ymm10, %ymm11
110 vmovmskps %ymm11, %edx
111
112/* dR2 = dR^2 */
113 vmulpd {rn-sae}, %zmm5, %zmm5, %zmm2
114 vfmadd231pd {rn-sae}, %zmm2, %zmm6, %zmm14
115
116/*
117 * Index and lookup
118 * j
119 */
120 vpandq _lIndexMask+__svml_dsinh_data_internal(%rip), %zmm12, %zmm15
121 vpsllq $4, %zmm15, %zmm1
122 vpmovqd %zmm1, %ymm0
123 vpxord %zmm11, %zmm11, %zmm11
124 vpxord %zmm10, %zmm10, %zmm10
125 vgatherdpd (%rax,%ymm0), %zmm11{%k1}
126 vgatherdpd -8(%rax,%ymm0), %zmm10{%k2}
127
128/* split j and N */
129 vpxorq %zmm15, %zmm12, %zmm3
130
131/*
132 * G1,G2,G3: dTdif,dTn * 2^N,2^(-N)
133 * lM now is an EXP(2^N)
134 */
135 vpsllq $45, %zmm3, %zmm3
136 vpaddq %zmm3, %zmm10, %zmm1
137
138/* */
139 vpaddq %zmm3, %zmm11, %zmm12
140
141/* */
142 vpsubq %zmm3, %zmm11, %zmm13
143
144/* dSinh_r = r^2*(a3+r^2*a5) */
145 vmulpd {rn-sae}, %zmm2, %zmm14, %zmm3
146
147/* dG2 = dTn*2^N - dTn*2^-N */
148 vsubpd {rn-sae}, %zmm13, %zmm12, %zmm15
149
150/* dG3 = dTn*2^N + dTn*2^-N */
151 vaddpd {rn-sae}, %zmm13, %zmm12, %zmm0
152
153/* dSinh_r = r + r*(r^2*(a3+r^2*a5)) */
154 vfmadd213pd {rn-sae}, %zmm5, %zmm5, %zmm3
155
156/*
157 * poly(r) = (dG2+dG1)+dG3*sinh(dR)+dG1*sinh(dR)+(dG1+dG2)*dR2*(a2 +a4*dR2)
158 * dOut = (a2 +a4*dR2)
159 */
160 vmovups _dPC2+__svml_dsinh_data_internal(%rip), %zmm5
161
162/* dG1 += dG3 */
163 vaddpd {rn-sae}, %zmm0, %zmm1, %zmm6
164 vfmadd231pd {rn-sae}, %zmm2, %zmm4, %zmm5
165
166/* dOut = dR2*(a2 +a4*dR2) */
167 vmulpd {rn-sae}, %zmm2, %zmm5, %zmm4
168
169/* dG2 += dG1 */
170 vaddpd {rn-sae}, %zmm15, %zmm1, %zmm2
171
172/* dOut = dG2*dR2*(a2 +a4*dR2) */
173 vmulpd {rn-sae}, %zmm2, %zmm4, %zmm4
174
175/* dOut = dG1*sinh(dR)+dG2*dR2*(a2 +a4*dR2) */
176 vfmadd213pd {rn-sae}, %zmm4, %zmm6, %zmm3
177
178/* dOut = dG2 + dG1*sinh(dR)+dG2*dR2*(a2 +a4*dR2) */
179 vaddpd {rn-sae}, %zmm2, %zmm3, %zmm0
180
181/* Ret H */
182 vorpd %zmm0, %zmm7, %zmm0
183 testl %edx, %edx
184
185/* Go to special inputs processing branch */
186 jne L(SPECIAL_VALUES_BRANCH)
187 # LOE rbx r12 r13 r14 r15 edx zmm0 zmm8
188
189/* Restore registers
190 * and exit the function
191 */
192
193L(EXIT):
194 movq %rbp, %rsp
195 popq %rbp
196 cfi_def_cfa(7, 8)
197 cfi_restore(6)
198 ret
199 cfi_def_cfa(6, 16)
200 cfi_offset(6, -16)
201
202/* Branch to process
203 * special inputs
204 */
205
206L(SPECIAL_VALUES_BRANCH):
207 vmovups %zmm8, 64(%rsp)
208 vmovups %zmm0, 128(%rsp)
209 # LOE rbx r12 r13 r14 r15 edx zmm0
210
211 xorl %eax, %eax
212 # LOE rbx r12 r13 r14 r15 eax edx
213
214 vzeroupper
215 movq %r12, 16(%rsp)
216 /* DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -176; DW_OP_plus) */
217 .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x50, 0xff, 0xff, 0xff, 0x22
218 movl %eax, %r12d
219 movq %r13, 8(%rsp)
220 /* DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -184; DW_OP_plus) */
221 .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x48, 0xff, 0xff, 0xff, 0x22
222 movl %edx, %r13d
223 movq %r14, (%rsp)
224 /* DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -192; DW_OP_plus) */
225 .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x40, 0xff, 0xff, 0xff, 0x22
226 # LOE rbx r15 r12d r13d
227
228/* Range mask
229 * bits check
230 */
231
232L(RANGEMASK_CHECK):
233 btl %r12d, %r13d
234
235/* Call scalar math function */
236 jc L(SCALAR_MATH_CALL)
237 # LOE rbx r15 r12d r13d
238
239/* Special inputs
240 * processing loop
241 */
242
243L(SPECIAL_VALUES_LOOP):
244 incl %r12d
245 cmpl $8, %r12d
246
247/* Check bits in range mask */
248 jl L(RANGEMASK_CHECK)
249 # LOE rbx r15 r12d r13d
250
251 movq 16(%rsp), %r12
252 cfi_restore(12)
253 movq 8(%rsp), %r13
254 cfi_restore(13)
255 movq (%rsp), %r14
256 cfi_restore(14)
257 vmovups 128(%rsp), %zmm0
258
259/* Go to exit */
260 jmp L(EXIT)
261 /* DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -176; DW_OP_plus) */
262 .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x50, 0xff, 0xff, 0xff, 0x22
263 /* DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -184; DW_OP_plus) */
264 .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x48, 0xff, 0xff, 0xff, 0x22
265 /* DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -192; DW_OP_plus) */
266 .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x40, 0xff, 0xff, 0xff, 0x22
267 # LOE rbx r12 r13 r14 r15 zmm0
268
269/* Scalar math fucntion call
270 * to process special input
271 */
272
273L(SCALAR_MATH_CALL):
274 movl %r12d, %r14d
275 movsd 64(%rsp,%r14,8), %xmm0
276 call sinh@PLT
277 # LOE rbx r14 r15 r12d r13d xmm0
278
279 movsd %xmm0, 128(%rsp,%r14,8)
280
281/* Process special inputs in loop */
282 jmp L(SPECIAL_VALUES_LOOP)
283 # LOE rbx r15 r12d r13d
284END(_ZGVeN8v_sinh_skx)
