19/*
20 * ALGORITHM DESCRIPTION:
21 *
22 * Compute acosh(x) as log(x + sqrt(x*x - 1))
23 *
24 * Special cases:
25 *
26 * acosh(NaN) = quiet NaN, and raise invalid exception
27 * acosh(-INF) = NaN
28 * acosh(+INF) = +INF
29 * acosh(x) = NaN if x < 1
30 * acosh(1) = +0
31 *
32 */
33
34/* Offsets for data table __svml_sacosh_data_internal
35 */
36#define sOne 0
37#define sPoly 32
38#define iBrkValue 288
39#define iOffExpoMask 320
40#define sBigThreshold 352
41#define sC2 384
42#define sC3 416
43#define sHalf 448
44#define sLargestFinite 480
45#define sThirtyOne 512
46#define sTopMask8 544
47#define XScale 576
48#define sLn2 608

50#include <sysdep.h>

52 .text
53 .section .text.avx2,"ax",@progbits
54ENTRY(_ZGVdN8v_acoshf_avx2)
55 pushq %rbp
56 cfi_def_cfa_offset(16)
57 movq %rsp, %rbp
58 cfi_def_cfa(6, 16)
59 cfi_offset(6, -16)
60 andq $-32, %rsp
61 subq $96, %rsp
62
63/* Load constants, always including One = 1 */
64 vmovups sOne+__svml_sacosh_data_internal(%rip), %ymm2
65
66/* Finally, express Y + W = U * V accurately where Y has <= 8 bits */
67 vmovups sTopMask8+__svml_sacosh_data_internal(%rip), %ymm9
68
69/*
70 * Now 1 / (1 + d)
71 * = 1 / (1 + (sqrt(1 - e) - 1))
72 * = 1 / sqrt(1 - e)
73 * = 1 + 1/2 * e + 3/8 * e^2 + 5/16 * e^3 + 35/128 * e^4 + ...
74 * So compute the first three nonconstant terms of that, so that
75 * we have a relative correction (1 + Corr) to apply to S etc.
76 * C1 = 1/2
77 * C2 = 3/8
78 * C3 = 5/16
79 */
80 vmovups sC3+__svml_sacosh_data_internal(%rip), %ymm14
81 vmovaps %ymm0, %ymm3
82 vmovaps %ymm2, %ymm7
83 vfmsub231ps %ymm3, %ymm3, %ymm7
84
85/*
86 * Check that 1 < X < +inf; otherwise go to the callout function.
87 * We need the callout for X = 1 to avoid division by zero below.
88 * This test ensures that callout handles NaN and either infinity.
89 */
90 vcmpnle_uqps sLargestFinite+__svml_sacosh_data_internal(%rip), %ymm3, %ymm4
91 vcmpngt_uqps %ymm2, %ymm3, %ymm5
92
93/*
94 * The following computation can go wrong for very large X, e.g.
95 * the X^2 - 1 = U * V can overflow. But for large X we have
96 * acosh(X) / log(2 X) - 1 =~= 1/(4 * X^2), so for X >= 2^30
97 * we can just later stick X back into the log and tweak up the exponent.
98 * Actually we scale X by 2^-30 and tweak the exponent up by 31,
99 * to stay in the safe range for the later log computation.
100 * Compute a flag now telling us when to do this.
101 */
102 vcmplt_oqps sBigThreshold+__svml_sacosh_data_internal(%rip), %ymm3, %ymm1
103 vandps %ymm9, %ymm7, %ymm10
104
105/*
106 * Compute R = 1/sqrt(Y + W) * (1 + d)
107 * Force R to <= 8 significant bits.
108 * This means that R * Y and R^2 * Y are exactly representable.
109 */
110 vrsqrtps %ymm10, %ymm8
111 vsubps %ymm10, %ymm7, %ymm11
112 vandps %ymm9, %ymm8, %ymm12
113
114/*
115 * Compute S = (Y/sqrt(Y + W)) * (1 + d)
116 * and T = (W/sqrt(Y + W)) * (1 + d)
117 * so that S + T = sqrt(Y + W) * (1 + d)
118 * S is exact, and the rounding error in T is OK.
119 */
120 vmulps %ymm12, %ymm10, %ymm15
121 vmulps %ymm11, %ymm12, %ymm0
122
123/* Now multiplex to the case X = 2^-30 * input, Xl = 0 in the "big" case. */
124 vmulps XScale+__svml_sacosh_data_internal(%rip), %ymm3, %ymm11
125
126/*
127 * Compute e = -(2 * d + d^2)
128 * The first FMR is exact, and the rounding error in the other is acceptable
129 * since d and e are ~ 2^-8
130 */
131 vmovaps %ymm2, %ymm13
132 vfnmadd231ps %ymm15, %ymm12, %ymm13
133 vfnmadd231ps %ymm0, %ymm12, %ymm13
134 vfmadd213ps sC2+__svml_sacosh_data_internal(%rip), %ymm13, %ymm14
135 vfmadd213ps sHalf+__svml_sacosh_data_internal(%rip), %ymm13, %ymm14
136 vmulps %ymm14, %ymm13, %ymm7
137 vorps %ymm5, %ymm4, %ymm6
138
139/*
140 * For low-accuracy versions, the computation can be done
141 * just as U + ((S + T) + (S + T) * Corr)
142 */
143 vaddps %ymm0, %ymm15, %ymm5
144
145/* sU is needed later on */
146 vsubps %ymm2, %ymm3, %ymm4
147 vfmadd213ps %ymm5, %ymm7, %ymm5
148 vmovmskps %ymm6, %edx
149 vaddps %ymm5, %ymm4, %ymm6
150
151/*
152 * Now resume the main code.
153 * reduction: compute r,n
154 */
155 vmovups iBrkValue+__svml_sacosh_data_internal(%rip), %ymm4
156
157/*
158 * Now we feed into the log1p code, using H in place of _VARG1 and
159 * also adding L into Xl.
160 * compute 1+x as high, low parts
161 */
162 vmaxps %ymm6, %ymm2, %ymm8
163 vminps %ymm6, %ymm2, %ymm9
164 vaddps %ymm9, %ymm8, %ymm12
165 vblendvps %ymm1, %ymm12, %ymm11, %ymm14
166 vsubps %ymm12, %ymm8, %ymm10
167 vpsubd %ymm4, %ymm14, %ymm15
168 vaddps %ymm10, %ymm9, %ymm13
169 vpand iOffExpoMask+__svml_sacosh_data_internal(%rip), %ymm15, %ymm14
170 vpsrad $23, %ymm15, %ymm15
171 vpaddd %ymm4, %ymm14, %ymm8
172 vpslld $23, %ymm15, %ymm5
173 vmovups sPoly+224+__svml_sacosh_data_internal(%rip), %ymm4
174 vcvtdq2ps %ymm15, %ymm0
175 vpsubd %ymm5, %ymm2, %ymm7
176
177/* polynomial evaluation */
178 vsubps %ymm2, %ymm8, %ymm2
179
180/* Add 31 to the exponent in the "large" case to get log(2 * input) */
181 vaddps sThirtyOne+__svml_sacosh_data_internal(%rip), %ymm0, %ymm5
182 vandps %ymm1, %ymm13, %ymm6
183 vmulps %ymm7, %ymm6, %ymm9
184 vblendvps %ymm1, %ymm0, %ymm5, %ymm0
185 vaddps %ymm2, %ymm9, %ymm2
186 vfmadd213ps sPoly+192+__svml_sacosh_data_internal(%rip), %ymm2, %ymm4
187 vfmadd213ps sPoly+160+__svml_sacosh_data_internal(%rip), %ymm2, %ymm4
188 vfmadd213ps sPoly+128+__svml_sacosh_data_internal(%rip), %ymm2, %ymm4
189 vfmadd213ps sPoly+96+__svml_sacosh_data_internal(%rip), %ymm2, %ymm4
190 vfmadd213ps sPoly+64+__svml_sacosh_data_internal(%rip), %ymm2, %ymm4
191 vfmadd213ps sPoly+32+__svml_sacosh_data_internal(%rip), %ymm2, %ymm4
192 vfmadd213ps sPoly+__svml_sacosh_data_internal(%rip), %ymm2, %ymm4
193 vmulps %ymm4, %ymm2, %ymm6
194 vfmadd213ps %ymm2, %ymm2, %ymm6
195
196/* final reconstruction */
197 vfmadd132ps sLn2+__svml_sacosh_data_internal(%rip), %ymm6, %ymm0
198 testl %edx, %edx
199
200/* Go to special inputs processing branch */
201 jne L(SPECIAL_VALUES_BRANCH)
202 # LOE rbx r12 r13 r14 r15 edx ymm0 ymm3
203
204/* Restore registers
205 * and exit the function
206 */
207
208L(EXIT):
209 movq %rbp, %rsp
210 popq %rbp
211 cfi_def_cfa(7, 8)
212 cfi_restore(6)
213 ret
214 cfi_def_cfa(6, 16)
215 cfi_offset(6, -16)
216
217/* Branch to process
218 * special inputs
219 */
220
221L(SPECIAL_VALUES_BRANCH):
222 vmovups %ymm3, 32(%rsp)
223 vmovups %ymm0, 64(%rsp)
224 # LOE rbx r12 r13 r14 r15 edx ymm0
225
226 xorl %eax, %eax
227 # LOE rbx r12 r13 r14 r15 eax edx
228
229 vzeroupper
230 movq %r12, 16(%rsp)
231 /* DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -80; DW_OP_plus) */
232 .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xb0, 0xff, 0xff, 0xff, 0x22
233 movl %eax, %r12d
234 movq %r13, 8(%rsp)
235 /* DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -88; DW_OP_plus) */
236 .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xa8, 0xff, 0xff, 0xff, 0x22
237 movl %edx, %r13d
238 movq %r14, (%rsp)
239 /* DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -96; DW_OP_plus) */
240 .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xa0, 0xff, 0xff, 0xff, 0x22
241 # LOE rbx r15 r12d r13d
242
243/* Range mask
244 * bits check
245 */
246
247L(RANGEMASK_CHECK):
248 btl %r12d, %r13d
249
250/* Call scalar math function */
251 jc L(SCALAR_MATH_CALL)
252 # LOE rbx r15 r12d r13d
253
254/* Special inputs
255 * processing loop
256 */
257
258L(SPECIAL_VALUES_LOOP):
259 incl %r12d
260 cmpl $8, %r12d
261
262/* Check bits in range mask */
263 jl L(RANGEMASK_CHECK)
264 # LOE rbx r15 r12d r13d
265
266 movq 16(%rsp), %r12
267 cfi_restore(12)
268 movq 8(%rsp), %r13
269 cfi_restore(13)
270 movq (%rsp), %r14
271 cfi_restore(14)
272 vmovups 64(%rsp), %ymm0
273
274/* Go to exit */
275 jmp L(EXIT)
276 /* DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -80; DW_OP_plus) */
277 .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xb0, 0xff, 0xff, 0xff, 0x22
278 /* DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -88; DW_OP_plus) */
279 .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xa8, 0xff, 0xff, 0xff, 0x22
280 /* DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -96; DW_OP_plus) */
281 .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xa0, 0xff, 0xff, 0xff, 0x22
282 # LOE rbx r12 r13 r14 r15 ymm0
283
284/* Scalar math fucntion call
285 * to process special input
286 */
287
288L(SCALAR_MATH_CALL):
289 movl %r12d, %r14d
290 movss 32(%rsp,%r14,4), %xmm0
291 call acoshf@PLT
292 # LOE rbx r14 r15 r12d r13d xmm0
293
294 movss %xmm0, 64(%rsp,%r14,4)
295
296/* Process special inputs in loop */
297 jmp L(SPECIAL_VALUES_LOOP)
298 # LOE rbx r15 r12d r13d
