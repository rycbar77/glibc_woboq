19/*
20 * ALGORITHM DESCRIPTION:
21 *
22 * Compute asinh(x) as log(x + sqrt(x*x + 1))
23 * using RSQRT instructions for starting the
24 * square root approximation, and small table lookups for log
25 * that map to AVX-512 permute instructions
26 *
27 * Special cases:
28 *
29 * asinh(NaN) = quiet NaN, and raise invalid exception
30 * asinh(INF) = that INF
31 * asinh(0) = that 0
32 *
33 */
34
35/* Offsets for data table __svml_sasinh_data_internal_avx512
36 */
37#define Log_tbl_H 0
38#define Log_tbl_L 128
39#define One 256
40#define AbsMask 320
41#define SmallThreshold 384
42#define Threshold 448
43#define LargeThreshold 512
44#define ca1 576
45#define c2s 640
46#define c1s 704
47#define AddB5 768
48#define RcpBitMask 832
49#define OneEighth 896
50#define Four 960
51#define poly_coeff3 1024
52#define poly_coeff2 1088
53#define poly_coeff1 1152
54#define L2H 1216
55#define L2L 1280

57#include <sysdep.h>

59 .text
60 .section .text.exex512,"ax",@progbits
61ENTRY(_ZGVeN16v_asinhf_skx)
62 pushq %rbp
63 cfi_def_cfa_offset(16)
64 movq %rsp, %rbp
65 cfi_def_cfa(6, 16)
66 cfi_offset(6, -16)
67 andq $-64, %rsp
68 subq $192, %rsp
69 vmovaps %zmm0, %zmm10
70
71/* x^2 */
72 vmulps {rn-sae}, %zmm10, %zmm10, %zmm0
73 vmovups One+__svml_sasinh_data_internal_avx512(%rip), %zmm2
74
75/* polynomial computation for small inputs */
76 vmovups ca1+__svml_sasinh_data_internal_avx512(%rip), %zmm1
77
78/* not a very small input ? */
79 vmovups SmallThreshold+__svml_sasinh_data_internal_avx512(%rip), %zmm11
80
81/* 1+x^2 */
82 vaddps {rn-sae}, %zmm2, %zmm0, %zmm7
83
84/* |input| */
85 vandps AbsMask+__svml_sasinh_data_internal_avx512(%rip), %zmm10, %zmm12
86
87/* A=max(x^2, 1); */
88 vmaxps {sae}, %zmm0, %zmm2, %zmm14
89 vrsqrt14ps %zmm7, %zmm8
90
91/* B=min(x^2, 1); */
92 vminps {sae}, %zmm0, %zmm2, %zmm15
93 vcmpps $21, {sae}, %zmm11, %zmm12, %k2
94
95/* B_high */
96 vsubps {rn-sae}, %zmm14, %zmm7, %zmm9
97
98/* sign bit */
99 vxorps %zmm10, %zmm12, %zmm13
100
101/* Sh ~sqrt(1+x^2) */
102 vmulps {rn-sae}, %zmm8, %zmm7, %zmm6
103 vmovups LargeThreshold+__svml_sasinh_data_internal_avx512(%rip), %zmm14
104
105/* B_low */
106 vsubps {rn-sae}, %zmm9, %zmm15, %zmm3
107
108/* Sh+x */
109 vaddps {rn-sae}, %zmm12, %zmm6, %zmm15
110
111/* (Yh*R0)_low */
112 vfmsub213ps {rn-sae}, %zmm6, %zmm8, %zmm7
113 vmulps {rn-sae}, %zmm1, %zmm0, %zmm9
114 vcmpps $22, {sae}, %zmm14, %zmm12, %k0
115 vmovups c1s+__svml_sasinh_data_internal_avx512(%rip), %zmm1
116
117/* polynomial computation for small inputs */
118 vfmadd213ps {rn-sae}, %zmm12, %zmm12, %zmm9
119 kmovw %k0, %edx
120
121/* (x^2)_low */
122 vmovaps %zmm10, %zmm4
123 vfmsub213ps {rn-sae}, %zmm0, %zmm10, %zmm4
124
125/* Yl = (x^2)_low + B_low */
126 vaddps {rn-sae}, %zmm4, %zmm3, %zmm5
127
128/* rel. error term: Eh=1-Sh*R0 */
129 vmovaps %zmm2, %zmm0
130 vfnmadd231ps {rn-sae}, %zmm6, %zmm8, %zmm0
131
132/* Sl = (Yh*R0)_low+(R0*Yl) */
133 vfmadd213ps {rn-sae}, %zmm7, %zmm8, %zmm5
134
135/* very large inputs ? */
136 vmovups Threshold+__svml_sasinh_data_internal_avx512(%rip), %zmm7
137
138/* rel. error term: Eh=(1-Sh*R0)-Sl*R0 */
139 vfnmadd231ps {rn-sae}, %zmm5, %zmm8, %zmm0
140
141/* sqrt(1+x^2) ~ Sh + Sl + Sh*Eh*poly_s */
142 vmovups c2s+__svml_sasinh_data_internal_avx512(%rip), %zmm8
143 vcmpps $21, {sae}, %zmm7, %zmm12, %k1
144
145/* Sh*Eh */
146 vmulps {rn-sae}, %zmm0, %zmm6, %zmm4
147 vfmadd231ps {rn-sae}, %zmm0, %zmm8, %zmm1
148
149/* Sl + Sh*Eh*poly_s */
150 vfmadd213ps {rn-sae}, %zmm5, %zmm1, %zmm4
151
152/* Xh */
153 vsubps {rn-sae}, %zmm6, %zmm15, %zmm5
154
155/* fixup for very large inputs */
156 vmovups OneEighth+__svml_sasinh_data_internal_avx512(%rip), %zmm6
157
158/* Xin0+Sl+Sh*Eh*poly_s ~ x+sqrt(1+x^2) */
159 vaddps {rn-sae}, %zmm4, %zmm15, %zmm3
160
161/* Xl */
162 vsubps {rn-sae}, %zmm5, %zmm12, %zmm5
163
164/* Sl_high */
165 vsubps {rn-sae}, %zmm15, %zmm3, %zmm0
166 vmulps {rn-sae}, %zmm6, %zmm12, %zmm3{%k1}
167
168/* -K*L2H + Th */
169 vmovups L2H+__svml_sasinh_data_internal_avx512(%rip), %zmm15
170
171/* Sl_l */
172 vsubps {rn-sae}, %zmm0, %zmm4, %zmm1
173 vrcp14ps %zmm3, %zmm6
174
175/* Table lookups */
176 vmovups __svml_sasinh_data_internal_avx512(%rip), %zmm0
177
178/* Xin_low */
179 vaddps {rn-sae}, %zmm5, %zmm1, %zmm7
180
181/* round reciprocal to 1+4b mantissas */
182 vpaddd AddB5+__svml_sasinh_data_internal_avx512(%rip), %zmm6, %zmm4
183 vmovups poly_coeff1+__svml_sasinh_data_internal_avx512(%rip), %zmm5
184 vandps RcpBitMask+__svml_sasinh_data_internal_avx512(%rip), %zmm4, %zmm8
185
186/* fixup for very large inputs */
187 vxorps %zmm7, %zmm7, %zmm7{%k1}
188
189/* polynomial */
190 vmovups poly_coeff3+__svml_sasinh_data_internal_avx512(%rip), %zmm4
191
192/* reduced argument for log(): (Rcp*Xin-1)+Rcp*Xin_low */
193 vfmsub231ps {rn-sae}, %zmm8, %zmm3, %zmm2
194 vmovups Four+__svml_sasinh_data_internal_avx512(%rip), %zmm3
195
196/* exponents */
197 vgetexpps {sae}, %zmm8, %zmm1
198
199/* Prepare table index */
200 vpsrld $18, %zmm8, %zmm14
201 vfmadd231ps {rn-sae}, %zmm8, %zmm7, %zmm2
202 vmovups poly_coeff2+__svml_sasinh_data_internal_avx512(%rip), %zmm7
203 vsubps {rn-sae}, %zmm3, %zmm1, %zmm1{%k1}
204 vpermt2ps Log_tbl_H+64+__svml_sasinh_data_internal_avx512(%rip), %zmm14, %zmm0
205 vmovups Log_tbl_L+__svml_sasinh_data_internal_avx512(%rip), %zmm3
206 vfmadd231ps {rn-sae}, %zmm2, %zmm4, %zmm7
207 vfnmadd231ps {rn-sae}, %zmm1, %zmm15, %zmm0
208
209/* R^2 */
210 vmulps {rn-sae}, %zmm2, %zmm2, %zmm6
211 vfmadd213ps {rn-sae}, %zmm5, %zmm2, %zmm7
212 vpermt2ps Log_tbl_L+64+__svml_sasinh_data_internal_avx512(%rip), %zmm14, %zmm3
213
214/* -K*L2L + Tl */
215 vmovups L2L+__svml_sasinh_data_internal_avx512(%rip), %zmm14
216 vfnmadd213ps {rn-sae}, %zmm3, %zmm14, %zmm1
217
218/* Tl + R^2*Poly */
219 vfmadd213ps {rn-sae}, %zmm1, %zmm6, %zmm7
220
221/* R+Tl + R^2*Poly */
222 vaddps {rn-sae}, %zmm2, %zmm7, %zmm2
223 vaddps {rn-sae}, %zmm2, %zmm0, %zmm9{%k2}
224 vxorps %zmm13, %zmm9, %zmm0
225 testl %edx, %edx
226
227/* Go to special inputs processing branch */
228 jne L(SPECIAL_VALUES_BRANCH)
229 # LOE rbx r12 r13 r14 r15 edx zmm0 zmm10
230
231/* Restore registers
232 * and exit the function
233 */
234
235L(EXIT):
236 movq %rbp, %rsp
237 popq %rbp
238 cfi_def_cfa(7, 8)
239 cfi_restore(6)
240 ret
241 cfi_def_cfa(6, 16)
242 cfi_offset(6, -16)
243
244/* Branch to process
245 * special inputs
246 */
247
248L(SPECIAL_VALUES_BRANCH):
249 vmovups %zmm10, 64(%rsp)
250 vmovups %zmm0, 128(%rsp)
251 # LOE rbx r12 r13 r14 r15 edx zmm0
252
253 xorl %eax, %eax
254 # LOE rbx r12 r13 r14 r15 eax edx
255
256 vzeroupper
257 movq %r12, 16(%rsp)
258 /* DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -176; DW_OP_plus) */
259 .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x50, 0xff, 0xff, 0xff, 0x22
260 movl %eax, %r12d
261 movq %r13, 8(%rsp)
262 /* DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -184; DW_OP_plus) */
263 .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x48, 0xff, 0xff, 0xff, 0x22
264 movl %edx, %r13d
265 movq %r14, (%rsp)
266 /* DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -192; DW_OP_plus) */
267 .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x40, 0xff, 0xff, 0xff, 0x22
268 # LOE rbx r15 r12d r13d
269
270/* Range mask
271 * bits check
272 */
273
274L(RANGEMASK_CHECK):
275 btl %r12d, %r13d
276
277/* Call scalar math function */
278 jc L(SCALAR_MATH_CALL)
279 # LOE rbx r15 r12d r13d
280
281/* Special inputs
282 * processing loop
283 */
284
285L(SPECIAL_VALUES_LOOP):
286 incl %r12d
287 cmpl $16, %r12d
288
289/* Check bits in range mask */
290 jl L(RANGEMASK_CHECK)
291 # LOE rbx r15 r12d r13d
292
293 movq 16(%rsp), %r12
294 cfi_restore(12)
295 movq 8(%rsp), %r13
296 cfi_restore(13)
297 movq (%rsp), %r14
298 cfi_restore(14)
299 vmovups 128(%rsp), %zmm0
300
301/* Go to exit */
302 jmp L(EXIT)
303 /* DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -176; DW_OP_plus) */
304 .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x50, 0xff, 0xff, 0xff, 0x22
305 /* DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -184; DW_OP_plus) */
306 .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x48, 0xff, 0xff, 0xff, 0x22
307 /* DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -192; DW_OP_plus) */
308 .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x40, 0xff, 0xff, 0xff, 0x22
309 # LOE rbx r12 r13 r14 r15 zmm0
310
311/* Scalar math fucntion call
312 * to process special input
313 */
314
315L(SCALAR_MATH_CALL):
316 movl %r12d, %r14d
317 movss 64(%rsp,%r14,4), %xmm0
318 call asinhf@PLT
319 # LOE rbx r14 r15 r12d r13d xmm0
320
321 movss %xmm0, 128(%rsp,%r14,4)
322
323/* Process special inputs in loop */
324 jmp L(SPECIAL_VALUES_LOOP)
325 # LOE rbx r15 r12d r13d
326END(_ZGVeN16v_asinhf_skx)
