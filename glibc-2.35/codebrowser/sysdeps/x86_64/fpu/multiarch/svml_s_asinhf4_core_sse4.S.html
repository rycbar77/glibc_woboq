19/*
20 * ALGORITHM DESCRIPTION:
21 *
22 * Compute asinh(x) as log(x + sqrt(x*x + 1))
23 *
24 * Special cases:
25 *
26 * asinh(NaN) = quiet NaN, and raise invalid exception
27 * asinh(INF) = that INF
28 * asinh(0) = that 0
29 *
30 */
31
32/* Offsets for data table __svml_sasinh_data_internal
33 */
34#define SgnMask 0
35#define sOne 16
36#define sPoly 32
37#define iBrkValue 160
38#define iOffExpoMask 176
39#define sBigThreshold 192
40#define sC2 208
41#define sC3 224
42#define sHalf 240
43#define sLargestFinite 256
44#define sLittleThreshold 272
45#define sSign 288
46#define sThirtyOne 304
47#define sTopMask11 320
48#define sTopMask8 336
49#define XScale 352
50#define sLn2 368
51
52#include <sysdep.h>
53
54 .text
55 .section .text.sse4,"ax",@progbits
56ENTRY(_ZGVbN4v_asinhf_sse4)
57 subq $72, %rsp
58 cfi_def_cfa_offset(80)
59 movaps %xmm0, %xmm8
60
61/*
62 * Split X into high and low parts, XHi (<= 11 bits) and XLo (<= 13 bits)
63 * We could use either X or |X| here, but it doesn't seem to matter
64 */
65 movups sTopMask11+__svml_sasinh_data_internal(%rip), %xmm10
66 movaps %xmm8, %xmm2
67 andps %xmm8, %xmm10
68
69/*
70 * Compute X^2 = (XHi + XLo)^2 = XHi^2 + XLo * (X + XHi)
71 * The two parts are shifted off by around 11 bits. So even though
72 * the low bit will not in general be exact, it's near enough
73 */
74 movaps %xmm10, %xmm3
75 subps %xmm10, %xmm2
76 mulps %xmm10, %xmm3
77 addps %xmm8, %xmm10
78
79/* Load the constant 1 and a sign mask */
80 movups sOne+__svml_sasinh_data_internal(%rip), %xmm7
81
82/*
83 * Finally, express Y + W = X^2 + 1 accurately where Y has <= 8 bits.
84 * If |X| <= 1 then |XHi| <= 1 and so |X2Hi| <= 1, so we can treat 1
85 * as the dominant component in the compensated summation. Otherwise,
86 * if |X| >= 1, then since X2Hi only has 22 significant bits, the basic
87 * addition will be exact anyway until we get to |X| >= 2^24. But by
88 * that time the log function is well-conditioned enough that the
89 * rounding error doesn't matter. Hence we can treat 1 as dominant even
90 * if it literally isn't.
91 */
92 movaps %xmm7, %xmm11
93 movaps %xmm7, %xmm4
94 movups sTopMask8+__svml_sasinh_data_internal(%rip), %xmm12
95 addps %xmm3, %xmm11
96 mulps %xmm10, %xmm2
97 subps %xmm11, %xmm4
98 movaps %xmm12, %xmm0
99 addps %xmm3, %xmm4
100
101/*
102 * Unfortunately, we can still be in trouble if |X| <= 2^-5, since
103 * the absolute error 2^-(7+24)-ish in sqrt(1 + X^2) gets scaled up
104 * by 1/X and comes close to our threshold. Hence if |X| <= 2^-4,
105 * perform an alternative computation
106 * sqrt(1 + X^2) - 1 = X^2/2 - X^4/8 + X^6/16
107 * X2 = X^2
108 */
109 addps %xmm2, %xmm3
110 addps %xmm2, %xmm4
111 andps %xmm11, %xmm0
112
113/*
114 * Compute R = 1/sqrt(Y + W) * (1 + d)
115 * Force R to <= 8 significant bits.
116 * This means that R * Y and R^2 * Y are exactly representable.
117 */
118 rsqrtps %xmm0, %xmm14
119 subps %xmm0, %xmm11
120 andps %xmm12, %xmm14
121 addps %xmm11, %xmm4
122
123/*
124 * Compute S = (Y/sqrt(Y + W)) * (1 + d)
125 * and T = (W/sqrt(Y + W)) * (1 + d)
126 * so that S + T = sqrt(Y + W) * (1 + d)
127 * S is exact, and the rounding error in T is OK.
128 */
129 mulps %xmm14, %xmm0
130 mulps %xmm14, %xmm4
131
132/*
133 * Get the absolute value of the input, since we will exploit antisymmetry
134 * and mostly assume X >= 0 in the core computation
135 */
136 movups SgnMask+__svml_sasinh_data_internal(%rip), %xmm6
137
138/*
139 * Compute e = -(2 * d + d^2)
140 * The first FMR is exact, and the rounding error in the other is acceptable
141 * since d and e are ~ 2^-8
142 */
143 movaps %xmm14, %xmm13
144 andps %xmm8, %xmm6
145
146/*
147 * Obtain sqrt(1 + X^2) - 1 in two pieces
148 * sqrt(1 + X^2) - 1
149 * = sqrt(Y + W) - 1
150 * = (S + T) * (1 + Corr) - 1
151 * = [S - 1] + [T + (S + T) * Corr]
152 * We need a compensated summation for the last part. We treat S - 1
153 * as the larger part; it certainly is until about X < 2^-4, and in that
154 * case, the error is affordable since X dominates over sqrt(1 + X^2) - 1
155 * Final sum is dTmp5 (hi) + dTmp7 (lo)
156 */
157 movaps %xmm0, %xmm1
158
159/*
160 * Check whether the input is finite, by checking |X| <= MaxFloat
161 * Otherwise set the rangemask so that the callout will get used.
162 * Note that this will also use the callout for NaNs since not(NaN <= MaxFloat)
163 */
164 movaps %xmm6, %xmm9
165
166/*
167 * The following computation can go wrong for very large X, basically
168 * because X^2 overflows. But for large X we have
169 * asinh(X) / log(2 X) - 1 =~= 1/(4 * X^2), so for X >= 2^30
170 * we can just later stick X back into the log and tweak up the exponent.
171 * Actually we scale X by 2^-30 and tweak the exponent up by 31,
172 * to stay in the safe range for the later log computation.
173 * Compute a flag now telling us when do do this.
174 */
175 movaps %xmm6, %xmm5
176 cmpnleps sLargestFinite+__svml_sasinh_data_internal(%rip), %xmm9
177 cmpltps sBigThreshold+__svml_sasinh_data_internal(%rip), %xmm5
178 mulps %xmm0, %xmm13
179 addps %xmm4, %xmm1
180 subps %xmm7, %xmm0
181 mulps %xmm4, %xmm14
182 movmskps %xmm9, %edx
183 movaps %xmm7, %xmm9
184
185/*
186 * Now 1 / (1 + d)
187 * = 1 / (1 + (sqrt(1 - e) - 1))
188 * = 1 / sqrt(1 - e)
189 * = 1 + 1/2 * e + 3/8 * e^2 + 5/16 * e^3 + 35/128 * e^4 + ...
190 * So compute the first three nonconstant terms of that, so that
191 * we have a relative correction (1 + Corr) to apply to S etc.
192 * C1 = 1/2
193 * C2 = 3/8
194 * C3 = 5/16
195 */
196 movups sC3+__svml_sasinh_data_internal(%rip), %xmm15
197 subps %xmm13, %xmm9
198 movups sHalf+__svml_sasinh_data_internal(%rip), %xmm10
199 subps %xmm14, %xmm9
200
201/* sX2over2 = X^2/2 */
202 mulps %xmm10, %xmm3
203 mulps %xmm9, %xmm15
204
205/* sX46 = -X^4/4 + X^6/8 */
206 movaps %xmm3, %xmm2
207 movaps %xmm3, %xmm12
208
209/*
210 * Now do another compensated sum to add |X| + [sqrt(1 + X^2) - 1].
211 * It's always safe to assume |X| is larger.
212 * This is the final 2-part argument to the log1p function
213 */
214 movaps %xmm6, %xmm14
215 addps sC2+__svml_sasinh_data_internal(%rip), %xmm15
216 mulps %xmm9, %xmm15
217 addps %xmm10, %xmm15
218 mulps %xmm15, %xmm9
219 mulps %xmm1, %xmm9
220
221/* Now multiplex to the case X = 2^-30 * input, Xl = sL = 0 in the "big" case. */
222 movups XScale+__svml_sasinh_data_internal(%rip), %xmm15
223 addps %xmm9, %xmm4
224 movaps %xmm4, %xmm11
225 addps %xmm0, %xmm11
226 subps %xmm11, %xmm0
227 addps %xmm0, %xmm4
228
229/* sX4over4 = X^4/4 */
230 movaps %xmm3, %xmm0
231 mulps %xmm3, %xmm0
232 mulps %xmm0, %xmm2
233 subps %xmm0, %xmm2
234
235/*
236 * Now we feed into the log1p code, using H in place of _VARG1 and
237 * also adding L into Xl.
238 * compute 1+x as high, low parts
239 */
240 movaps %xmm7, %xmm0
241
242/* sX46over2 = -X^4/8 + x^6/16 */
243 mulps %xmm2, %xmm10
244 movaps %xmm7, %xmm2
245 addps %xmm10, %xmm12
246 subps %xmm12, %xmm3
247 addps %xmm3, %xmm10
248
249/* Now multiplex the two possible computations */
250 movaps %xmm6, %xmm3
251 cmpleps sLittleThreshold+__svml_sasinh_data_internal(%rip), %xmm3
252 movaps %xmm3, %xmm13
253 andps %xmm3, %xmm12
254 andnps %xmm11, %xmm13
255 movaps %xmm3, %xmm1
256 orps %xmm12, %xmm13
257 andnps %xmm4, %xmm1
258 andps %xmm3, %xmm10
259 movaps %xmm6, %xmm4
260 orps %xmm10, %xmm1
261 addps %xmm13, %xmm14
262 mulps %xmm15, %xmm6
263 maxps %xmm14, %xmm0
264 minps %xmm14, %xmm2
265 subps %xmm14, %xmm4
266 movaps %xmm0, %xmm3
267 addps %xmm4, %xmm13
268 addps %xmm2, %xmm3
269 addps %xmm13, %xmm1
270 subps %xmm3, %xmm0
271 movaps %xmm5, %xmm4
272 andps %xmm5, %xmm3
273 andnps %xmm6, %xmm4
274 addps %xmm0, %xmm2
275
276/*
277 * Now resume the main code.
278 * reduction: compute r,n
279 */
280 movdqu iBrkValue+__svml_sasinh_data_internal(%rip), %xmm6
281 orps %xmm3, %xmm4
282 psubd %xmm6, %xmm4
283 movaps %xmm7, %xmm0
284 addps %xmm2, %xmm1
285 movdqu iOffExpoMask+__svml_sasinh_data_internal(%rip), %xmm2
286 pand %xmm4, %xmm2
287 psrad $23, %xmm4
288 cvtdq2ps %xmm4, %xmm3
289 pslld $23, %xmm4
290 andps %xmm5, %xmm1
291 paddd %xmm6, %xmm2
292 psubd %xmm4, %xmm0
293 mulps %xmm0, %xmm1
294
295/* polynomial evaluation */
296 subps %xmm7, %xmm2
297 movups sPoly+112+__svml_sasinh_data_internal(%rip), %xmm7
298 addps %xmm2, %xmm1
299 mulps %xmm1, %xmm7
300 movaps %xmm5, %xmm2
301
302/* Add 31 to the exponent in the "large" case to get log(2 * input) */
303 movups sThirtyOne+__svml_sasinh_data_internal(%rip), %xmm0
304 addps sPoly+96+__svml_sasinh_data_internal(%rip), %xmm7
305 addps %xmm3, %xmm0
306 mulps %xmm1, %xmm7
307 andnps %xmm0, %xmm2
308 andps %xmm5, %xmm3
309 orps %xmm3, %xmm2
310 addps sPoly+80+__svml_sasinh_data_internal(%rip), %xmm7
311
312/* final reconstruction */
313 mulps sLn2+__svml_sasinh_data_internal(%rip), %xmm2
314 mulps %xmm1, %xmm7
315
316/* Finally, reincorporate the original sign. */
317 movups sSign+__svml_sasinh_data_internal(%rip), %xmm0
318 andps %xmm8, %xmm0
319 addps sPoly+64+__svml_sasinh_data_internal(%rip), %xmm7
320 mulps %xmm1, %xmm7
321 addps sPoly+48+__svml_sasinh_data_internal(%rip), %xmm7
322 mulps %xmm1, %xmm7
323 addps sPoly+32+__svml_sasinh_data_internal(%rip), %xmm7
324 mulps %xmm1, %xmm7
325 addps sPoly+16+__svml_sasinh_data_internal(%rip), %xmm7
326 mulps %xmm1, %xmm7
327 addps sPoly+__svml_sasinh_data_internal(%rip), %xmm7
328 mulps %xmm1, %xmm7
329 mulps %xmm1, %xmm7
330 addps %xmm7, %xmm1
331 addps %xmm2, %xmm1
332 pxor %xmm1, %xmm0
333 testl %edx, %edx
334
335/* Go to special inputs processing branch */
336 jne L(SPECIAL_VALUES_BRANCH)
337 # LOE rbx rbp r12 r13 r14 r15 edx xmm0 xmm8
338
339/* Restore registers
340 * and exit the function
341 */
342
343L(EXIT):
344 addq $72, %rsp
345 cfi_def_cfa_offset(8)
346 ret
347 cfi_def_cfa_offset(80)
348
349/* Branch to process
350 * special inputs
351 */
352
353L(SPECIAL_VALUES_BRANCH):
354 movups %xmm8, 32(%rsp)
355 movups %xmm0, 48(%rsp)
356 # LOE rbx rbp r12 r13 r14 r15 edx
357
358 xorl %eax, %eax
359 movq %r12, 16(%rsp)
360 cfi_offset(12, -64)
361 movl %eax, %r12d
362 movq %r13, 8(%rsp)
363 cfi_offset(13, -72)
364 movl %edx, %r13d
365 movq %r14, (%rsp)
366 cfi_offset(14, -80)
367 # LOE rbx rbp r15 r12d r13d
368
369/* Range mask
370 * bits check
371 */
372
373L(RANGEMASK_CHECK):
374 btl %r12d, %r13d
375
376/* Call scalar math function */
377 jc L(SCALAR_MATH_CALL)
378 # LOE rbx rbp r15 r12d r13d
379
380/* Special inputs
381 * processing loop
382 */
383
384L(SPECIAL_VALUES_LOOP):
385 incl %r12d
386 cmpl $4, %r12d
387
388/* Check bits in range mask */
389 jl L(RANGEMASK_CHECK)
390 # LOE rbx rbp r15 r12d r13d
391
392 movq 16(%rsp), %r12
393 cfi_restore(12)
394 movq 8(%rsp), %r13
395 cfi_restore(13)
396 movq (%rsp), %r14
397 cfi_restore(14)
398 movups 48(%rsp), %xmm0
399
400/* Go to exit */
401 jmp L(EXIT)
402 cfi_offset(12, -64)
403 cfi_offset(13, -72)
404 cfi_offset(14, -80)
405 # LOE rbx rbp r12 r13 r14 r15 xmm0
406
407/* Scalar math fucntion call
408 * to process special input
409 */
410
411L(SCALAR_MATH_CALL):
412 movl %r12d, %r14d
413 movss 32(%rsp,%r14,4), %xmm0
414 call asinhf@PLT
415 # LOE rbx rbp r14 r15 r12d r13d xmm0
416
417 movss %xmm0, 48(%rsp,%r14,4)
418
419/* Process special inputs in loop */
420 jmp L(SPECIAL_VALUES_LOOP)
421 # LOE rbx rbp r15 r12d r13d
422END(_ZGVbN4v_asinhf_sse4)
