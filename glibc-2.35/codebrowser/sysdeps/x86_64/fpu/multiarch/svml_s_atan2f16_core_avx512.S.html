19/*
20 * ALGORITHM DESCRIPTION:
21 * For 0.0 <= x <= 7.0/16.0: atan(x) = atan(0.0) + atan(s), where s=(x-0.0)/(1.0+0.0*x)
22 * For 7.0/16.0 <= x <= 11.0/16.0: atan(x) = atan(0.5) + atan(s), where s=(x-0.5)/(1.0+0.5*x)
23 * For 11.0/16.0 <= x <= 19.0/16.0: atan(x) = atan(1.0) + atan(s), where s=(x-1.0)/(1.0+1.0*x)
24 * For 19.0/16.0 <= x <= 39.0/16.0: atan(x) = atan(1.5) + atan(s), where s=(x-1.5)/(1.0+1.5*x)
25 * For 39.0/16.0 <= x <= inf : atan(x) = atan(inf) + atan(s), where s=-1.0/x
26 * Where atan(s) ~= s+s^3*Poly11(s^2) on interval |s|<7.0/0.16.
27 *
28 *
29 */

55ENTRY(_ZGVeN16vv_atan2f_skx)
56 pushq %rbp
57 cfi_def_cfa_offset(16)
58 movq %rsp, %rbp
59 cfi_def_cfa(6, 16)
60 cfi_offset(6, -16)
61 andq $-64, %rsp
62 subq $256, %rsp
63 xorl %edx, %edx
64
65/*
66 * #define NO_VECTOR_ZERO_ATAN2_ARGS
67 * Declarations
68 * Variables
69 * Constants
70 * The end of declarations
71 * Implementation
72 * Arguments signs
73 */
74 vmovups sABS_MASK+__svml_satan2_data_internal(%rip), %zmm6
75 vmovups sONE+__svml_satan2_data_internal(%rip), %zmm3
76
77/* Testing on working interval. */
78 vmovups iCHK_WORK_SUB+__svml_satan2_data_internal(%rip), %zmm9
79 vmovups iCHK_WORK_CMP+__svml_satan2_data_internal(%rip), %zmm14

81/*
82 * 1) If y<x then a= y, b=x, PIO2=0
83 * 2) If y>x then a=-x, b=y, PIO2=Pi/2
84 */
85 vmovups sPIO2+__svml_satan2_data_internal(%rip), %zmm4
86 vpternlogd $255, %zmm13, %zmm13, %zmm13
87 vmovaps %zmm1, %zmm8
88 vandps %zmm6, %zmm8, %zmm2
89 vandps %zmm6, %zmm0, %zmm1
90 vorps sSIGN_MASK+__svml_satan2_data_internal(%rip), %zmm2, %zmm5
91 vpsubd %zmm9, %zmm2, %zmm10
92 vpsubd %zmm9, %zmm1, %zmm12
93 vxorps %zmm2, %zmm8, %zmm7
94 vxorps %zmm1, %zmm0, %zmm6
95 vcmpps $17, {sae}, %zmm2, %zmm1, %k1
96 vpcmpgtd %zmm10, %zmm14, %k2
97 vpcmpgtd %zmm12, %zmm14, %k3
98 vmovups sPC6+__svml_satan2_data_internal(%rip), %zmm14
99 vblendmps %zmm1, %zmm5, %zmm11{%k1}
100 vblendmps %zmm2, %zmm1, %zmm5{%k1}
101 vxorps %zmm4, %zmm4, %zmm4{%k1}
102
103/*
104 * Division a/b.
105 * Enabled when FMA is available and
106 * performance is better with NR iteration
107 */
108 vrcp14ps %zmm5, %zmm15
109 vfnmadd231ps {rn-sae}, %zmm5, %zmm15, %zmm3
110 vfmadd213ps {rn-sae}, %zmm15, %zmm3, %zmm15
111 vmulps {rn-sae}, %zmm15, %zmm11, %zmm3
112 vfnmadd231ps {rn-sae}, %zmm5, %zmm3, %zmm11
113 vfmadd213ps {rn-sae}, %zmm3, %zmm11, %zmm15
114 vmovups sPC8+__svml_satan2_data_internal(%rip), %zmm11
115 vpternlogd $255, %zmm3, %zmm3, %zmm3
116
117/* Polynomial. */
118 vmulps {rn-sae}, %zmm15, %zmm15, %zmm9
119 vpandnd %zmm10, %zmm10, %zmm13{%k2}
120 vmulps {rn-sae}, %zmm9, %zmm9, %zmm10
121 vfmadd231ps {rn-sae}, %zmm10, %zmm11, %zmm14
122 vmovups sPC5+__svml_satan2_data_internal(%rip), %zmm11
123 vpandnd %zmm12, %zmm12, %zmm3{%k3}
124 vpord %zmm3, %zmm13, %zmm3
125 vmovups sPC4+__svml_satan2_data_internal(%rip), %zmm13
126 vmovups sPC7+__svml_satan2_data_internal(%rip), %zmm12
127 vptestmd %zmm3, %zmm3, %k0
128 vfmadd213ps {rn-sae}, %zmm13, %zmm10, %zmm14
129 vfmadd231ps {rn-sae}, %zmm10, %zmm12, %zmm11
130 vmovups sPC3+__svml_satan2_data_internal(%rip), %zmm12
131 vmovups sPC2+__svml_satan2_data_internal(%rip), %zmm13
132
133/* Special branch for fast (vector) processing of zero arguments */
134 kortestw %k0, %k0
135 vfmadd213ps {rn-sae}, %zmm12, %zmm10, %zmm11
136 vmovups sPC1+__svml_satan2_data_internal(%rip), %zmm12
137 vfmadd213ps {rn-sae}, %zmm13, %zmm10, %zmm14
138 vmovups sPC0+__svml_satan2_data_internal(%rip), %zmm13
139 vfmadd213ps {rn-sae}, %zmm12, %zmm10, %zmm11
140 vfmadd213ps {rn-sae}, %zmm13, %zmm10, %zmm14
141 vfmadd213ps {rn-sae}, %zmm14, %zmm9, %zmm11
142
143/* Reconstruction. */
144 vfmadd213ps {rn-sae}, %zmm4, %zmm15, %zmm11
145
146/* if x<0, sPI = Pi, else sPI =0 */
147 vmovups __svml_satan2_data_internal(%rip), %zmm15
148 vorps %zmm7, %zmm11, %zmm9
149 vcmpps $18, {sae}, %zmm15, %zmm8, %k4
150 vmovups sPI+__svml_satan2_data_internal(%rip), %zmm11
151 vaddps {rn-sae}, %zmm11, %zmm9, %zmm9{%k4}
152 vorps %zmm6, %zmm9, %zmm10
153
154/* Go to auxilary branch */
155 jne L(AUX_BRANCH)
156 # LOE rbx r12 r13 r14 r15 edx zmm0 zmm1 zmm2 zmm3 zmm4 zmm5 zmm6 zmm7 zmm8 zmm10 zmm11
157
158/* Return from auxilary branch
159 * for out of main path inputs
160 */
161
162L(AUX_BRANCH_RETURN):
163/*
164 * Special branch for fast (vector) processing of zero arguments
165 * The end of implementation
166 */
167 testl %edx, %edx
168
169/* Go to special inputs processing branch */
170 jne L(SPECIAL_VALUES_BRANCH)
171 # LOE rbx r12 r13 r14 r15 edx zmm0 zmm8 zmm10
172
173/* Restore registers
174 * and exit the function
175 */
176
177L(EXIT):
178 vmovaps %zmm10, %zmm0
179 movq %rbp, %rsp
180 popq %rbp
181 cfi_def_cfa(7, 8)
182 cfi_restore(6)
183 ret
184 cfi_def_cfa(6, 16)
185 cfi_offset(6, -16)
186
187/* Branch to process
188 * special inputs
189 */
190
191L(SPECIAL_VALUES_BRANCH):
192 vmovups %zmm0, 64(%rsp)
193 vmovups %zmm8, 128(%rsp)
194 vmovups %zmm10, 192(%rsp)
195 # LOE rbx r12 r13 r14 r15 edx zmm10
196
197 xorl %eax, %eax
198 # LOE rbx r12 r13 r14 r15 eax edx
199
200 vzeroupper
201 movq %r12, 16(%rsp)
202 /* DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -240; DW_OP_plus) */
203 .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x10, 0xff, 0xff, 0xff, 0x22
204 movl %eax, %r12d
205 movq %r13, 8(%rsp)
206 /* DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -248; DW_OP_plus) */
207 .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x08, 0xff, 0xff, 0xff, 0x22
208 movl %edx, %r13d
209 movq %r14, (%rsp)
210 /* DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -256; DW_OP_plus) */
211 .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x00, 0xff, 0xff, 0xff, 0x22
212 # LOE rbx r15 r12d r13d
213
214/* Range mask
215 * bits check
216 */
217
218L(RANGEMASK_CHECK):
219 btl %r12d, %r13d
220
221/* Call scalar math function */
222 jc L(SCALAR_MATH_CALL)
223 # LOE rbx r15 r12d r13d
224
225/* Special inputs
226 * processing loop
227 */
228
229L(SPECIAL_VALUES_LOOP):
230 incl %r12d
231 cmpl $16, %r12d
232
233/* Check bits in range mask */
234 jl L(RANGEMASK_CHECK)
235 # LOE rbx r15 r12d r13d
236
237 movq 16(%rsp), %r12
238 cfi_restore(12)
239 movq 8(%rsp), %r13
240 cfi_restore(13)
241 movq (%rsp), %r14
242 cfi_restore(14)
243 vmovups 192(%rsp), %zmm10
244
245/* Go to exit */
246 jmp L(EXIT)
247 /* DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -240; DW_OP_plus) */
248 .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x10, 0xff, 0xff, 0xff, 0x22
249 /* DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -248; DW_OP_plus) */
250 .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x08, 0xff, 0xff, 0xff, 0x22
251 /* DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -256; DW_OP_plus) */
252 .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x00, 0xff, 0xff, 0xff, 0x22
253 # LOE rbx r12 r13 r14 r15 zmm10
254
255/* Scalar math fucntion call
256 * to process special input
257 */
258
259L(SCALAR_MATH_CALL):
260 movl %r12d, %r14d
261 movss 64(%rsp,%r14,4), %xmm0
262 movss 128(%rsp,%r14,4), %xmm1
263 call atan2f@PLT
264 # LOE rbx r14 r15 r12d r13d xmm0
265
266 movss %xmm0, 192(%rsp,%r14,4)
267
268/* Process special inputs in loop */
269 jmp L(SPECIAL_VALUES_LOOP)
270 cfi_restore(12)
271 cfi_restore(13)
272 cfi_restore(14)
273 # LOE rbx r15 r12d r13d
274
275/* Auxilary branch
276 * for out of main path inputs
277 */
278
279L(AUX_BRANCH):
280/* Check if at least on of Y or Y is zero: iAXAYZERO */
281 vmovups __svml_satan2_data_internal(%rip), %zmm9
282
283/* Check if both X & Y are not NaNs: iXYnotNAN */
284 vcmpps $3, {sae}, %zmm8, %zmm8, %k1
285 vcmpps $3, {sae}, %zmm0, %zmm0, %k2
286 vpcmpd $4, %zmm9, %zmm2, %k3
287 vpcmpd $4, %zmm9, %zmm1, %k4
288
289/*
290 * Path for zero arguments (at least one of both)
291 * Check if both args are zeros (den. is zero)
292 */
293 vcmpps $4, {sae}, %zmm9, %zmm5, %k5
294
295/* Res = sign(Y)*(X<0)?(PIO2+PI):PIO2 */
296 vpcmpgtd %zmm8, %zmm9, %k6
297 vpternlogd $255, %zmm14, %zmm14, %zmm14
298 vpternlogd $255, %zmm12, %zmm12, %zmm12
299 vpternlogd $255, %zmm13, %zmm13, %zmm13
300 vpandnd %zmm2, %zmm2, %zmm14{%k3}
301 vpternlogd $255, %zmm2, %zmm2, %zmm2
302 vpandnd %zmm1, %zmm1, %zmm2{%k4}
303 vpord %zmm2, %zmm14, %zmm15
304 vpternlogd $255, %zmm2, %zmm2, %zmm2
305 vpandnd %zmm5, %zmm5, %zmm2{%k5}
306
307/* Set sPIO2 to zero if den. is zero */
308 vpandnd %zmm4, %zmm2, %zmm4
309 vpandd %zmm2, %zmm9, %zmm5
310 vpord %zmm5, %zmm4, %zmm2
311 vorps %zmm7, %zmm2, %zmm7
312 vaddps {rn-sae}, %zmm11, %zmm7, %zmm7{%k6}
313 vorps %zmm6, %zmm7, %zmm6
314 vpandnd %zmm8, %zmm8, %zmm12{%k1}
315 vpandnd %zmm0, %zmm0, %zmm13{%k2}
316 vandps %zmm13, %zmm12, %zmm12
317
318/* Check if at least on of Y or Y is zero and not NaN: iAXAYZEROnotNAN */
319 vpandd %zmm12, %zmm15, %zmm1
320
321/* Exclude from previous callout mask zero (and not NaN) arguments */
322 vpandnd %zmm3, %zmm1, %zmm3
323
324/* Go to callout */
325 vptestmd %zmm3, %zmm3, %k0
326 kmovw %k0, %edx
327
328/* Merge results from main and spec path */
329 vpandnd %zmm10, %zmm1, %zmm10
330 vpandd %zmm1, %zmm6, %zmm11
331 vpord %zmm11, %zmm10, %zmm10
332
333/* Return to main vector processing path */
334 jmp L(AUX_BRANCH_RETURN)
335 # LOE rbx r12 r13 r14 r15 edx zmm0 zmm8 zmm10
336END(_ZGVeN16vv_atan2f_skx)
