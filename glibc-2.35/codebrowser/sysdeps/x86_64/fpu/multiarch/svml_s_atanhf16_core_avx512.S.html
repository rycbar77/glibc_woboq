19/*
20 * ALGORITHM DESCRIPTION:
21 *
22 * Compute atanh(x) as 0.5 * log((1 + x)/(1 - x))
23 * using small lookup table that map to AVX-512 permute instructions
24 *
25 * Special cases:
26 *
27 * atanh(0) = 0
28 * atanh(+1) = +INF
29 * atanh(-1) = -INF
30 * atanh(x) = NaN if |x| > 1, or if x is a NaN or INF
31 *
32 */
33
34/* Offsets for data table __svml_satanh_data_internal_avx512
35 */
36#define Log_tbl_H 0
37#define Log_tbl_L 128
38#define One 256
39#define AbsMask 320
40#define AddB5 384
41#define RcpBitMask 448
42#define poly_coeff3 512
43#define poly_coeff2 576
44#define poly_coeff1 640
45#define poly_coeff0 704
46#define Half 768
47#define L2H 832
48#define L2L 896

50#include <sysdep.h>

52 .text
53 .section .text.exex512,"ax",@progbits
54ENTRY(_ZGVeN16v_atanhf_skx)
55 pushq %rbp
56 cfi_def_cfa_offset(16)
57 movq %rsp, %rbp
58 cfi_def_cfa(6, 16)
59 cfi_offset(6, -16)
60 andq $-64, %rsp
61 subq $192, %rsp
62 vmovups One+__svml_satanh_data_internal_avx512(%rip), %zmm4
63
64/* round reciprocals to 1+5b mantissas */
65 vmovups AddB5+__svml_satanh_data_internal_avx512(%rip), %zmm14
66 vmovups RcpBitMask+__svml_satanh_data_internal_avx512(%rip), %zmm1
67 vmovaps %zmm0, %zmm11
68 vandps AbsMask+__svml_satanh_data_internal_avx512(%rip), %zmm11, %zmm6
69
70/* 1+y */
71 vaddps {rn-sae}, %zmm4, %zmm6, %zmm9
72
73/* 1-y */
74 vsubps {rn-sae}, %zmm6, %zmm4, %zmm8
75 vxorps %zmm6, %zmm11, %zmm10
76
77/* Yp_high */
78 vsubps {rn-sae}, %zmm4, %zmm9, %zmm2
79
80/* -Ym_high */
81 vsubps {rn-sae}, %zmm4, %zmm8, %zmm5
82
83/* RcpP ~ 1/Yp */
84 vrcp14ps %zmm9, %zmm12
85
86/* RcpM ~ 1/Ym */
87 vrcp14ps %zmm8, %zmm13
88
89/* input outside (-1, 1) ? */
90 vcmpps $21, {sae}, %zmm4, %zmm6, %k0
91 vpaddd %zmm14, %zmm12, %zmm15
92 vpaddd %zmm14, %zmm13, %zmm0
93
94/* Yp_low */
95 vsubps {rn-sae}, %zmm2, %zmm6, %zmm3
96 vandps %zmm1, %zmm15, %zmm7
97 vandps %zmm1, %zmm0, %zmm12
98
99/* Ym_low */
100 vaddps {rn-sae}, %zmm5, %zmm6, %zmm5
101
102/* Reduced argument: Rp = (RcpP*Yp - 1)+RcpP*Yp_low */
103 vfmsub213ps {rn-sae}, %zmm4, %zmm7, %zmm9
104
105/* Reduced argument: Rm = (RcpM*Ym - 1)+RcpM*Ym_low */
106 vfmsub231ps {rn-sae}, %zmm12, %zmm8, %zmm4
107 vmovups Log_tbl_L+__svml_satanh_data_internal_avx512(%rip), %zmm8
108 vmovups Log_tbl_L+64+__svml_satanh_data_internal_avx512(%rip), %zmm13
109
110/* exponents */
111 vgetexpps {sae}, %zmm7, %zmm15
112 vfmadd231ps {rn-sae}, %zmm7, %zmm3, %zmm9
113
114/* Table lookups */
115 vmovups __svml_satanh_data_internal_avx512(%rip), %zmm6
116 vgetexpps {sae}, %zmm12, %zmm14
117 vfnmadd231ps {rn-sae}, %zmm12, %zmm5, %zmm4
118
119/* Prepare table index */
120 vpsrld $18, %zmm7, %zmm3
121 vpsrld $18, %zmm12, %zmm2
122 vmovups Log_tbl_H+64+__svml_satanh_data_internal_avx512(%rip), %zmm7
123 vmovups poly_coeff1+__svml_satanh_data_internal_avx512(%rip), %zmm12
124
125/* Km-Kp */
126 vsubps {rn-sae}, %zmm15, %zmm14, %zmm1
127 kmovw %k0, %edx
128 vmovaps %zmm3, %zmm0
129 vpermi2ps %zmm13, %zmm8, %zmm3
130 vpermt2ps %zmm13, %zmm2, %zmm8
131 vpermi2ps %zmm7, %zmm6, %zmm0
132 vpermt2ps %zmm7, %zmm2, %zmm6
133 vsubps {rn-sae}, %zmm3, %zmm8, %zmm5
134
135/* K*L2H + Th */
136 vmovups L2H+__svml_satanh_data_internal_avx512(%rip), %zmm2
137
138/* K*L2L + Tl */
139 vmovups L2L+__svml_satanh_data_internal_avx512(%rip), %zmm3
140
141/* polynomials */
142 vmovups poly_coeff3+__svml_satanh_data_internal_avx512(%rip), %zmm7
143 vmovups poly_coeff0+__svml_satanh_data_internal_avx512(%rip), %zmm13
144
145/* table values */
146 vsubps {rn-sae}, %zmm0, %zmm6, %zmm0
147 vfmadd231ps {rn-sae}, %zmm1, %zmm2, %zmm0
148 vfmadd213ps {rn-sae}, %zmm5, %zmm3, %zmm1
149 vmovups poly_coeff2+__svml_satanh_data_internal_avx512(%rip), %zmm3
150 vmovaps %zmm3, %zmm2
151 vfmadd231ps {rn-sae}, %zmm9, %zmm7, %zmm2
152 vfmadd231ps {rn-sae}, %zmm4, %zmm7, %zmm3
153 vfmadd213ps {rn-sae}, %zmm12, %zmm9, %zmm2
154 vfmadd213ps {rn-sae}, %zmm12, %zmm4, %zmm3
155 vfmadd213ps {rn-sae}, %zmm13, %zmm9, %zmm2
156 vfmadd213ps {rn-sae}, %zmm13, %zmm4, %zmm3
157
158/* (K*L2L + Tl) + Rp*PolyP */
159 vfmadd213ps {rn-sae}, %zmm1, %zmm9, %zmm2
160 vorps Half+__svml_satanh_data_internal_avx512(%rip), %zmm10, %zmm9
161
162/* (K*L2L + Tl) + Rp*PolyP -Rm*PolyM */
163 vfnmadd213ps {rn-sae}, %zmm2, %zmm4, %zmm3
164 vaddps {rn-sae}, %zmm3, %zmm0, %zmm4
165 vmulps {rn-sae}, %zmm9, %zmm4, %zmm0
166 testl %edx, %edx
167
168/* Go to special inputs processing branch */
169 jne L(SPECIAL_VALUES_BRANCH)
170 # LOE rbx r12 r13 r14 r15 edx zmm0 zmm11
171
172/* Restore registers
173 * and exit the function
174 */
175
176L(EXIT):
177 movq %rbp, %rsp
178 popq %rbp
179 cfi_def_cfa(7, 8)
180 cfi_restore(6)
181 ret
182 cfi_def_cfa(6, 16)
183 cfi_offset(6, -16)
184
185/* Branch to process
186 * special inputs
187 */
188
189L(SPECIAL_VALUES_BRANCH):
190 vmovups %zmm11, 64(%rsp)
191 vmovups %zmm0, 128(%rsp)
192 # LOE rbx r12 r13 r14 r15 edx zmm0
193
194 xorl %eax, %eax
195 # LOE rbx r12 r13 r14 r15 eax edx
196
197 vzeroupper
198 movq %r12, 16(%rsp)
199 /* DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -176; DW_OP_plus) */
200 .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x50, 0xff, 0xff, 0xff, 0x22
201 movl %eax, %r12d
202 movq %r13, 8(%rsp)
203 /* DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -184; DW_OP_plus) */
204 .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x48, 0xff, 0xff, 0xff, 0x22
205 movl %edx, %r13d
206 movq %r14, (%rsp)
207 /* DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -192; DW_OP_plus) */
208 .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x40, 0xff, 0xff, 0xff, 0x22
209 # LOE rbx r15 r12d r13d
210
211/* Range mask
212 * bits check
213 */
214
215L(RANGEMASK_CHECK):
216 btl %r12d, %r13d
217
218/* Call scalar math function */
219 jc L(SCALAR_MATH_CALL)
220 # LOE rbx r15 r12d r13d
221
222/* Special inputs
223 * processing loop
224 */
225
226L(SPECIAL_VALUES_LOOP):
227 incl %r12d
228 cmpl $16, %r12d
229
230/* Check bits in range mask */
231 jl L(RANGEMASK_CHECK)
232 # LOE rbx r15 r12d r13d
233
234 movq 16(%rsp), %r12
235 cfi_restore(12)
236 movq 8(%rsp), %r13
237 cfi_restore(13)
238 movq (%rsp), %r14
239 cfi_restore(14)
240 vmovups 128(%rsp), %zmm0
241
242/* Go to exit */
243 jmp L(EXIT)
244 /* DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -176; DW_OP_plus) */
245 .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x50, 0xff, 0xff, 0xff, 0x22
246 /* DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -184; DW_OP_plus) */
247 .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x48, 0xff, 0xff, 0xff, 0x22
248 /* DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -192; DW_OP_plus) */
249 .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x40, 0xff, 0xff, 0xff, 0x22
250 # LOE rbx r12 r13 r14 r15 zmm0
251
252/* Scalar math fucntion call
253 * to process special input
254 */
255
256L(SCALAR_MATH_CALL):
257 movl %r12d, %r14d
258 movss 64(%rsp,%r14,4), %xmm0
259 call atanhf@PLT
260 # LOE rbx r14 r15 r12d r13d xmm0
261
262 movss %xmm0, 128(%rsp,%r14,4)
263
264/* Process special inputs in loop */
265 jmp L(SPECIAL_VALUES_LOOP)
266 # LOE rbx r15 r12d r13d
267END(_ZGVeN16v_atanhf_skx)
