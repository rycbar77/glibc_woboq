19/*
20 * ALGORITHM DESCRIPTION:
21 *
22 * x=2^{3*k+j} * 1.b1 b2 ... b5 b6 ... b52
23 * Let r=(x*2^{-3k-j} - 1.b1 b2 ... b5 1)* rcp[b1 b2 ..b5],
24 * where rcp[b1 b2 .. b5]=1/(1.b1 b2 b3 b4 b5 1) in single precision
25 * cbrtf(2^j * 1. b1 b2 .. b5 1) is approximated as T[j][b1..b5]+D[j][b1..b5]
26 * (T stores the high 24 bits, D stores the low order bits)
27 * Result=2^k*T+(2^k*T*r)*P+2^k*D
28 * where P=p1+p2*r+..
29 *
30 */
31

56ENTRY(_ZGVbN4v_cbrtf_sse4)
57 subq $72, %rsp
58 cfi_def_cfa_offset(80)
59
60/*
61 * Load constants
62 * Reciprocal index calculation
63 */
64 movaps %xmm0, %xmm2
65 movdqu _iRcpIndexMask+__svml_scbrt_data_internal(%rip), %xmm3
66 psrld $16, %xmm2
67 pand %xmm2, %xmm3
68
69/* Load reciprocal value */
70 lea __svml_scbrt_data_internal(%rip), %rdx
71 pshufd $1, %xmm3, %xmm5
72
73/* Get signed biased exponent */
74 psrld $7, %xmm2
75 movd %xmm3, %eax
76 movd %xmm5, %ecx
77
78/* Get absolute biased exponent */
79 movdqu _iBExpMask+__svml_scbrt_data_internal(%rip), %xmm15
80
81/*
82 * Calculate exponent/3
83 * i555Exp=(2^{12}-1)/3*exponent
84 */
85 movdqu _i555+__svml_scbrt_data_internal(%rip), %xmm14
86 pand %xmm2, %xmm15
87 movslq %eax, %rax
88 movdqa %xmm14, %xmm5
89 movslq %ecx, %rcx
90 psrlq $32, %xmm14
91 pmuludq %xmm15, %xmm5
92 movd (%rdx,%rax), %xmm4
93 movd (%rdx,%rcx), %xmm6
94 punpckldq %xmm6, %xmm4
95 movdqa %xmm15, %xmm6
96 psrlq $32, %xmm15
97 pmuludq %xmm14, %xmm15
98 pshufd $2, %xmm3, %xmm7
99 psllq $32, %xmm15
100 pshufd $3, %xmm3, %xmm8
101 movd %xmm7, %esi
102 movd %xmm8, %edi
103
104/* Argument reduction */
105 movups _sMantissaMask+__svml_scbrt_data_internal(%rip), %xmm12
106 movups _sMantissaMask1+__svml_scbrt_data_internal(%rip), %xmm11
107 andps %xmm0, %xmm12
108 pand .FLT_17(%rip), %xmm5
109 andps %xmm0, %xmm11
110 movslq %esi, %rsi
111 por %xmm15, %xmm5
112 movslq %edi, %rdi
113
114/* Get K (exponent=3*k+j) */
115 psrld $12, %xmm5
116 orps _sExpMask+__svml_scbrt_data_internal(%rip), %xmm12
117 orps _sExpMask1+__svml_scbrt_data_internal(%rip), %xmm11
118 psubd _iOne+__svml_scbrt_data_internal(%rip), %xmm6
119
120/* r=y-y` */
121 subps %xmm11, %xmm12
122
123/* Get J */
124 psubd %xmm5, %xmm6
125 movdqu _iAbsMask+__svml_scbrt_data_internal(%rip), %xmm1
126 psubd %xmm5, %xmm6
127 movd (%rdx,%rsi), %xmm10
128 pand %xmm0, %xmm1
129 movd (%rdx,%rdi), %xmm9
130 psubd %xmm5, %xmm6
131 punpckldq %xmm9, %xmm10
132
133/* Get 128*J */
134 pslld $7, %xmm6
135 punpcklqdq %xmm10, %xmm4
136
137/*
138 * iCbrtIndex=4*l+128*j
139 * Zero index if callout expected
140 */
141 paddd %xmm6, %xmm3
142 psubd _iSubConst+__svml_scbrt_data_internal(%rip), %xmm1
143 pcmpgtd _iCmpConst+__svml_scbrt_data_internal(%rip), %xmm1
144
145/* r=(y-y`)*rcp_table(y`) */
146 mulps %xmm12, %xmm4
147 movmskps %xmm1, %eax
148
149/* Biased exponent-1 */
150 movdqu _iSignMask+__svml_scbrt_data_internal(%rip), %xmm13
151 pandn %xmm3, %xmm1
152
153/*
154 * Add 2/3*(bias-1)+1 to (k+1/3*(bias-1))
155 * Attach sign to exponent
156 */
157 movdqu _iBias+__svml_scbrt_data_internal(%rip), %xmm12
158 pand %xmm13, %xmm2
159 paddd %xmm5, %xmm12
160
161/* Load Cbrt table Hi & Lo values */
162 movd %xmm1, %r8d
163 por %xmm2, %xmm12
164 pshufd $1, %xmm1, %xmm2
165 pslld $23, %xmm12
166 pshufd $2, %xmm1, %xmm7
167 pshufd $3, %xmm1, %xmm1
168 movd %xmm2, %r9d
169 movd %xmm7, %r10d
170 movd %xmm1, %r11d
171
172/* Polynomial: p1+r*(p2*r+r*(p3+r*p4)) */
173 movups _sP2+__svml_scbrt_data_internal(%rip), %xmm11
174 mulps %xmm4, %xmm11
175 movslq %r8d, %r8
176 addps _sP1+__svml_scbrt_data_internal(%rip), %xmm11
177 movslq %r9d, %r9
178 movslq %r10d, %r10
179 movslq %r11d, %r11
180 movd 128(%rdx,%r8), %xmm10
181 movd 128(%rdx,%r9), %xmm3
182 movd 128(%rdx,%r10), %xmm9
183 movd 128(%rdx,%r11), %xmm8
184 punpckldq %xmm3, %xmm10
185 punpckldq %xmm8, %xmm9
186 punpcklqdq %xmm9, %xmm10
187
188/* sCbrtHi *= 2^k */
189 mulps %xmm10, %xmm12
190
191/* T`*r */
192 mulps %xmm12, %xmm4
193
194/* (T`*r)*P */
195 mulps %xmm4, %xmm11
196
197/*
198 * T`*r*P+D`
199 * result = T`+(T`*r*P+D`)
200 */
201 addps %xmm11, %xmm12
202 testl %eax, %eax
203
204/* Go to special inputs processing branch */
205 jne L(SPECIAL_VALUES_BRANCH)
206 # LOE rbx rbp r12 r13 r14 r15 eax xmm0 xmm12
207
208/* Restore registers
209 * and exit the function
210 */
211
212L(EXIT):
213 movaps %xmm12, %xmm0
214 addq $72, %rsp
215 cfi_def_cfa_offset(8)
216 ret
217 cfi_def_cfa_offset(80)
218
219/* Branch to process
220 * special inputs
221 */
222
223L(SPECIAL_VALUES_BRANCH):
224 movups %xmm0, 32(%rsp)
225 movups %xmm12, 48(%rsp)
226 # LOE rbx rbp r12 r13 r14 r15 eax
227
228 xorl %edx, %edx
229 movq %r12, 16(%rsp)
230 cfi_offset(12, -64)
231 movl %edx, %r12d
232 movq %r13, 8(%rsp)
233 cfi_offset(13, -72)
234 movl %eax, %r13d
235 movq %r14, (%rsp)
236 cfi_offset(14, -80)
237 # LOE rbx rbp r15 r12d r13d
238
239/* Range mask
240 * bits check
241 */
242
243L(RANGEMASK_CHECK):
244 btl %r12d, %r13d
245
246/* Call scalar math function */
247 jc L(SCALAR_MATH_CALL)
248 # LOE rbx rbp r15 r12d r13d
249
250/* Special inputs
251 * processing loop
252 */
253
254L(SPECIAL_VALUES_LOOP):
255 incl %r12d
256 cmpl $4, %r12d
257
258/* Check bits in range mask */
259 jl L(RANGEMASK_CHECK)
260 # LOE rbx rbp r15 r12d r13d
261
262 movq 16(%rsp), %r12
263 cfi_restore(12)
264 movq 8(%rsp), %r13
265 cfi_restore(13)
266 movq (%rsp), %r14
267 cfi_restore(14)
268 movups 48(%rsp), %xmm12
269
270/* Go to exit */
271 jmp L(EXIT)
272 cfi_offset(12, -64)
273 cfi_offset(13, -72)
274 cfi_offset(14, -80)
275 # LOE rbx rbp r12 r13 r14 r15 xmm12
276
277/* Scalar math fucntion call
278 * to process special input
279 */
280
281L(SCALAR_MATH_CALL):
282 movl %r12d, %r14d
283 movss 32(%rsp,%r14,4), %xmm0
284 call cbrtf@PLT
285 # LOE rbx rbp r14 r15 r12d r13d xmm0
286
287 movss %xmm0, 48(%rsp,%r14,4)
288
289/* Process special inputs in loop */
290 jmp L(SPECIAL_VALUES_LOOP)
291 # LOE rbx rbp r15 r12d r13d
292END(_ZGVbN4v_cbrtf_sse4)
