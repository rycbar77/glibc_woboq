19/*
20 * ALGORITHM DESCRIPTION:
21 *
22 * x=2^{3*k+j} * 1.b1 b2 ... b5 b6 ... b52
23 * Let r=(x*2^{-3k-j} - 1.b1 b2 ... b5 1)* rcp[b1 b2 ..b5],
24 * where rcp[b1 b2 .. b5]=1/(1.b1 b2 b3 b4 b5 1) in single precision
25 * cbrtf(2^j * 1. b1 b2 .. b5 1) is approximated as T[j][b1..b5]+D[j][b1..b5]
26 * (T stores the high 24 bits, D stores the low order bits)
27 * Result=2^k*T+(2^k*T*r)*P+2^k*D
28 * where P=p1+p2*r+..
29 *
30 */
31

56ENTRY(_ZGVdN8v_cbrtf_avx2)
57 pushq %rbp
58 cfi_def_cfa_offset(16)
59 movq %rsp, %rbp

61 cfi_offset(6, -16)
62 andq $-32, %rsp
63 subq $96, %rsp
64
65/* Load reciprocal value */
66 lea __svml_scbrt_data_internal(%rip), %rdx
67 vmovaps %ymm0, %ymm5
68
69/*
70 * Load constants
71 * Reciprocal index calculation
72 */
73 vpsrld $16, %ymm5, %ymm3
74 vpand _iRcpIndexMask+__svml_scbrt_data_internal(%rip), %ymm3, %ymm4
75 vextractf128 $1, %ymm4, %xmm15
76 vmovd %xmm4, %eax
77 vmovd %xmm15, %r8d
78 vpextrd $1, %xmm15, %r9d
79 vpextrd $2, %xmm15, %r10d
80 vpextrd $3, %xmm15, %r11d
81 movslq %r8d, %r8
82 movslq %r9d, %r9
83 movslq %r10d, %r10
84 movslq %r11d, %r11
85 vpextrd $1, %xmm4, %ecx
86 vpextrd $2, %xmm4, %esi
87 vpextrd $3, %xmm4, %edi
88 movslq %eax, %rax
89 movslq %ecx, %rcx
90 movslq %esi, %rsi
91 movslq %edi, %rdi
92 vmovd (%rdx,%r8), %xmm13
93 vmovd (%rdx,%r9), %xmm14
94 vmovd (%rdx,%r10), %xmm1
95 vmovd (%rdx,%r11), %xmm0
96 vpunpckldq %xmm14, %xmm13, %xmm2
97 vpunpckldq %xmm0, %xmm1, %xmm13
98
99/* Get signed biased exponent */
100 vpsrld $7, %ymm3, %ymm0
101 vmovd (%rdx,%rax), %xmm6
102 vmovd (%rdx,%rcx), %xmm7
103 vmovd (%rdx,%rsi), %xmm8
104 vmovd (%rdx,%rdi), %xmm9
105 vpunpckldq %xmm7, %xmm6, %xmm10
106 vpunpckldq %xmm9, %xmm8, %xmm11
107 vpunpcklqdq %xmm11, %xmm10, %xmm12
108 vpunpcklqdq %xmm13, %xmm2, %xmm6
109 vandps _iAbsMask+__svml_scbrt_data_internal(%rip), %ymm5, %ymm3
110
111/* Argument reduction */
112 vandps _sMantissaMask+__svml_scbrt_data_internal(%rip), %ymm5, %ymm8
113 vandps _sMantissaMask1+__svml_scbrt_data_internal(%rip), %ymm5, %ymm9
114 vpsubd _iSubConst+__svml_scbrt_data_internal(%rip), %ymm3, %ymm7
115 vorps _sExpMask+__svml_scbrt_data_internal(%rip), %ymm8, %ymm10
116 vorps _sExpMask1+__svml_scbrt_data_internal(%rip), %ymm9, %ymm11
117
118/* r=y-y` */
119 vsubps %ymm11, %ymm10, %ymm15
120
121/* Biased exponent-1 */
122 vpand _iSignMask+__svml_scbrt_data_internal(%rip), %ymm0, %ymm8
123 vpcmpgtd _iCmpConst+__svml_scbrt_data_internal(%rip), %ymm7, %ymm2
124 vmovmskps %ymm2, %eax
125 vinsertf128 $1, %xmm6, %ymm12, %ymm14
126
127/* Get absolute biased exponent */
128 vpand _iBExpMask+__svml_scbrt_data_internal(%rip), %ymm0, %ymm6
129
130/* r=(y-y`)*rcp_table(y`) */
131 vmulps %ymm15, %ymm14, %ymm1
132 vpsubd _iOne+__svml_scbrt_data_internal(%rip), %ymm6, %ymm10
133
134/*
135 * Calculate exponent/3
136 * i555Exp=(2^{12}-1)/3*exponent
137 */
138 vpmulld _i555+__svml_scbrt_data_internal(%rip), %ymm6, %ymm3
139
140/* Get K (exponent=3*k+j) */
141 vpsrld $12, %ymm3, %ymm13
142
143/* Get J */
144 vpsubd %ymm13, %ymm10, %ymm11
145
146/* Add 2/3*(bias-1)+1 to (k+1/3*(bias-1)) */
147 vpaddd _iBias+__svml_scbrt_data_internal(%rip), %ymm13, %ymm7
148 vpsubd %ymm13, %ymm11, %ymm12
149
150/* Attach sign to exponent */
151 vpor %ymm8, %ymm7, %ymm9
152 vpsubd %ymm13, %ymm12, %ymm14
153 vpslld $23, %ymm9, %ymm0
154
155/* Get 128*J */
156 vpslld $7, %ymm14, %ymm15
157
158/* iCbrtIndex=4*l+128*j */
159 vpaddd %ymm15, %ymm4, %ymm4
160
161/* Zero index if callout expected */
162 vpandn %ymm4, %ymm2, %ymm4
163
164/* Load Cbrt table Hi & Lo values */
165 vmovd %xmm4, %ecx
166 vextractf128 $1, %ymm4, %xmm13
167 vpextrd $1, %xmm4, %esi
168 movslq %ecx, %rcx
169 movslq %esi, %rsi
170 vmovd %xmm13, %r9d
171 vmovd 128(%rdx,%rcx), %xmm2
172 vpextrd $2, %xmm4, %edi
173 vpextrd $3, %xmm4, %r8d
174 vmovd 128(%rdx,%rsi), %xmm3
175 vpextrd $1, %xmm13, %r10d
176 vpextrd $2, %xmm13, %ecx
177 vpextrd $3, %xmm13, %esi
178 movslq %edi, %rdi
179 movslq %r8d, %r8
180 movslq %r9d, %r9
181 movslq %r10d, %r10
182 movslq %ecx, %rcx
183 movslq %esi, %rsi
184 vmovd 128(%rdx,%rdi), %xmm6
185 vmovd 128(%rdx,%r8), %xmm7
186 vmovd 128(%rdx,%r9), %xmm11
187 vmovd 128(%rdx,%r10), %xmm12
188 vmovd 128(%rdx,%rcx), %xmm14
189 vmovd 128(%rdx,%rsi), %xmm15
190 vpunpckldq %xmm3, %xmm2, %xmm8
191 vpunpckldq %xmm7, %xmm6, %xmm9
192 vpunpckldq %xmm12, %xmm11, %xmm4
193 vpunpckldq %xmm15, %xmm14, %xmm11
194 vpunpcklqdq %xmm9, %xmm8, %xmm10
195 vpunpcklqdq %xmm11, %xmm4, %xmm2
196 vinsertf128 $1, %xmm2, %ymm10, %ymm3
197
198/* sCbrtHi *= 2^k */
199 vmulps %ymm3, %ymm0, %ymm2
200
201/* Polynomial: p1+r*(p2*r+r*(p3+r*p4)) */
202 vmovups _sP2+__svml_scbrt_data_internal(%rip), %ymm0
203 vfmadd213ps _sP1+__svml_scbrt_data_internal(%rip), %ymm1, %ymm0
204
205/* T`*r */
206 vmulps %ymm2, %ymm1, %ymm1
207
208/* (T`*r)*P */
209 vmulps %ymm1, %ymm0, %ymm0
210
211/*
212 * T`*r*P+D`
213 * result = T`+(T`*r*P+D`)
214 */
215 vaddps %ymm0, %ymm2, %ymm0
216 testl %eax, %eax
217
218/* Go to special inputs processing branch */
219 jne L(SPECIAL_VALUES_BRANCH)
220 # LOE rbx r12 r13 r14 r15 eax ymm0 ymm5
221
222/* Restore registers
223 * and exit the function
224 */
225
226L(EXIT):
227 movq %rbp, %rsp
228 popq %rbp
229 cfi_def_cfa(7, 8)
230 cfi_restore(6)
231 ret
232 cfi_def_cfa(6, 16)
233 cfi_offset(6, -16)
234
235/* Branch to process
236 * special inputs
237 */
238
239L(SPECIAL_VALUES_BRANCH):
240 vmovups %ymm5, 32(%rsp)
241 vmovups %ymm0, 64(%rsp)
242 # LOE rbx r12 r13 r14 r15 eax ymm0
243
244 xorl %edx, %edx
245 # LOE rbx r12 r13 r14 r15 eax edx
246
247 vzeroupper
248 movq %r12, 16(%rsp)
249 /* DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -80; DW_OP_plus) */
250 .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xb0, 0xff, 0xff, 0xff, 0x22
251 movl %edx, %r12d
252 movq %r13, 8(%rsp)
253 /* DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -88; DW_OP_plus) */
254 .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xa8, 0xff, 0xff, 0xff, 0x22
255 movl %eax, %r13d
256 movq %r14, (%rsp)
257 /* DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -96; DW_OP_plus) */
258 .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xa0, 0xff, 0xff, 0xff, 0x22
259 # LOE rbx r15 r12d r13d
260
261/* Range mask
262 * bits check
263 */
264
265L(RANGEMASK_CHECK):
266 btl %r12d, %r13d
267
268/* Call scalar math function */
269 jc L(SCALAR_MATH_CALL)
270 # LOE rbx r15 r12d r13d
271
272/* Special inputs
273 * processing loop
274 */
275
276L(SPECIAL_VALUES_LOOP):
277 incl %r12d
278 cmpl $8, %r12d
279
280/* Check bits in range mask */
281 jl L(RANGEMASK_CHECK)
282 # LOE rbx r15 r12d r13d
283
284 movq 16(%rsp), %r12
285 cfi_restore(12)
286 movq 8(%rsp), %r13
287 cfi_restore(13)
288 movq (%rsp), %r14
289 cfi_restore(14)
290 vmovups 64(%rsp), %ymm0
291
292/* Go to exit */
293 jmp L(EXIT)
294 /* DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -80; DW_OP_plus) */
295 .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xb0, 0xff, 0xff, 0xff, 0x22
296 /* DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -88; DW_OP_plus) */
297 .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xa8, 0xff, 0xff, 0xff, 0x22
298 /* DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -96; DW_OP_plus) */
299 .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xa0, 0xff, 0xff, 0xff, 0x22
300 # LOE rbx r12 r13 r14 r15 ymm0
301
302/* Scalar math fucntion call
303 * to process special input
304 */
305
306L(SCALAR_MATH_CALL):
307 movl %r12d, %r14d
308 movss 32(%rsp,%r14,4), %xmm0
309 call cbrtf@PLT
310 # LOE rbx r14 r15 r12d r13d xmm0
311
312 movss %xmm0, 64(%rsp,%r14,4)
313
314/* Process special inputs in loop */
315 jmp L(SPECIAL_VALUES_LOOP)
316 # LOE rbx r15 r12d r13d
