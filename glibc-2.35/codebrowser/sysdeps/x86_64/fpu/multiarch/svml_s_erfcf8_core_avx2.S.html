19/*
20 * ALGORITHM DESCRIPTION:
21 *
22 * Approximation formula:
23 * erfc(x) ~ erfc(x0) - 2/sqrt(pi)*exp(-x0^2)*D * [ 1 + (x0*D)*p1(x0*D) + D^2 * p3(x0*D) ]
24 * D = x - x0
25 * erfc(x0) and 2/sqrt(pi)*exp(-x0^2)/(2*x0) are tabulated
26 *
27 *
28 */

54ENTRY(_ZGVdN8v_erfcf_avx2)

61 subq $96, %rsp

63/* vector gather: erfc_h(x0), (erfc_l(x0), 2/sqrt(pi)*exp(-x0^2)) */

65 vmovaps %ymm0, %ymm6
66 vmovups _SRound+__svml_serfc_data_internal(%rip), %ymm1
67 vmovups _TwoM48+__svml_serfc_data_internal(%rip), %ymm10
68 vmovups _One+__svml_serfc_data_internal(%rip), %ymm0
69 vandps _AbsMask+__svml_serfc_data_internal(%rip), %ymm6, %ymm5
70
71/*
72 * erfc(10.125) underflows to 0
73 * can compute all results in the main path
74 */
75 vminps _MaxThreshold+__svml_serfc_data_internal(%rip), %ymm5, %ymm4
76 vaddps %ymm1, %ymm4, %ymm8
77 vmaxps %ymm10, %ymm4, %ymm2
78 vsubps %ymm1, %ymm8, %ymm12
79 vpslld $3, %ymm8, %ymm7
80 vandps _SgnMask+__svml_serfc_data_internal(%rip), %ymm6, %ymm13
81 vorps %ymm13, %ymm0, %ymm3
82
83/* 2.0 if x<0, 0.0 otherwise */
84 vsubps %ymm3, %ymm0, %ymm5
85 vsubps %ymm12, %ymm2, %ymm0
86
87/* Start polynomial evaluation */
88 vmovups _poly1_0+__svml_serfc_data_internal(%rip), %ymm3
89 vmovups _poly3_0+__svml_serfc_data_internal(%rip), %ymm2
90 vmulps %ymm0, %ymm12, %ymm1
91 vfmadd213ps _poly1_1+__svml_serfc_data_internal(%rip), %ymm1, %ymm3
92 vfmadd213ps _poly3_1+__svml_serfc_data_internal(%rip), %ymm1, %ymm2
93 vfmadd213ps _poly1_2+__svml_serfc_data_internal(%rip), %ymm1, %ymm3
94
95/* 2^(-64) with sign of input */
96 vorps %ymm13, %ymm10, %ymm4
97 vfmadd213ps _poly1_3+__svml_serfc_data_internal(%rip), %ymm1, %ymm3
98 vextractf128 $1, %ymm7, %xmm12
99 vmovd %xmm7, %edx
100 vmovd %xmm12, %r8d
101 vpextrd $1, %xmm7, %ecx
102 vpextrd $2, %xmm7, %esi
103 vpextrd $3, %xmm7, %edi
104 vpextrd $1, %xmm12, %r9d
105 vpextrd $2, %xmm12, %r10d
106 vpextrd $3, %xmm12, %r11d
107 movslq %edx, %rdx
108 movslq %ecx, %rcx
109 movslq %esi, %rsi
110 movslq %edi, %rdi
111 movslq %r8d, %r8
112 movslq %r9d, %r9
113 movslq %r10d, %r10
114 movslq %r11d, %r11
115 vmovq (%rax,%rdx), %xmm13
116 vmovq (%rax,%rcx), %xmm11
117 vmovq (%rax,%rsi), %xmm14
118 vmovq (%rax,%rdi), %xmm15
119 vmovq (%rax,%r8), %xmm7
120 vmovq (%rax,%r9), %xmm8
121 vmovq (%rax,%r10), %xmm9
122 vmovq (%rax,%r11), %xmm10
123 vunpcklps %xmm14, %xmm13, %xmm12
124 vunpcklps %xmm15, %xmm11, %xmm11
125 vunpcklps %xmm9, %xmm7, %xmm9
126 vunpcklps %xmm10, %xmm8, %xmm14
127 vinsertf128 $1, %xmm9, %ymm12, %ymm15
128 vinsertf128 $1, %xmm14, %ymm11, %ymm7
129 vunpcklps %ymm7, %ymm15, %ymm9
130 vunpckhps %ymm7, %ymm15, %ymm8
131
132/* Diff^2 */
133 vmulps %ymm0, %ymm0, %ymm15
134
135/* P3*D2 */
136 vmulps %ymm15, %ymm2, %ymm2
137
138/* P1 = P1*T + P3*D2 */
139 vfmadd213ps %ymm2, %ymm1, %ymm3
140
141/* Special arguments (for flags only) */
142 vcmplt_oqps _UF_Threshold+__svml_serfc_data_internal(%rip), %ymm6, %ymm1
143
144/* EXP_X0H * (1+P1) */
145 vfmadd213ps %ymm8, %ymm8, %ymm3
146
147/* erfc_high(x0) - Diff * (2/sqrt(pi)*exp(-x0^2))*(1+P1) */
148 vfnmadd213ps %ymm9, %ymm3, %ymm0
149
150/* combine and get argument value range mask */
151 vmovmskps %ymm1, %edx
152 notl %edx
153 movzbl %dl, %edx
154 vfmadd213ps %ymm5, %ymm4, %ymm0
155 testl %edx, %edx
156
157/* Go to special inputs processing branch */
158 jne L(SPECIAL_VALUES_BRANCH)
159 # LOE rbx r12 r13 r14 r15 edx ymm0 ymm6
160
161/* Restore registers
162 * and exit the function
163 */
164
165L(EXIT):
166 movq %rbp, %rsp
167 popq %rbp
168 cfi_def_cfa(7, 8)
169 cfi_restore(6)
170 ret
171 cfi_def_cfa(6, 16)
172 cfi_offset(6, -16)
173
174/* Branch to process
175 * special inputs
176 */
177
178L(SPECIAL_VALUES_BRANCH):
179 vmovups %ymm6, 32(%rsp)
180 vmovups %ymm0, 64(%rsp)
181 # LOE rbx r12 r13 r14 r15 edx ymm0
182
183 xorl %eax, %eax
184 # LOE rbx r12 r13 r14 r15 eax edx
185
186 vzeroupper
187 movq %r12, 16(%rsp)
188 /* DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -80; DW_OP_plus) */
189 .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xb0, 0xff, 0xff, 0xff, 0x22
190 movl %eax, %r12d
191 movq %r13, 8(%rsp)
192 /* DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -88; DW_OP_plus) */
193 .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xa8, 0xff, 0xff, 0xff, 0x22
194 movl %edx, %r13d
195 movq %r14, (%rsp)
196 /* DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -96; DW_OP_plus) */
197 .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xa0, 0xff, 0xff, 0xff, 0x22
198 # LOE rbx r15 r12d r13d
199
200/* Range mask
201 * bits check
202 */
203
204L(RANGEMASK_CHECK):
205 btl %r12d, %r13d
206
207/* Call scalar math function */
208 jc L(SCALAR_MATH_CALL)
209 # LOE rbx r15 r12d r13d
210
211/* Special inputs
212 * processing loop
213 */
214
215L(SPECIAL_VALUES_LOOP):
216 incl %r12d
217 cmpl $8, %r12d
218
219/* Check bits in range mask */
220 jl L(RANGEMASK_CHECK)
221 # LOE rbx r15 r12d r13d
222
223 movq 16(%rsp), %r12
224 cfi_restore(12)
225 movq 8(%rsp), %r13
226 cfi_restore(13)
227 movq (%rsp), %r14
228 cfi_restore(14)
229 vmovups 64(%rsp), %ymm0
230
231/* Go to exit */
232 jmp L(EXIT)
233 /* DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -80; DW_OP_plus) */
234 .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xb0, 0xff, 0xff, 0xff, 0x22
235 /* DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -88; DW_OP_plus) */
236 .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xa8, 0xff, 0xff, 0xff, 0x22
237 /* DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -96; DW_OP_plus) */
238 .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xa0, 0xff, 0xff, 0xff, 0x22
239 # LOE rbx r12 r13 r14 r15 ymm0
240
241/* Scalar math fucntion call
242 * to process special input
243 */
244
245L(SCALAR_MATH_CALL):
246 movl %r12d, %r14d
247 movss 32(%rsp,%r14,4), %xmm0
248 call erfcf@PLT
249 # LOE rbx r14 r15 r12d r13d xmm0
250
251 movss %xmm0, 64(%rsp,%r14,4)
252
253/* Process special inputs in loop */
254 jmp L(SPECIAL_VALUES_LOOP)
255 # LOE rbx r15 r12d r13d
256END(_ZGVdN8v_erfcf_avx2)
