19/*
20 * ALGORITHM DESCRIPTION:
21 *
22 * Basic formula is
23 * erf(x) ~ erf(x0) +
24 * + exp(-x0*x0)*D*(1+c0+T*P1(T)+D^2*P3(T)+D^4*p5)
25 * where D=x-x0, T=x0*D
26 * x0 is x rounded to a specified number of fractional bits (in this case 8),
27 * except that x0=0 for |x|<3.5/256.0 (using x0=0 for first 4 table entries)
28 *
29 * Data table packs both erf(x0)_high and a few bits of erf(x0)_low in one
30 * entry (in place of redundant exponent bits)
31 *
32 */
33
34/* Offsets for data table __svml_serf_data_internal
35 */
36#define _erf_tbl 0
37#define _AbsMask 4032
38#define _MaxThreshold 4064
39#define _SRound 4096
40#define _U2Threshold 4128
41#define _poly3_0 4160
42
43/* Lookup bias for data table __svml_serf_data_internal. */
44#define Table_Lookup_Bias -0x3c000000
45
46#include <sysdep.h>
47
48 .text
49 .section .text.avx2,"ax",@progbits
50ENTRY(_ZGVdN8v_erff_avx2)
51 lea Table_Lookup_Bias+__svml_serf_data_internal(%rip), %rax
52
53/*
54 * vector gather:
55 * erf(x0), exp(-x0*x0)*2.0/sqrt(pi)
56 */
57 vmovups _SRound+__svml_serf_data_internal(%rip), %ymm7
58 vandps _AbsMask+__svml_serf_data_internal(%rip), %ymm0, %ymm6
59
60/*
61 * erf(x) rounds to 1.0 for x>_MaxThreshold (3.9375)
62 * can compute all results in the main path
63 */
64 vminps _MaxThreshold+__svml_serf_data_internal(%rip), %ymm6, %ymm8
65 vaddps %ymm7, %ymm8, %ymm10
66 vcmpgt_oqps _U2Threshold+__svml_serf_data_internal(%rip), %ymm8, %ymm9
67 vpslld $3, %ymm10, %ymm11
68 vsubps %ymm7, %ymm10, %ymm4
69 vsubps %ymm4, %ymm8, %ymm3
70 vandps %ymm9, %ymm3, %ymm2
71
72/* NaN fixup */
73 vminps %ymm6, %ymm3, %ymm3
74
75/* D2 = Diff^2 */
76 vmulps %ymm2, %ymm2, %ymm2
77
78/* save sign */
79 vxorps %ymm0, %ymm6, %ymm5
80 vmovd %xmm11, %edx
81 vextractf128 $1, %ymm11, %xmm12
82 vpextrd $2, %xmm11, %esi
83 movslq %edx, %rdx
84 movslq %esi, %rsi
85 vmovd %xmm12, %r8d
86 vmovq (%rax,%rdx), %xmm13
87 vmovq (%rax,%rsi), %xmm14
88 vunpcklps %xmm14, %xmm13, %xmm10
89 vmovups _poly3_0+__svml_serf_data_internal(%rip), %ymm14
90 vpextrd $1, %xmm11, %ecx
91 vpextrd $3, %xmm11, %edi
92 vpextrd $1, %xmm12, %r9d
93 vpextrd $2, %xmm12, %r10d
94 vpextrd $3, %xmm12, %r11d
95
96/*
97 * Start polynomial evaluation
98 * P1
99 */
100 vfmsub231ps %ymm14, %ymm3, %ymm4
101 movslq %ecx, %rcx
102 movslq %edi, %rdi
103 movslq %r8d, %r8
104 movslq %r9d, %r9
105 movslq %r10d, %r10
106 movslq %r11d, %r11
107 vmovq (%rax,%rcx), %xmm1
108 vmovq (%rax,%rdi), %xmm15
109
110/*
111 * branch-free
112 * (exp_h(x0) * Diff) * (poly + 1.0)
113 */
114 vfmadd213ps %ymm3, %ymm2, %ymm4
115 vmovq (%rax,%r8), %xmm7
116 vmovq (%rax,%r9), %xmm0
117 vmovq (%rax,%r10), %xmm8
118 vmovq (%rax,%r11), %xmm9
119 vunpcklps %xmm15, %xmm1, %xmm11
120 vunpcklps %xmm8, %xmm7, %xmm1
121 vunpcklps %xmm9, %xmm0, %xmm0
122 vinsertf128 $1, %xmm1, %ymm10, %ymm12
123 vinsertf128 $1, %xmm0, %ymm11, %ymm13
124 vunpcklps %ymm13, %ymm12, %ymm0
125 vunpckhps %ymm13, %ymm12, %ymm15
126
127/* Final result */
128 vfmadd213ps %ymm0, %ymm15, %ymm4
129
130/* set sign */
131 vorps %ymm5, %ymm4, %ymm0
132 ret
133
134END(_ZGVdN8v_erff_avx2)
