19/*
20 * ALGORITHM DESCRIPTION:
21 *
22 * ( optimized for throughput, with small table lookup, works when HW FMA is available )
23 *
24 * Implementation reduces argument x to |R|<pi/64
25 * 32-entry tables used to store high and low parts of tan(x0)
26 * Argument x = N*pi + x0 + (R); x0 = k*pi/32, with k in {0, 1, ..., 31}
27 * (very large arguments reduction resolved in _vsreduction_core.i)
28 * Compute result as (tan(x0) + tan(R))/(1-tan(x0)*tan(R))
29 * _HA_ version keeps extra precision for numerator, denominator, and during
30 * final NR-iteration computing quotient.
31 *
32 *
33 */
34

73 * Main path

75 * start arg. reduction

81 vmovaps %zmm0, %zmm11

87 *
88 * End of main path
89 */

93 vsubps {rn-sae}, %zmm1, %zmm10, %zmm5
94 vfnmadd213ps {rn-sae}, %zmm11, %zmm5, %zmm4

98/* Go to auxilary branch */
99 jne L(AUX_BRANCH)
100 # LOE rbx r12 r13 r14 r15 edx zmm0 zmm5 zmm10 zmm11 k6

102/* Return from auxilary branch
103 * for out of main path inputs
104 */
105
106L(AUX_BRANCH_RETURN):
107/* Table lookup */
108 vmovups Th_tbl_uisa+__svml_stan_data_internal(%rip), %zmm3
109 vmovups _sPC3_uisa+__svml_stan_data_internal(%rip), %zmm0
110 vmulps {rn-sae}, %zmm5, %zmm5, %zmm1
111 vpermt2ps Th_tbl_uisa+64+__svml_stan_data_internal(%rip), %zmm10, %zmm3
112 vmovups _sPC5_uisa+__svml_stan_data_internal(%rip), %zmm10
113 vfmadd231ps {rn-sae}, %zmm1, %zmm10, %zmm0
114 vmulps {rn-sae}, %zmm5, %zmm0, %zmm4
115 vfmadd213ps {rn-sae}, %zmm5, %zmm1, %zmm4
116
117/*
118 * Computer Denominator:
119 * sDenominator - sDlow ~= 1-(sTh+sTl)*(sP+sPlow)
120 */
121 vmovups _sOne+__svml_stan_data_internal(%rip), %zmm5
122 vmulps {rn-sae}, %zmm4, %zmm3, %zmm7
123
124/*
125 * Compute Numerator:
126 * sNumerator + sNlow ~= sTh+sTl+sP+sPlow
127 */
128 vaddps {rn-sae}, %zmm3, %zmm4, %zmm8
129 vsubps {rn-sae}, %zmm7, %zmm5, %zmm9
130 vsubps {rn-sae}, %zmm3, %zmm8, %zmm2
131
132/*
133 * Now computes (sNumerator + sNlow)/(sDenominator - sDlow)
134 * Choose NR iteration instead of hardware division
135 */
136 vrcp14ps %zmm9, %zmm14
137 vsubps {rn-sae}, %zmm5, %zmm9, %zmm6
138 vsubps {rn-sae}, %zmm2, %zmm4, %zmm13
139 vmulps {rn-sae}, %zmm8, %zmm14, %zmm15
140 vaddps {rn-sae}, %zmm7, %zmm6, %zmm12
141
142/* One NR iteration to refine sQuotient */
143 vfmsub213ps {rn-sae}, %zmm8, %zmm15, %zmm9
144 vfnmadd213ps {rn-sae}, %zmm9, %zmm15, %zmm12
145 vsubps {rn-sae}, %zmm13, %zmm12, %zmm0
146 vfnmadd213ps {rn-sae}, %zmm15, %zmm14, %zmm0
147 testl %edx, %edx
148
149/* Go to special inputs processing branch */
150 jne L(SPECIAL_VALUES_BRANCH)
151 # LOE rbx r12 r13 r14 r15 edx zmm0 zmm11
152
153/* Restore registers
154 * and exit the function
155 */
156
157L(EXIT):
158 movq %rbp, %rsp
159 popq %rbp
160 cfi_def_cfa(7, 8)
161 cfi_restore(6)
162 ret
163 cfi_def_cfa(6, 16)
164 cfi_offset(6, -16)
165
166/* Branch to process
167 * special inputs
168 */
169
170L(SPECIAL_VALUES_BRANCH):
171 vmovups %zmm11, 64(%rsp)
172 vmovups %zmm0, 128(%rsp)
173 # LOE rbx r12 r13 r14 r15 edx zmm0
174
175 xorl %eax, %eax
176 # LOE rbx r12 r13 r14 r15 eax edx
177
178 vzeroupper
179 movq %r12, 16(%rsp)
180 /* DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -176; DW_OP_plus) */
181 .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x50, 0xff, 0xff, 0xff, 0x22
182 movl %eax, %r12d
183 movq %r13, 8(%rsp)
184 /* DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -184; DW_OP_plus) */
185 .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x48, 0xff, 0xff, 0xff, 0x22
186 movl %edx, %r13d
187 movq %r14, (%rsp)
188 /* DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -192; DW_OP_plus) */
189 .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x40, 0xff, 0xff, 0xff, 0x22
190 # LOE rbx r15 r12d r13d
191
192/* Range mask
193 * bits check
194 */
195
196L(RANGEMASK_CHECK):
197 btl %r12d, %r13d
198
199/* Call scalar math function */
200 jc L(SCALAR_MATH_CALL)
201 # LOE rbx r15 r12d r13d
202
203/* Special inputs
204 * processing loop
205 */
206
207L(SPECIAL_VALUES_LOOP):
208 incl %r12d
209 cmpl $16, %r12d
210
211/* Check bits in range mask */
212 jl L(RANGEMASK_CHECK)
213 # LOE rbx r15 r12d r13d
214
215 movq 16(%rsp), %r12
216 cfi_restore(12)
217 movq 8(%rsp), %r13
218 cfi_restore(13)
219 movq (%rsp), %r14
220 cfi_restore(14)
221 vmovups 128(%rsp), %zmm0
222
223/* Go to exit */
224 jmp L(EXIT)
225 /* DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -176; DW_OP_plus) */
226 .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x50, 0xff, 0xff, 0xff, 0x22
227 /* DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -184; DW_OP_plus) */
228 .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x48, 0xff, 0xff, 0xff, 0x22
229 /* DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -192; DW_OP_plus) */
230 .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x40, 0xff, 0xff, 0xff, 0x22
231 # LOE rbx r12 r13 r14 r15 zmm0

233/* Scalar math fucntion call
234 * to process special input
235 */
236
237L(SCALAR_MATH_CALL):
238 movl %r12d, %r14d
239 movss 64(%rsp,%r14,4), %xmm0
240 call tanf@PLT
241 # LOE rbx r14 r15 r12d r13d xmm0
242
243 movss %xmm0, 128(%rsp,%r14,4)
244
245/* Process special inputs in loop */
246 jmp L(SPECIAL_VALUES_LOOP)
247 cfi_restore(12)
248 cfi_restore(13)
249 cfi_restore(14)
250 # LOE rbx r15 r12d r13d

252/* Auxilary branch
253 * for out of main path inputs
254 */
255
256L(AUX_BRANCH):
257 vmovups _sRangeVal+__svml_stan_data_internal(%rip), %zmm6
258
259/*
260 * Get the (2^a / 2pi) mod 1 values from the table.
261 * Because doesn't have I-type gather, we need a trivial cast
262 */
263 lea __svml_stan_reduction_data_internal(%rip), %rax
264 vmovups %zmm5, (%rsp)
265 vandps %zmm0, %zmm6, %zmm14
266 vcmpps $0, {sae}, %zmm6, %zmm14, %k0
267
268/*
269 * Break the P_xxx and m into 16-bit chunks ready for
270 * the long multiplication via 16x16->32 multiplications
271 */
272 vmovups .FLT_15(%rip), %zmm6
273 kxnorw %k0, %k0, %k1
274 kxnorw %k0, %k0, %k2
275 kxnorw %k0, %k0, %k3
276 kmovw %k0, %edx
277 vpandd .FLT_12(%rip), %zmm11, %zmm5
278 vpsrld $23, %zmm5, %zmm7
279 vpslld $1, %zmm7, %zmm8
280 vpaddd %zmm7, %zmm8, %zmm9
281 vpslld $2, %zmm9, %zmm4
282 vpxord %zmm3, %zmm3, %zmm3
283 vpxord %zmm15, %zmm15, %zmm15
284 vpxord %zmm2, %zmm2, %zmm2
285 vgatherdps (%rax,%zmm4), %zmm3{%k1}
286 vgatherdps 4(%rax,%zmm4), %zmm15{%k2}
287 vgatherdps 8(%rax,%zmm4), %zmm2{%k3}
288 vpsrld $16, %zmm3, %zmm5
289 vpsrld $16, %zmm2, %zmm13
290
291/*
292 * Also get the significand as an integer
293 * NB: adding in the integer bit is wrong for denorms!
294 * To make this work for denorms we should do something slightly different
295 */
296 vpandd .FLT_13(%rip), %zmm11, %zmm0
297 vpaddd .FLT_14(%rip), %zmm0, %zmm1
298 vpsrld $16, %zmm15, %zmm0
299 vpsrld $16, %zmm1, %zmm8
300 vpandd %zmm6, %zmm3, %zmm9
301 vpandd %zmm6, %zmm15, %zmm12
302 vpandd %zmm6, %zmm2, %zmm7
303 vpandd %zmm6, %zmm1, %zmm14
304
305/* Now do the big multiplication and carry propagation */
306 vpmulld %zmm9, %zmm8, %zmm4
307 vpmulld %zmm0, %zmm8, %zmm3
308 vpmulld %zmm12, %zmm8, %zmm2
309 vpmulld %zmm13, %zmm8, %zmm1
310 vpmulld %zmm7, %zmm8, %zmm8
311 vpmulld %zmm5, %zmm14, %zmm7
312 vpmulld %zmm9, %zmm14, %zmm5
313 vpmulld %zmm0, %zmm14, %zmm9
314 vpmulld %zmm12, %zmm14, %zmm0
315 vpmulld %zmm13, %zmm14, %zmm12
316 vpsrld $16, %zmm12, %zmm14
317 vpsrld $16, %zmm0, %zmm13
318 vpsrld $16, %zmm9, %zmm15
319 vpsrld $16, %zmm5, %zmm12
320 vpsrld $16, %zmm8, %zmm8
321 vpaddd %zmm14, %zmm1, %zmm1
322 vpaddd %zmm13, %zmm2, %zmm2
323 vpaddd %zmm15, %zmm3, %zmm15
324 vpaddd %zmm12, %zmm4, %zmm3
325 vpandd %zmm6, %zmm0, %zmm13
326 vpaddd %zmm1, %zmm13, %zmm4
327 vpaddd %zmm4, %zmm8, %zmm14
328 vpsrld $16, %zmm14, %zmm0
329 vpandd %zmm6, %zmm9, %zmm9
330 vpaddd %zmm2, %zmm9, %zmm1
331 vpaddd %zmm1, %zmm0, %zmm8
332
333/*
334 * Now round at the 2^-8 bit position for reduction mod pi/2^7
335 * instead of the original 2pi (but still with the same 2pi scaling).
336 * Use a shifter of 2^15 + 2^14.
337 * The N we get is our final version; it has an offset of
338 * 2^8 because of the implicit integer bit, and anyway for negative
339 * starting value it's a 2s complement thing. But we need to mask
340 * off the exponent part anyway so it's fine.
341 */
342 vmovups .FLT_18(%rip), %zmm1
343 vpandd %zmm6, %zmm7, %zmm7
344 vpaddd %zmm3, %zmm7, %zmm13
345 vpsrld $16, %zmm8, %zmm3
346 vpandd %zmm6, %zmm5, %zmm5
347 vpaddd %zmm15, %zmm5, %zmm2
348 vpaddd %zmm2, %zmm3, %zmm15
349 vpsrld $16, %zmm15, %zmm12
350 vpaddd %zmm13, %zmm12, %zmm5
351
352/* Assemble reduced argument from the pieces */
353 vpandd %zmm6, %zmm14, %zmm9
354 vpandd %zmm6, %zmm15, %zmm7
355 vpslld $16, %zmm5, %zmm6
356 vpslld $16, %zmm8, %zmm5
357 vpaddd %zmm7, %zmm6, %zmm4
358 vpaddd %zmm9, %zmm5, %zmm9
359 vpsrld $9, %zmm4, %zmm6
360
361/*
362 * We want to incorporate the original sign now too.
363 * Do it here for convenience in getting the right N value,
364 * though we could wait right to the end if we were prepared
365 * to modify the sign of N later too.
366 * So get the appropriate sign mask now (or sooner).
367 */
368 vpandd .FLT_16(%rip), %zmm11, %zmm0
369 vpandd .FLT_21(%rip), %zmm9, %zmm13
370 vpslld $5, %zmm13, %zmm14
371
372/*
373 * Create floating-point high part, implicitly adding integer bit 1
374 * Incorporate overall sign at this stage too.
375 */
376 vpxord .FLT_17(%rip), %zmm0, %zmm8
377 vpord %zmm8, %zmm6, %zmm2
378 vaddps {rn-sae}, %zmm2, %zmm1, %zmm12
379 vsubps {rn-sae}, %zmm1, %zmm12, %zmm3
380 vsubps {rn-sae}, %zmm3, %zmm2, %zmm7
381
382/*
383 * Create floating-point low and medium parts, respectively
384 * lo_17, ... lo_0, 0, ..., 0
385 * hi_8, ... hi_0, lo_31, ..., lo_18
386 * then subtract off the implicitly added integer bits,
387 * 2^-46 and 2^-23, respectively.
388 * Put the original sign into all of them at this stage.
389 */
390 vpxord .FLT_20(%rip), %zmm0, %zmm6
391 vpord %zmm6, %zmm14, %zmm15
392 vpandd .FLT_23(%rip), %zmm4, %zmm4
393 vsubps {rn-sae}, %zmm6, %zmm15, %zmm8
394 vandps .FLT_26(%rip), %zmm11, %zmm15
395 vpsrld $18, %zmm9, %zmm6
396
397/*
398 * If the magnitude of the input is <= 2^-20, then
399 * just pass through the input, since no reduction will be needed and
400 * the main path will only work accurately if the reduced argument is
401 * about >= 2^-40 (which it is for all large pi multiples)
402 */
403 vmovups .FLT_27(%rip), %zmm14
404 vcmpps $26, {sae}, %zmm14, %zmm15, %k4
405 vcmpps $22, {sae}, %zmm14, %zmm15, %k5
406 vpxord .FLT_22(%rip), %zmm0, %zmm1
407 vpslld $14, %zmm4, %zmm0
408 vpord %zmm6, %zmm0, %zmm0
409 vpord %zmm1, %zmm0, %zmm4
410 vsubps {rn-sae}, %zmm1, %zmm4, %zmm2
411 vpternlogd $255, %zmm6, %zmm6, %zmm6
412
413/* Now add them up into 2 reasonably aligned pieces */
414 vaddps {rn-sae}, %zmm2, %zmm7, %zmm13
415 vsubps {rn-sae}, %zmm13, %zmm7, %zmm7
416 vaddps {rn-sae}, %zmm7, %zmm2, %zmm3
417
418/*
419 * The output is _VRES_R (high) + _VRES_E (low), and the integer part is _VRES_IND
420 * Set sRp2 = _VRES_R^2 and then resume the original code.
421 */
422 vmovups .FLT_28(%rip), %zmm2
423 vaddps {rn-sae}, %zmm8, %zmm3, %zmm1
424 vmovups .FLT_25(%rip), %zmm8
425
426/* Grab our final N value as an integer, appropriately masked mod 2^8 */
427 vpandd .FLT_19(%rip), %zmm12, %zmm5
428
429/*
430 * Now multiply those numbers all by 2 pi, reasonably accurately.
431 * (RHi + RLo) * (pi_lead + pi_trail) ~=
432 * RHi * pi_lead + (RHi * pi_trail + RLo * pi_lead)
433 */
434 vmovups .FLT_24(%rip), %zmm12
435 vmulps {rn-sae}, %zmm12, %zmm13, %zmm0
436 vmovaps %zmm12, %zmm9
437 vfmsub213ps {rn-sae}, %zmm0, %zmm13, %zmm9
438 vfmadd213ps {rn-sae}, %zmm9, %zmm8, %zmm13
439 vmovaps %zmm6, %zmm8
440 vfmadd213ps {rn-sae}, %zmm13, %zmm12, %zmm1
441 vpandnd %zmm15, %zmm15, %zmm8{%k4}
442 vpandnd %zmm15, %zmm15, %zmm6{%k5}
443 vandps %zmm11, %zmm6, %zmm14
444 vandps %zmm0, %zmm8, %zmm15
445 vandps %zmm1, %zmm8, %zmm12
446 vorps %zmm15, %zmm14, %zmm6
447 vpsrld $31, %zmm6, %zmm3
448 vpsubd %zmm3, %zmm2, %zmm4
449 vpaddd %zmm4, %zmm5, %zmm7
450 vpsrld $2, %zmm7, %zmm13
451 vpslld $2, %zmm13, %zmm9
452
453/*
454 *
455 * End of large arguments path
456 *
457 * Merge results from main and large paths:
458 */
459 vblendmps %zmm13, %zmm10, %zmm10{%k6}
460 vpsubd %zmm9, %zmm5, %zmm5
461 vmovups .FLT_29(%rip), %zmm9
462 vcvtdq2ps {rn-sae}, %zmm5, %zmm0
463 vmovups .FLT_30(%rip), %zmm5
464 vfmadd231ps {rn-sae}, %zmm0, %zmm5, %zmm12
465 vmovups (%rsp), %zmm5
466 vaddps {rn-sae}, %zmm6, %zmm12, %zmm6
467 vfmadd213ps {rn-sae}, %zmm6, %zmm9, %zmm0
468 vblendmps %zmm0, %zmm5, %zmm5{%k6}
469
470/* Return to main vector processing path */
471 jmp L(AUX_BRANCH_RETURN)
472 # LOE rbx r12 r13 r14 r15 edx zmm5 zmm10 zmm11
473END(_ZGVeN16v_tanf_skx)
