7/* the following events that user-space can register for */
8#define FAN_ACCESS 0x00000001 /* File was accessed */
9#define FAN_MODIFY 0x00000002 /* File was modified */
10#define FAN_ATTRIB 0x00000004 /* Metadata changed */
11#define FAN_CLOSE_WRITE 0x00000008 /* Writtable file closed */
12#define FAN_CLOSE_NOWRITE 0x00000010 /* Unwrittable file closed */
13#define FAN_OPEN 0x00000020 /* File was opened */
14#define FAN_MOVED_FROM 0x00000040 /* File was moved from X */
15#define FAN_MOVED_TO 0x00000080 /* File was moved to Y */
16#define FAN_CREATE 0x00000100 /* Subfile was created */
17#define FAN_DELETE 0x00000200 /* Subfile was deleted */
18#define FAN_DELETE_SELF 0x00000400 /* Self was deleted */
19#define FAN_MOVE_SELF 0x00000800 /* Self was moved */
20#define FAN_OPEN_EXEC 0x00001000 /* File was opened for exec */
21
22#define FAN_Q_OVERFLOW 0x00004000 /* Event queued overflowed */
23
24#define FAN_OPEN_PERM 0x00010000 /* File open in perm check */
25#define FAN_ACCESS_PERM 0x00020000 /* File accessed in perm check */
26#define FAN_OPEN_EXEC_PERM 0x00040000 /* File open/exec in perm check */
27
28#define FAN_EVENT_ON_CHILD 0x08000000 /* Interested in child events */
29
30#define FAN_ONDIR 0x40000000 /* Event occurred against dir */
31
32/* helper events */
33#define FAN_CLOSE (FAN_CLOSE_WRITE | FAN_CLOSE_NOWRITE) /* close */
34#define FAN_MOVE (FAN_MOVED_FROM | FAN_MOVED_TO) /* moves */

36/* flags used for fanotify_init() */
37#define FAN_CLOEXEC 0x00000001
38#define FAN_NONBLOCK 0x00000002

40/* These are NOT bitwise flags. Both bits are used together. */
41#define FAN_CLASS_NOTIF 0x00000000
42#define FAN_CLASS_CONTENT 0x00000004
43#define FAN_CLASS_PRE_CONTENT 0x00000008
44
45/* Deprecated - do not use this in programs and do not add new flags here! */
46#define FAN_ALL_CLASS_BITS (FAN_CLASS_NOTIF | FAN_CLASS_CONTENT | \
47 FAN_CLASS_PRE_CONTENT)
48
49#define FAN_UNLIMITED_QUEUE 0x00000010
50#define FAN_UNLIMITED_MARKS 0x00000020
51#define FAN_ENABLE_AUDIT 0x00000040
52
53/* Flags to determine fanotify event format */
54#define FAN_REPORT_PIDFD 0x00000080 /* Report pidfd for event->pid */
55#define FAN_REPORT_TID 0x00000100 /* event->pid is thread id */
56#define FAN_REPORT_FID 0x00000200 /* Report unique file id */
57#define FAN_REPORT_DIR_FID 0x00000400 /* Report unique directory id */
58#define FAN_REPORT_NAME 0x00000800 /* Report events with name */
59
60/* Convenience macro - FAN_REPORT_NAME requires FAN_REPORT_DIR_FID */
61#define FAN_REPORT_DFID_NAME (FAN_REPORT_DIR_FID | FAN_REPORT_NAME)
62
63/* Deprecated - do not use this in programs and do not add new flags here! */
64#define FAN_ALL_INIT_FLAGS (FAN_CLOEXEC | FAN_NONBLOCK | \
65 FAN_ALL_CLASS_BITS | FAN_UNLIMITED_QUEUE |\
66 FAN_UNLIMITED_MARKS)
67
68/* flags used for fanotify_modify_mark() */
69#define FAN_MARK_ADD 0x00000001
70#define FAN_MARK_REMOVE 0x00000002
71#define FAN_MARK_DONT_FOLLOW 0x00000004
72#define FAN_MARK_ONLYDIR 0x00000008
73/* FAN_MARK_MOUNT is 0x00000010 */
74#define FAN_MARK_IGNORED_MASK 0x00000020
75#define FAN_MARK_IGNORED_SURV_MODIFY 0x00000040
76#define FAN_MARK_FLUSH 0x00000080
77/* FAN_MARK_FILESYSTEM is 0x00000100 */
78
79/* These are NOT bitwise flags. Both bits can be used togther. */
80#define FAN_MARK_INODE 0x00000000
81#define FAN_MARK_MOUNT 0x00000010
82#define FAN_MARK_FILESYSTEM 0x00000100
83

112#define FANOTIFY_METADATA_VERSION 3

114struct fanotify_event_metadata {
115 __u32 event_len;
116 __u8 vers;
117 __u8 reserved;
118 __u16 metadata_len;
119 __aligned_u64 mask;
120 __s32 fd;
121 __s32 pid;
122};
123
124#define FAN_EVENT_INFO_TYPE_FID 1
125#define FAN_EVENT_INFO_TYPE_DFID_NAME 2
126#define FAN_EVENT_INFO_TYPE_DFID 3
127#define FAN_EVENT_INFO_TYPE_PIDFD 4
128
129/* Variable length info record following event metadata */
130struct fanotify_event_info_header {
131 __u8 info_type;
132 __u8 pad;
133 __u16 len;
134};
135
136/*
137 * Unique file identifier info record.
138 * This structure is used for records of types FAN_EVENT_INFO_TYPE_FID,
139 * FAN_EVENT_INFO_TYPE_DFID and FAN_EVENT_INFO_TYPE_DFID_NAME.
140 * For FAN_EVENT_INFO_TYPE_DFID_NAME there is additionally a null terminated
141 * name immediately after the file handle.
142 */
143struct fanotify_event_info_fid {
144 struct fanotify_event_info_header hdr;
145 __kernel_fsid_t fsid;
146 /*
147 * Following is an opaque struct file_handle that can be passed as
148 * an argument to open_by_handle_at(2).
149 */
150 unsigned char handle[0];
151};
152
153/*
154 * This structure is used for info records of type FAN_EVENT_INFO_TYPE_PIDFD.
155 * It holds a pidfd for the pid that was responsible for generating an event.
156 */
157struct fanotify_event_info_pidfd {
158 struct fanotify_event_info_header hdr;
159 __s32 pidfd;
160};
161
162struct fanotify_response {
163 __s32 fd;
164 __u32 response;
165};
166
167/* Legit userspace responses to a _PERM event */
168#define FAN_ALLOW 0x01
169#define FAN_DENY 0x02
170#define FAN_AUDIT 0x10 /* Bit mask to create audit record for result */
171
172/* No fd set in event */
173#define FAN_NOFD -1
174#define FAN_NOPIDFD FAN_NOFD
175#define FAN_EPIDFD -2
176
177/* Helper functions to deal with fanotify_event_metadata buffers */
178#define FAN_EVENT_METADATA_LEN (sizeof(struct fanotify_event_metadata))
179
180#define FAN_EVENT_NEXT(meta, len) ((len) -= (meta)->event_len, \
181 (struct fanotify_event_metadata*)(((char *)(meta)) + \
182 (meta)->event_len))
183
184#define FAN_EVENT_OK(meta, len) ((long)(len) >= (long)FAN_EVENT_METADATA_LEN && \
185 (long)(meta)->event_len >= (long)FAN_EVENT_METADATA_LEN && \
186 (long)(meta)->event_len <= (long)(len))
